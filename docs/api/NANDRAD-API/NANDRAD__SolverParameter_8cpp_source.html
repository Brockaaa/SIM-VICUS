67 bool SolverParameter::operator!=(const SolverParameter & other) const {
68  if (m_integrator != other.m_integrator) return true;
69  if (m_lesSolver != other.m_lesSolver) return true;
70  if (m_preconditioner != other.m_preconditioner) return true;
71 
72  for (unsigned int i=0; i<NUM_P; ++i)
73  if (m_para[i] != other.m_para[i])
74  return true;
75 
76  for (unsigned int i=0; i<NUM_F; ++i)
77  if (m_flag[i] != other.m_flag[i])
78  return true;
79 
80  return false; // this and other hold the same data
81 }
