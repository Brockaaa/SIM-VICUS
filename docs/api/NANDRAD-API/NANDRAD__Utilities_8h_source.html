78 template <typename T>
79 void readVector(const TiXmlElement * element, const std::string & name, std::vector<T> & vec) {
80  FUNCID(NANDRAD::readVector);
81  std::string text = element->GetText();
82  IBK::replace_string(text, ",", " ");
83  try {
84  IBK::string2valueVector(text, vec);
85  } catch (IBK::Exception & ex) {
86  throw IBK::Exception( ex, IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
87  IBK::FormatString("Error reading vector element '%1'.").arg(name) ), FUNC_ID);
88  }
89 }

92 template <typename T>
93 void writeVector(TiXmlElement * parent, const std::string & name, const std::vector<T> & vec) {
94  if (!vec.empty()) {
95  TiXmlElement * child = new TiXmlElement(name);
96  parent->LinkEndChild(child);
97 
98  std::stringstream vals;
99  for (unsigned int i=0; i<vec.size(); ++i) {
100  vals << vec[i];
101  if (i<vec.size()-1) vals << ",";
102  }
103  TiXmlText * text = new TiXmlText( vals.str() );
104  child->LinkEndChild( text );
105  }
106 }
