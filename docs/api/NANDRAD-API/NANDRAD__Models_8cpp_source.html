29 void Models::readXML(const TiXmlElement * element) {
30  FUNCID(Models::readXML);
31 
32  // loop over all elements in this XML element
33  for (const TiXmlElement * e=element->FirstChildElement(); e; e = e->NextSiblingElement()) {
34 
35  // get element name
36  std::string name = e->Value();
37  // handle known elements
38 
39  if (name == "NaturalVentilationModel") {
40  NaturalVentilationModel model;
41  try {
42  model.readXML(e);
43  } catch (IBK::Exception & ex) {
44  throw IBK::Exception(ex, "Error reading NaturalVentilationModel in Models tag.", FUNC_ID);
45  }
46  m_naturalVentilationModels.push_back(model);
47  }
48  else {
49  IBK::IBK_Message(IBK::FormatString(
50  "Unknown element '%1' in Models section.").arg(name), IBK::MSG_WARNING);
51  }
52 
53  }
54 }

58 TiXmlElement * Models::writeXML(TiXmlElement * parent) const {
59 
60  // write nothing if all containers are empty
61  if (m_naturalVentilationModels.empty()) return nullptr;
62 
63  TiXmlComment::addComment(parent, "Model parameterization blocks");
64 
65  TiXmlElement * e1 = new TiXmlElement( "Models" );
66  parent->LinkEndChild( e1 );
67 
68  // now write all models as they are defined
69  for (auto m : m_naturalVentilationModels)
70  m.writeXML(e1);
71 
72  TiXmlComment::addSeparatorComment(parent);
73  return e1;
74 }
