33 void ObjectList::readXML(const TiXmlElement * element) {
34  FUNCID(ObjectList::readXML);
35 
36  // read base stuff
37  readXMLPrivate(element);
38 
39  try {
40  // read filter ID
41  for (const TiXmlElement * c = element->FirstChildElement(); c; c = c->NextSiblingElement()) {
42  // determine data based on element name
43  std::string cname = c->Value();
44  // location specification is found
45  if (cname == "FilterID") {
46  m_filterID.setEncodedString(c->GetText());
47  }
48  else if (cname == "ReferenceType") {
49  if (!KeywordList::KeywordExists("ModelInputReference::referenceType_t", c->GetText()) ) {
50  throw IBK::Exception( IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
51  IBK::FormatString("Invalid or unknown keyword '"+std::string(c->GetText())+"'.") ), FUNC_ID);
52  }
53  m_referenceType = (ModelInputReference::referenceType_t)
54  KeywordList::Enumeration("ModelInputReference::referenceType_t", c->GetText());
55  }
56  }
57  // check if we have found a type filter
58  if (m_referenceType == ModelInputReference::NUM_MRT) {
59  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
60  IBK::FormatString("Missing tag 'ReferenceType'.") ), FUNC_ID);
61  }
62  // need either ID filter or name filter
63  // exception: Filter type location and filter type schedule need no id filter
64  if (m_referenceType != ModelInputReference::MRT_LOCATION && m_referenceType != ModelInputReference::MRT_SCHEDULE) {
65  if (m_filterID.empty()) {
66  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
67  IBK::FormatString("'FilterID' tag is required.") ), FUNC_ID);
68  }
69  }
70  }
71  catch (IBK::Exception &ex) {
72  throw IBK::Exception(ex, IBK::FormatString("Error reading 'ObjectList' element."), FUNC_ID);
73  }
74  catch (std::exception &ex2) {
75  throw IBK::Exception(IBK::FormatString("%1\nError reading 'ObjectList' element.").arg(ex2.what()), FUNC_ID);
76  }
77 }

80 TiXmlElement * ObjectList::writeXML(TiXmlElement * parent) const {
81 
82  // first basic code
83  TiXmlElement * e = writeXMLPrivate(parent);
84 
85  if (!m_filterID.empty())
86  TiXmlElement::appendSingleAttributeElement(e, "FilterID", nullptr, std::string(), m_filterID.encodedString());
87 
88  TiXmlElement::appendSingleAttributeElement( e, "ReferenceType", nullptr, std::string(),
89  KeywordList::Keyword("ModelInputReference::referenceType_t", m_referenceType));
90 
91  return e;
92 }
