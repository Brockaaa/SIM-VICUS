55 bool SimulationParameter::operator!=(const SimulationParameter & other) const {
56  for (unsigned int i=0; i<NUM_P; ++i) {
57  if (m_para[i] != other.m_para[i])
58  return true;
59  }
60  if (m_interval != other.m_interval) return true;
61 
62  for (unsigned int i=0; i<NUM_F; ++i)
63  if (m_flags[i] != other.m_flags[i]) return true;
64  for (unsigned int i=0; i<NUM_P; ++i)
65  if (m_para[i] != other.m_para[i]) return true;
66  for (unsigned int i=0; i<NUM_IP; ++i)
67  if (m_intPara[i] != other.m_intPara[i]) return true;
68 
69  return false; // this and other hold the same data
70 }
