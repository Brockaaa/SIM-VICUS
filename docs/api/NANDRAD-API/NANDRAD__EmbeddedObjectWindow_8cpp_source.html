147 void EmbeddedObjectWindow::writeXML(TiXmlElement * parent) const {
148  TiXmlElement * e = new TiXmlElement("Window");
149  parent->LinkEndChild(e);
150 
151  e->SetAttribute( "model", KeywordList::Keyword("EmbeddedObjectWindow::modelType_t", m_modelType) );
152 
153  // write EmbeddedObjectWindow parameters
154  for (unsigned int i=0; i<NUM_P; ++i) {
155  if(m_para[i].name.empty()) continue;
156  if(detailedOutput)
157  TiXmlComment::addComment(e,KeywordList::Description("EmbeddedObjectWindow::para_t",i));
158  TiXmlElement::appendIBKParameterElement(e,
159  m_para[i].name,
160  m_para[i].IO_unit.name(),
161  m_para[i].get_value());
162  }
163 
164  // write all generic parameters
165  writeGenericParameters(e);
166 }
