34 /*! Class LinearSplineParameter stores a linear spline curve, the corresponding parameter name
35  and a unit name of both dependend and independend quantity (xUnit, yUnit).
36 
37  Note that the read and write functions perform unit conversions on the values, so that
38  the linear spline actually holds values always in Base-SI units (according to the
39  IBK Unit-Definition).
40 
41  So when reading a linear spline that defines temperatures in C, the actual spline will
42  then contain the values in K.
43 */
44 class LinearSplineParameter {
45 public:
46 
47  /*! Interpolation method to be used for this linear spline parameter. */
48  enum interpolationMethod_t {
49  I_CONSTANT, // Keyword: constant
50  I_LINEAR, // Keyword: linear
51  NUM_I
52  };
53 
54  /*! How to treat the values in multi-year simulations. */
55  enum wrapMethod_t {
56  C_CONTINUOUS, // Keyword: continuous 'Continuous data'
57  C_CYCLIC, // Keyword: cyclic 'Annual cycle'
58  NUM_C
59  };
60 
61  // *** PUBLIC MEMBER FUNCTIONS ***
62 
63  NANDRAD_READWRITE
64  NANDRAD_COMP(LinearSplineParameter)
65 
66  // *** PUBLIC MEMBER VARIABLES ***
67 
68  /*! Parameter name (in context of schedules used as scheduled quantity). */
69  std::string m_name; // XML:A
70  /*! Interpolation method to be used when computing values of this spline. */
71  interpolationMethod_t m_interpolationMethod = I_LINEAR; // XML:A
72  /*! Whether to wrap time around in multi-year simulations (cyclic use) or to assume continuous data. */
73  wrapMethod_t m_wrapMethod = C_CYCLIC; // XML:A
74  /*! Data vectors including linear spline functionality (i.e. interpolation at any given value).
75  Values are stored in the respective Base-SI units of the input/output units m_xUnit and
76  m_yUnit. For example, if m_yUnit is 'C' (degree C), then the spline holds values in
77  'K' (degree Kelvin). The functions readXML() and writeXML() perform the necessary conversions.
78  */
79  IBK::LinearSpline m_values; // XML:E
80  /*! Unit of the x-values. */
81  IBK::Unit m_xUnit; // XML:A
82  /*! Unit of the y-values. */
83  IBK::Unit m_yUnit; // XML:A
84 };
