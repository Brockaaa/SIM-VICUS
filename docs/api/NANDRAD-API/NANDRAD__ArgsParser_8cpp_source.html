26 ArgsParser::ArgsParser() {
27  setAppName("NandradSolver");
28  addOption(0, "fmu-export",
29  "(optional) Generate FMU based on project file specification.",
30  "FMU file name",
31  "");
32  addFlag(0, "fmu-modelica-wrapper",
33  "(optional) If given, a Modelica wrapper for the exported FMU is created. Only works in combination with 'fmu-export'.");
34 
35  // adjust options for les-solver, integrator and precond to show only the options available for NANDRAD
36  for (OptionType & ot : m_knownOptions) {
37  if (ot.m_longVersion == "integrator") {
38  ot.m_description = "auto|CVode|ImplicitEuler|ExplicitEuler";
39  }
40  else if (ot.m_longVersion == "les-solver") {
41  ot.m_description = "auto|Dense|KLU|GMRES|BiCGStab";
42  }
43  else if (ot.m_longVersion == "precond") {
44  ot.m_description = "auto|ILU";
45  }
46  }
47 }

50 void ArgsParser::printHelp(std::ostream & out) const {
51  // re-use implementation of original help function
52  out << "Syntax: " << m_appname << " [options] <project file>\n\n";
53  const unsigned int TEXT_WIDTH = 79;
54  const unsigned int TAB_WIDTH = 26;
55  printFlags(out,TEXT_WIDTH,TAB_WIDTH);
56  printOptions(out,TEXT_WIDTH,TAB_WIDTH);
57  // write examples
58  out << "\nExamples:\n\n"
59  "Starting solver\n"
60  "> "<< m_appname << " <project file>\n\n"
61  "Running with 8 threads\n"
62  "> "<< m_appname << " -p=8 <project file>\n\n"
63  "Starting solver with different LES solver\n"
64  "> "<< m_appname << " --les-solver=GMRES <project file>\n\n"
65  "Starting solver with BiCGStab iterative solver, Krylov subspace limit of 50 and ILU preconditioner\n"
66  "> "<< m_appname << " --les-solver=BiCGStab(50) --precond=ILU <project file>\n\n";
67 }
