40 /*! An Interface identifies a surface of a wall and stores all data that are needed for boundary
41  condition calculation. Each ConstructionInstance contains two member variables of type
42  Interface, which uniquely define the boundary condition parameters at side
43  A or B of the construction (A - layer index 0, B - other side).
44 
45  The zone at the other side of the wall surface is identified by its zone ID.
46  The zone ID = 0 indicates ambient climate. If the construction is adiabatic,
47  there must not be an interface defined for this side.
48 */
49 class Interface {
50 public:
51 
52  // *** PUBLIC MEMBER FUNCTIONS ***
53 
54  NANDRAD_READWRITE
55  NANDRAD_COMPARE_WITH_ID
56 
57  /*! Special form of comparison operator, tests if parameters that have an
58  impact on result calculation are the same (zone, location, physical parameters).
59  */
60  bool behavesLike(const Interface & other) const {
61  return (m_zoneId == other.m_zoneId &&
62  m_heatConduction == other.m_heatConduction &&
63  m_solarAbsorption == other.m_solarAbsorption &&
64  m_longWaveEmission == other.m_longWaveEmission &&
65  m_vaporDiffusion == other.m_vaporDiffusion);
66  }
67 
68  /*! Updates the comment regarding the zone this interface is connected to. */
69  void updateComment(const std::vector<Zone> & zones);
70 
71  /*! Returns true, if any of the parameters blocks have boundary conditions defined.
72  If the model ID m_id is INVALID_ID, the function returns false (the default).
73  A boundary condition parameter block counts as defined, if the modelType is not NUM_MT.
74  */
75  bool haveBCParameters() const;
76 
77  // *** PUBLIC MEMBER VARIABLES ***
78 
79  /*! ID of the referenced surface/interface. */
80  unsigned int m_id = INVALID_ID; // XML:A:required
81  /*! The id number of the neighboring zone. */
82  unsigned int m_zoneId = 0; // XML:A
83 
84  // Boundary condition parameters
85  /*! Model for convective heat transfer. */
86  InterfaceHeatConduction m_heatConduction; // XML:E
87  /*! Model for solar absorption. */
88  InterfaceSolarAbsorption m_solarAbsorption; // XML:E
89  /*! Model for long wave emissivity. */
90  InterfaceLongWaveEmission m_longWaveEmission; // XML:E
91  /*! Model for vapor diffusion (requires enabled moisture balance). */
92  InterfaceVaporDiffusion m_vaporDiffusion; // XML:E
93  /*! Model for air flow calculation. */
