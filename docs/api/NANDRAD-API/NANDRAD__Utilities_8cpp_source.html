38 TiXmlElement * openXMLFile(const std::map<std::string,IBK::Path> & pathPlaceHolders, const IBK::Path & filename,
39  const std::string & parentXmlTag, TiXmlDocument & doc)
40 {
41  FUNCID(NANDRAD::openXMLFile);
42  // replace path placeholders
43  IBK::Path fname = filename.withReplacedPlaceholders( pathPlaceHolders );
44 
45  if ( !fname.isFile() )
46  throw IBK::Exception(IBK::FormatString("File '%1' does not exist or cannot be opened for reading.")
47  .arg(fname), FUNC_ID);
48 
49  if (!doc.LoadFile(fname.str().c_str(), TIXML_ENCODING_UTF8)) {
50  throw IBK::Exception(IBK::FormatString("Error in line %1 of project file '%2':\n%3")
51  .arg(doc.ErrorRow())
52  .arg(filename)
53  .arg(doc.ErrorDesc()), FUNC_ID);
54  }
55 
56  // we use a handle so that NULL pointer checks are done during the query functions
57  TiXmlHandle xmlHandleDoc(&doc);
58 
59  // read root element
60  TiXmlElement * xmlElem = xmlHandleDoc.FirstChildElement().Element();
61  if (!xmlElem)
62  return nullptr; // empty file?
63  std::string rootnode = xmlElem->Value();
64  if (rootnode != parentXmlTag)
65  throw IBK::Exception( IBK::FormatString("Expected '%1' as root node in XML file.").arg(parentXmlTag), FUNC_ID);
66 
67  return xmlElem;
68 }

71 void readLinearSplineElement(const TiXmlElement * element,
72  IBK::LinearSpline & spl, std::string & name, IBK::Unit * xunit, IBK::Unit * yunit)
73 {
74  FUNCID(NANDRAD::readLinearSplineElement);
75  std::string xunitstr, yunitstr, interpolationMethod;
76  std::vector<double> x,y;
77  try {
78  TiXmlElement::readIBKLinearSplineElement(element, name, interpolationMethod, xunitstr, x, yunitstr, y);
79  }
80  catch (std::runtime_error & ex) {
81  throw IBK::Exception( ex, IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
82  IBK::FormatString("Error reading 'IBK:LinearSpline' tag.") ), FUNC_ID);
83  }
84  try {
85  if (xunit != nullptr)
86  *xunit = IBK::Unit(xunitstr);
87  if (yunit != nullptr)
88  *yunit = IBK::Unit(yunitstr);
89  spl.setValues(x,y);
90  }
91  catch (IBK::Exception & ex) {
92  throw IBK::Exception(ex, IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
93  IBK::FormatString("Error reading 'IBK:LinearSpline' tag.") ), FUNC_ID);
94  }
95 }

103 void readParameterElement(const TiXmlElement * element, IBK::Parameter & p) {
104  FUNCID(NANDRAD::readParameterElement);
105  const TiXmlAttribute* attrib = TiXmlAttribute::attributeByName(element, "name");
106  if (attrib == nullptr)
107  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg("Missing 'name' attribute in IBK:Parameter element."), FUNC_ID);
108  const std::string & namestr = attrib->ValueStr();
109  attrib = TiXmlAttribute::attributeByName(element, "unit");
110  if (attrib == nullptr)
111  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg("Missing 'unit' attribute in IBK:Parameter element."), FUNC_ID);
112  const std::string & unit = attrib->ValueStr();
113  IBK::Unit u;
114  try {
115  u.set(unit);
116  } catch (...) {
117  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg("Unknown/invalid unit '"+unit+"' in IBK:Parameter element."), FUNC_ID);
118  }
119  const char * const str = element->GetText();
120  try {
121  double val = IBK::string2val<double>(str);
122  p.set(namestr, val, u);
123  } catch (...) {
124  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg("Cannot read value in IBK:Parameter element."), FUNC_ID);
125  }
126 }

145 void readFlagElement(const TiXmlElement * element, IBK::Flag & f) {
146  FUNCID(NANDRAD::readFlagElement);
147  std::string namestr;
148  std::string valueStr;
149  TiXmlElement::readSingleAttributeElement(element, "name", namestr, valueStr);
150  if (valueStr == "true" || valueStr == "1")
151  f.set(namestr, true);
152  else if (valueStr == "false" || valueStr == "0")
153  f.set(namestr, false);
154  else {
155  throw IBK::Exception( IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
156  IBK::FormatString("Error reading 'IBK:Flag' tag, expected 'true' or 'false' as value.") ), FUNC_ID);
157  }
158 
159 }
