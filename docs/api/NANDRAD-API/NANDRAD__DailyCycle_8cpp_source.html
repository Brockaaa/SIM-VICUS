30 void DailyCycle::prepareCalculation() {
31  FUNCID(DailyCycle::prepareCalculation);
32 
33  // empty daily cycle? throw exception
34  if (m_values.m_values.empty())
35  throw IBK::Exception("Missing DataTable values in DailyCycle.", FUNC_ID);
36 
37  // empty time points? throw exception
38  if (m_timePoints.empty())
39  throw IBK::Exception("Missing time points values in DailyCycle.", FUNC_ID);
40 
41  if (m_interpolation == NUM_IT)
42  m_interpolation = IT_LINEAR;
43  else if (m_interpolation != IT_CONSTANT)
44  throw IBK::Exception("Invalid/undefined interpolation type in DailyCycle.", FUNC_ID);
45 
46  // check if time points follow the rules
47  if (m_timePoints[0] != 0.0)
48  throw IBK::Exception("Invalid time points in DailyCycle (must start with 0).", FUNC_ID);
49  if (m_interpolation == IT_LINEAR && m_timePoints.back() != 24.0)
50  throw IBK::Exception("Invalid time points in DailyCycle with linear interpolation (must have at least two time points and last time point must be 24 h).", FUNC_ID);
51 
52  // check for monitonically increasing time points
53  double last = m_timePoints[0];
54  for (unsigned int i=1; i<m_timePoints.size(); ++i) {
55  if (last >= m_timePoints[i])
56  throw IBK::Exception("Invalid time points in DailyCycle : time points must follow in strictly monotonic order.", FUNC_ID);
57  last = m_timePoints[i];
58  }
59 
60 
61  // first check correct number of values
62  unsigned int valueCount = m_values.m_values.begin()->second.size();
63  if (valueCount != m_timePoints.size())
64  throw IBK::Exception(IBK::FormatString("Mismatching number of values (=%1) compared to time points (=%2) in DailyCycle.")
65  .arg(valueCount).arg(m_timePoints.size()), FUNC_ID);
66 
67 
68  // This map holds the units matching the values in m_values.
69 
70  // process column headers - extract units and variable names
71  for (std::map<std::string, std::vector<double> >::const_iterator valIT = m_values.m_values.begin();
72  valIT != m_values.m_values.end(); ++valIT)
73  {
74  // extract unit from parenthesis
75  std::string::size_type p1 = valIT->first.rfind("[");
76  std::string::size_type p2 = valIT->first.rfind("]");
77  if (p1 != std::string::npos || p2 != std::string::npos) {
78  if (p1 != std::string::npos && p2 != std::string::npos && p2 > p1) {
79  std::string varName = valIT->first.substr(0, p1);
80  IBK::trim(varName);
81  std::string unitName = valIT->first.substr(p1+1, p2-p1-1);
82  IBK::trim(unitName);
83  if (!varName.empty() && !unitName.empty()) {
84  // now convert unit
85  try {
86  IBK::Unit u(unitName);
87  // register as variable, if it does not yet exist
88  if (std::find(m_valueData.begin(), m_valueData.end(), varName) != m_valueData.end())
89  throw IBK::Exception(IBK::FormatString("Duplicate variable name '%1' "
90  "in DataTable of daily cycle.").arg(varName), FUNC_ID);
91  // store variable
92  m_valueData.push_back( valueData_t(varName, u, &valIT->second) ); // persistent pointer is guaranteed here
93  } catch (...) {
94  throw IBK::Exception(IBK::FormatString("Invalid/unrecognized unit '%1' "
95  "in scheduled variable caption.").arg(unitName), FUNC_ID);
96  }
97  continue; // next column
98  }
99  }
100  throw IBK::Exception(IBK::FormatString("Invalid format of variable caption '%1', "
101  "expected format 'varname [unit]'.").arg(valIT->first), FUNC_ID);
102  }
103  // no unit - assume unitless parameter
104  m_valueData.push_back( valueData_t(valIT->first, IBK::Unit("-"), &valIT->second) ); // persistent pointer is guaranteed here
105  }
106 }

109 bool DailyCycle::operator!=(const DailyCycle & other) const {
110  // Note: we do not compare solver runtime parameters here!
111  if (m_interpolation != other.m_interpolation) return true;
112  if (m_timePoints != other.m_timePoints) return true;
113  if (m_values != other.m_values) return true;
114  return false;
