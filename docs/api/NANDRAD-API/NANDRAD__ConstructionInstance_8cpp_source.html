38 void ConstructionInstance::checkParameters(const std::vector<ConstructionType> & conTypes) {
39  FUNCID(ConstructionInstance::checkParameters);
40 
41  // check and resolve construction type reference
42  std::vector<ConstructionType>::const_iterator it = std::find(conTypes.begin(), conTypes.end(), m_constructionTypeId);
43  if (it == conTypes.end())
44  throw IBK::Exception( IBK::FormatString("Invalid/unknown construction type ID %1.").arg(m_constructionTypeId), FUNC_ID);
45  m_constructionType = &(*it); // store pointer
46 
47  // check parameters
48  if (m_para[P_AREA].name.empty())
49  throw IBK::Exception( "Missing parameter 'Area'.", FUNC_ID);
50  if (m_para[P_AREA].get_value("m2") <= 0) // get_value ensures unit conversion
51  throw IBK::Exception( "Invalid value for parameter 'Area'.", FUNC_ID);
52 
53  // Note: parameters orientation and inclination are only needed when an outdoor interface with solar radiation
54  // model is defined, so first look for such an interface.
55 
56  bool haveRadiationBC = false;
57  if (m_interfaceA.m_zoneId == 0 && m_interfaceA.m_solarAbsorption.m_modelType != InterfaceSolarAbsorption::NUM_MT)
58  haveRadiationBC = true;
59  if (m_interfaceB.m_zoneId == 0 && m_interfaceB.m_solarAbsorption.m_modelType != InterfaceSolarAbsorption::NUM_MT)
60  haveRadiationBC = true;
61 
62  if (haveRadiationBC) {
63  // we have solar radiation to outside - and we need orientation and inclination for that
64  // check parameters
65  if (m_para[P_ORIENTATION].name.empty())
66  throw IBK::Exception( "Missing parameter 'Orientation'.", FUNC_ID);
67  double orientationInDeg = m_para[P_ORIENTATION].get_value("Deg");
68  if (orientationInDeg < 0 || orientationInDeg > 360)
69  throw IBK::Exception( "Parameter 'Orientation' outside allowed value range [0,360] Deg.", FUNC_ID);
70 
71  if (m_para[P_INCLINATION].name.empty())
72  throw IBK::Exception( "Missing parameter 'Inclination'.", FUNC_ID);
73  double inclinationInDeg = m_para[P_INCLINATION].get_value("Deg");
74  if (inclinationInDeg < 0 || inclinationInDeg > 180)
75  throw IBK::Exception( "Parameter 'Inclination' outside allowed value range [0,180] Deg.", FUNC_ID);
76  }
77 }

79 bool ConstructionInstance::behavesLike(const ConstructionInstance & other) const {
80  if (m_constructionTypeId != other.m_constructionTypeId)
81  return false;
82 
83  // now compare interfaces
84  if (!m_interfaceA.behavesLike( other.m_interfaceA) )
85  return false;
86 
87  if (!m_interfaceB.behavesLike( other.m_interfaceB) )
88  return false;
89 
90  // compare parameters
91  if (m_para[P_AREA] != other.m_para[P_AREA])
92  return false;
93  if (m_para[P_ORIENTATION] != other.m_para[P_ORIENTATION])
94  return false;
95  if (m_para[P_INCLINATION] != other.m_para[P_INCLINATION])
96  return false;
97 
98  return true; // both construction instances would calculate effectively the same
99 }
