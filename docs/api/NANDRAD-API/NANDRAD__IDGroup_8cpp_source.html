33 void IDGroup::setEncodedString(const std::string & encodedString) {
34  FUNCID(IDGroup::setEncodedString);
35  // clear all entries
36  m_allIDs = false;
37  m_ids.clear();
38  m_idIntervals.clear();
39  // No data? return
40  if (encodedString.empty())
41  return;
42 
43  try {
44  // a model interval is asked for output
45  std::list<std::string> idTokens;
46  IBK::explode(encodedString,idTokens,',');
47  // for each token check for special cases and occurancy of - sign
48  for (std::list<std::string>::const_iterator it = idTokens.begin(); it != idTokens.end(); ++it) {
49  std::string tokenTrimmed = *it;
50  IBK::trim(tokenTrimmed);
51 
52  if (tokenTrimmed.empty()) {
53  throw IBK::Exception( IBK::FormatString("Invalid encoding format."), FUNC_ID);
54  }
55  if (tokenTrimmed == "*") {
56  // check if set or vector is not empty
57  if (!m_idIntervals.empty() || !m_ids.empty())
58  throw IBK::Exception(IBK::FormatString("String '*' can only be used stand-alone."), FUNC_ID);
59  m_allIDs = true;
60  }
61  else {
62  // check if m_allIDs is still false
63  if (m_allIDs)
64  throw IBK::Exception( IBK::FormatString("String '*' can only be used stand-alone."), FUNC_ID);
65 
66  if (tokenTrimmed.find("-")!= std::string::npos) {
67  std::list<std::string> intervalTokens;
68  IBK::explode(tokenTrimmed,intervalTokens,'-');
69  // check for correct interval format
70  if (intervalTokens.size() != 2)
71  throw IBK::Exception(IBK::FormatString("Wrong interval format."),FUNC_ID);
72 
73  std::pair<unsigned int, unsigned int> idInterval = std::make_pair<unsigned int, unsigned int>
74  (IBK::string2val<unsigned int>(intervalTokens.front()),
75  IBK::string2val<unsigned int>(intervalTokens.back()) );
76  m_idIntervals.push_back(idInterval);
77  }
78  else {
79  m_ids.insert(IBK::string2val<unsigned int> (tokenTrimmed) );
80  }
81  }
82  }
83  }
84  catch (IBK::Exception & ex) {
85  throw IBK::Exception( ex, IBK::FormatString("Error initializing IDGroup from String '%1'."), FUNC_ID);
86  }
87  catch (std::exception & ex2) {
88  throw IBK::Exception( IBK::FormatString("%1\nError initializing IDGroup from String '%1'.").arg(ex2.what()).arg(encodedString)
89  , FUNC_ID);
90  }
91 }

94 std::string IDGroup::encodedString() const {
95  std::string str;
96  // first check if all desitnation are chosen
97  if (m_allIDs) {
98  return std::string("*");
99  }
100  // select all intervals
101  if (!m_idIntervals.empty()) {
102  for(unsigned int i=0; i < m_idIntervals.size(); ++i) {
103  if (i != 0)
104  str += ",";
105  str += IBK::val2string<unsigned int> (m_idIntervals[i].first);
106  str += "-";
107  str += IBK::val2string<unsigned int> (m_idIntervals[i].second);
108  }
109  }
110  // second all single ids
111  if (!m_ids.empty()) {
112  // set a delimiter between id-intervals and ids
113  if (!m_idIntervals.empty() )
114  str += ",";
115 
116  for (std::set<unsigned int>::const_iterator it = m_ids.begin(); it != m_ids.end(); ++it) {
117  if (it != m_ids.begin())
118  str += ",";
119  str += IBK::val2string<unsigned int> (*it);
120  }
121  }
122 
123  return str;
124 }

132 bool IDGroup::contains(unsigned int id) const {
133  if (m_allIDs)
134  return true;
135  if (m_ids.find(id) != m_ids.end() )
136  return true;
137  for (unsigned int i = 0; i < m_idIntervals.size(); ++i) {
138  const std::pair<unsigned int, unsigned int> &interval = m_idIntervals[i];
139  // id is inside interval
140  if (id >= interval.first && id <= interval.second)
141  return true;
142  }
143  // all checks failed
144  return false;
145 }

148 const IDGroup IDGroup::operator+(const IDGroup &group) {
149  IDGroup mergedIDGroup;
150  // merge id groups
151  if (group.m_allIDs || m_allIDs) {
152  mergedIDGroup.m_allIDs = true;
153  }
154  else {
155  // merge all single id numbers
156  std::set<unsigned int>::const_iterator idItA = m_ids.begin();
157  for (; idItA != m_ids.end(); ++idItA) {
158  unsigned int idA = *idItA;
159  // check if model id is part of an interval
160  std::vector<std::pair<unsigned int, unsigned int> >::const_iterator intervalItB = group.m_idIntervals.begin();
161  while (intervalItB != group.m_idIntervals.end() &&
162  (idA < intervalItB->first || idA > intervalItB->second))
163  ++ intervalItB;
164  // id number not part of an interval
165  if (intervalItB == group.m_idIntervals.end())
166  mergedIDGroup.m_ids.insert(idA);
167  }
168 
169  for (std::set<unsigned int>::const_iterator idItB = group.m_ids.begin(); idItB != group.m_ids.end(); ++idItB) {
170  unsigned int idB = *idItB;
171  // check if model id is part of an interval
172  std::vector<std::pair<unsigned int, unsigned int> >::const_iterator intervalItA = m_idIntervals.begin();
173  while (intervalItA != m_idIntervals.end() &&
174  (idB < intervalItA->first || idB > intervalItA->second))
175  ++ intervalItA;
176  // id number not part of an interval
177  if (intervalItA == m_idIntervals.end())
178  mergedIDGroup.m_ids.insert(idB);
179  }
180 
181  // cut all id interval from group B out of intervals from group A numbers
182  std::vector<std::pair<unsigned int, unsigned int> > complementIntervals;
183  std::vector<std::pair<unsigned int, unsigned int> >::const_iterator intervalItA = m_idIntervals.begin();
184  for(; intervalItA != m_idIntervals.end(); ++intervalItA) {
185  std::pair<unsigned int, unsigned int> intervalA = *intervalItA;
186  std::vector<std::pair<unsigned int, unsigned int> >::const_iterator intervalItB = group.m_idIntervals.begin();
187  unsigned int k = 0;
188  bool complementIsEmpty = false;
189  // check if model id crosses an id interval, enlarge if necessary and store interval index
190  for (; intervalItB != group.m_idIntervals.end(); ++ intervalItB, ++k) {
191  const unsigned int lowerId = intervalItB->first;
192  const unsigned int upperId = intervalItB->second;
193  // interval A crosses interval B -> reduce interval A
194  if (intervalA.first >= lowerId && intervalA.first <= upperId) {
195  intervalA.first = std::max(intervalA.second + 1,upperId + 1);
196  }
197  if (intervalA.second >= lowerId && intervalA.second <= upperId) {
198  IBK_ASSERT(lowerId != 0 && intervalA.first != 0);
199  intervalA.second = std::min(intervalA.first - 1,lowerId - 1);
200  }
201  // intervalA is empty after complement operation
202  if (intervalA.second == 0 || intervalA.second < intervalA.first) {
203  complementIsEmpty = true;
204  break;
205  }
206  }
207  // only add filled intervals
208  if (!complementIsEmpty)
209  complementIntervals.push_back(intervalA);
210  }
211  mergedIDGroup.m_idIntervals = complementIntervals;
212  // now add all remaining intervals of id group B
213  for (unsigned int i = 0; i < group.m_idIntervals.size(); ++i) {
214  std::pair<unsigned int, unsigned int> intervalB = group.m_idIntervals[i];
215  mergedIDGroup.m_idIntervals.push_back(intervalB);
216  }
217  }
218  return mergedIDGroup;
219 }
