30 /*! The class KeywordList provides conversion functionality between keyword strings
31  and their respective enumeration values.
32  Keyword strings are used in the Delphin project files as identifyer. In the code
33  we use the enumeration values. You can obtain a keyword for a given enumeration value
34  with the static function Keyword(). The corresponding enumeration value can be
35  obtained with Enumeration(), as shown in the example below.
36  \code
37  // to get the keyword string use
38  string kw = KeywordList::Keyword("Assignment::type_t", Assignment::ST_MATERIAL);
39  // and to get the enumeration value
40  Assignment::type_t t = KeywordList::Enumeration("Assignment::type_t}", kw);
41  // the string "Assignment::type_t" is the full enumeration type and formed from the class
42  // name and the enumeration type name
43  \endcode
44  Remember to pass the correct and full enumeration type as first argument to Keyword.
45  \note Instead of overloading the Keyword() function for different types
46  in the current implementation we use a generic form in order to avoid
47  adding compilation dependencies for
48  all CPP files in the project that include the keyword list.
49 */
50 class KeywordList {
51 public:
52  /*!Returns a keyword for an enum value t of type enumtype. */
53  static const char * Keyword(const char * const enumtype, int t);
54 
55  /*! Returns a description for an enum value t of type enumtype.
56  This function throws an exception if the enumeration type is invalid or unknown.
57  If no descrption is given, the keyword itself is returned.
58  \param enumtype The full enumeration type including the class name.
59  \param t The enumeration type cast in an int.
60  \param no_description The optional argument is set to true, if there was no description
61  for this keyword, otherwise to false.
62  */
63  static const char * Description(const char * const enumtype, int t, bool * no_description = nullptr);
64 
65  /*! Returns a default unit for an enum value t of type enumtype.
66  This function throws an exception if the enumeration type is invalid or unknown.
67  Returns an empty string if no default unit was specified.
68  \param enumtype The full enumeration type including the class name.
69  \param t The enumeration type cast in an int.
70  */
71  static const char * Unit(const char * const enumtype, int t);
72 
73  /*! Returns a color string for an enum value t of type enumtype.
74  This function throws an exception if the enumeration type is invalid or unknown.
75  Returns the color code for 'white' if no color value was specified.
76  \param enumtype The full enumeration type including the class name.
77  \param t The enumeration type cast in an int.
78  */
79  static const char * Color(const char * const enumtype, int t);
80 
81  /*! Returns a default value for an enum value t of type enumtype.
82  This function throws an exception if the enumeration type is invalid or unknown.
83  Returns an nan if no default value was specified.
84  \param enumtype The full enumeration type including the class name.
85  \param t The enumeration type cast in an int.
86  */
87  static double DefaultValue(const char * const enumtype, int t);
88 
89  /*! Returns an enumeration value for a given keyword kw of type enumtype.
90  This function throws an exception if the keyword or the enumeration type is invalid or unknown.
91  \param enumtype The full enumeration type including the class name.
92  \param kw The keyword string.
93  \param deprecated The optional argument is set the true if the keyword kw is deprecated.
94  */
95  static int Enumeration(const char * const enumtype, const std::string & kw, bool * deprecated = nullptr);
96 
97 
98  /*! Returns the maximum index for entries of a category in the keyword list.
99  This function throws an exception if the enumeration type is invalid or unknown.
100  \param enumtype The full enumeration type including the class name.
101  */
102  static int MaxIndex(const char * const enumtype);
103 
104 
105  /*! Returns the number of keywords in this category.
106  This function throws an exception if the enumeration type is invalid or unknown.
107  \param enumtype The full enumeration type including the class name.
108  */
109  static unsigned int Count(const char * const enumtype);
110 
111  /*! Checks whether a keyword exists in the enumeration of type enumtype.
112  \return Returns true if the keyword is valid, otherwise false.
113  */
114  static bool KeywordExists(const char * const enumtype, const std::string & kw);
115  /*! Checks whether a category of type enumtype exists.
116  \return Returns true if the category/enum type exists, otherwise false.
117  */
118  static bool CategoryExists(const char * const enumtype);
119 };
