32 void DataTable::setEncodedString(const std::string & str) {
33  FUNCID(DataTable::setEncodedString);
34  std::vector<std::string> tokens;
35  IBK::explode(str, tokens, ';', IBK::EF_TrimTokens);
36  m_values.clear();
37  int count = -1;
38  for (const std::string & colStr : tokens) {
39  std::vector<std::string> parts;
40  IBK::explode_in2(colStr, parts, ':');
41  if (parts.size() != 2)
42  throw IBK::Exception(IBK::FormatString("Invalid data in table, expected ':' as separator in column data '%1'").arg(colStr), FUNC_ID);
43  std::vector<double> val;
44  std::string line_with_spaces = parts[1];
45  IBK::replace_string(parts[1], ",", " ");
46  try {
47  IBK::string2valueVector(line_with_spaces, val);
48  } catch (IBK::Exception & ex) {
49  throw IBK::Exception(ex, IBK::FormatString("Invalid data in table, in column data '%1'").arg(colStr), FUNC_ID);
50  }
51  // check for correct column count
52  if (count == -1)
53  count = val.size();
54  else if (count != (int)val.size())
55  throw IBK::Exception(IBK::FormatString("Mismatching number of values (=%1) in column '%2' compared to previous columns.").arg(val.size()).arg(parts[0]), FUNC_ID);
56  // check that column name isn't duplicated
57  if (m_values.find(parts[0]) != m_values.end())
58  throw IBK::Exception(IBK::FormatString("Duplicate column ID '%1' in table.").arg(parts[0]), FUNC_ID);
59  m_values[parts[0]] = val;
60  }
61 }

64 std::string DataTable::encodedString() const {
65  // encode data table
66  std::stringstream strm;
67  for (const std::pair<const std::string, std::vector<double> > & column : m_values) {
68  strm << IBK::trim_copy(column.first) << ":";
69  for (unsigned int i=0; i<column.second.size(); ++i) {
70  strm << column.second[i];
71  if (i < column.second.size()-1)
72  strm << ",";
73  }
74 
75  strm << ";";
76  }
77  return strm.str();
78 }
