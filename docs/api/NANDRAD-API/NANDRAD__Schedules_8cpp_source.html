45 void Schedules::readXML(const TiXmlElement * element) {
46  FUNCID(Schedules::readXML);
47 
48  // reading elements
49  const TiXmlElement * c = element->FirstChildElement();
50  while (c) {
51  const std::string & cName = c->ValueStr();
52  if (cName == "Holidays") {
53  IBK::Time::TimeFormatInfo fmt = IBK::Time::formatInfo("dd.MM.");
54  const TiXmlElement * c2 = c->FirstChildElement();
55  while (c2) {
56  const std::string & c2Name = c2->ValueStr();
57  if (c2Name != "FirstDayOfYear")
58  throw IBK::Exception(IBK::FormatString(XML_READ_UNKNOWN_ELEMENT).arg(c2Name).arg(c2->Row()), FUNC_ID);
59 
60  IBK::Time t = IBK::Time::fromString(c2->GetText(), fmt, false);
61  if (!t.isValid())
62  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(c2->Row())
63  .arg("Invalid date format '"+std::string(c2->GetText())+"', expected date in format 'dd.MM.'."), FUNC_ID);
64  unsigned int dayOfYear = static_cast<unsigned int>(t.secondsOfYear()/(3600*24));
65  if (m_holidays.find(dayOfYear) != m_holidays.end())
66  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(c2->Row())
67  .arg("Duplicate holiday date '"+std::string(c2->GetText())+"'"), FUNC_ID);
68  m_holidays.insert(dayOfYear);
69 
70  c2 = c2->NextSiblingElement();
71  }
72 
73  }
74  else if (cName == "WeekEndDays") {
75  std::string days = c->GetText();
76  // split days at ,
77  std::vector<std::string> dayvec;
78  IBK::explode(days, dayvec, ",", IBK::EF_TrimTokens);
79  // convert all days into enums
80  for (const std::string & d : dayvec) {
81  try {
82  day_t dt = (day_t)KeywordList::Enumeration("Schedules::day_t", d);
83  m_weekEndDays.insert(dt);
84  }
85  catch (IBK::Exception & ex) {
86  throw IBK::Exception(ex, IBK::FormatString(XML_READ_ERROR).arg(c->Row())
87  .arg("Invalid day name in 'WeekEndDays' tag."), FUNC_ID);
88  }
89  }
90  }
91  else if (cName == "IBK:Flag") {
92  IBK::Flag f;
93  readFlagElement(c, f);
94  bool success = false;
95  try {
96  flag_t ftype = (flag_t)KeywordList::Enumeration("Schedules::flag_t", f.name());
97  m_flags[ftype] = f; success=true;
98  }
99  catch (...) { /* intentional fail */ }
100  if (!success)
101  IBK::IBK_Message(IBK::FormatString(XML_READ_UNKNOWN_NAME).arg(f.name()).arg(cName).arg(c->Row()), IBK::MSG_WARNING, FUNC_ID, IBK::VL_STANDARD);
102  }
103  else if (cName == "ScheduleGroups") {
104  const TiXmlElement * c2 = c->FirstChildElement();
105  while (c2) {
106  const std::string & c2Name = c2->ValueStr();
107  if (c2Name != "ScheduleGroup")
108  throw IBK::Exception(IBK::FormatString(XML_READ_UNKNOWN_ELEMENT).arg(c2Name).arg(c2->Row()), FUNC_ID);
109 
110  const TiXmlAttribute * attrib = TiXmlAttribute::attributeByName(c2, "objectList");
111  if (attrib == nullptr)
112  throw IBK::Exception( IBK::FormatString(XML_READ_ERROR).arg(c2->Row()).arg(
113  "Missing required 'objectList' attribute."), FUNC_ID);
114 
115  std::string objectListName = attrib->ValueStr();
116  // ensure that we do not have duplicate definitions
117  if (m_scheduleGroups.find(objectListName) != m_scheduleGroups.end())
118  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(c2->Row()).arg(
119  "ObjectList '"+objectListName+"' used for multiple ScheduleGroup elements."), FUNC_ID);
120 
121  std::vector<Schedule> schedules;
122  // now read all the schedule subtags
123 
124  const TiXmlElement * c3 = c2->FirstChildElement();
125  while (c3) {
126  const std::string & c3Name = c3->ValueStr();
127  if (c3Name != "Schedule")
128  throw IBK::Exception(IBK::FormatString(XML_READ_UNKNOWN_ELEMENT).arg(c3Name).arg(c3->Row()), FUNC_ID);
129 
130  Schedule s;
131  try {
132  s.readXML(c3);
133  } catch (IBK::Exception & ex) {
134  throw IBK::Exception(ex, "Error reading 'ScheduleGroup' element.", FUNC_ID);
135  }
136  schedules.push_back(s);
137 
138  c3 = c3->NextSiblingElement();
139  }
140 
141  m_scheduleGroups[objectListName] = schedules;
142 
143  c2 = c2->NextSiblingElement();
144  }
145  }
146  else if (cName == "AnnualSchedules") {
147  const TiXmlElement * c2 = c->FirstChildElement();
148  while (c2) {
149  const std::string & c2Name = c2->ValueStr();
150  if (c2Name != "ScheduleGroup")
151  throw IBK::Exception(IBK::FormatString(XML_READ_UNKNOWN_ELEMENT).arg(c2Name).arg(c2->Row()), FUNC_ID);
152 
153  const TiXmlAttribute * attrib = TiXmlAttribute::attributeByName(c2, "objectList");
154  if (attrib == nullptr)
155  throw IBK::Exception( IBK::FormatString(XML_READ_ERROR).arg(c2->Row()).arg(
156  "Missing required 'objectList' attribute."), FUNC_ID);
157 
158  std::string objectListName = attrib->ValueStr();
159  // ensure that we do not have duplicate definitions
160  if (m_scheduleGroups.find(objectListName) != m_scheduleGroups.end())
161  throw IBK::Exception(IBK::FormatString(XML_READ_ERROR).arg(c2->Row()).arg(
162  "ObjectList '"+objectListName+"' used for multiple ScheduleGroup elements."), FUNC_ID);
163 
164  std::vector<NANDRAD::LinearSplineParameter> schedules;
165  // now read all the schedule subtags
166 
167  const TiXmlElement * c3 = c2->FirstChildElement();
168  while (c3) {
169  const std::string & c3Name = c3->ValueStr();
170  if (c3Name != "AnnualSchedule")
171  throw IBK::Exception(IBK::FormatString(XML_READ_UNKNOWN_ELEMENT).arg(c3Name).arg(c3->Row()), FUNC_ID);
172 
173  NANDRAD::LinearSplineParameter spl;
174  try {
175  spl.readXML(c2);
176  }
177  catch (IBK::Exception & ex) {
178  throw IBK::Exception(ex, IBK::FormatString(XML_READ_ERROR).arg(c2->Row())
179  .arg("Invalid data in 'IBK:LinearSpline' tag."), FUNC_ID);
180  }
181  schedules.push_back(spl);
182 
183  c3 = c3->NextSiblingElement();
184  }
185 
186  m_annualSchedules[objectListName] = schedules;
187 
188  c2 = c2->NextSiblingElement();
189  }
190  }
191  else {
192  IBK::IBK_Message(IBK::FormatString(XML_READ_UNKNOWN_ELEMENT).arg(cName).arg(c->Row()), IBK::MSG_WARNING, FUNC_ID, IBK::VL_STANDARD);
193  }
194  c = c->NextSiblingElement();
195  }
196 }

199 TiXmlElement * Schedules::writeXML(TiXmlElement * parent) const {
200  TiXmlElement * e = new TiXmlElement("Schedules");
201  parent->LinkEndChild(e);
202 
203  if (!m_holidays.empty()) {
204  TiXmlElement * c = new TiXmlElement("Holidays");
205  e->LinkEndChild(c);
206  // encode days
207  for (unsigned int dayOfYear : m_holidays) {
208  // compose IBK::Time object
209  IBK::Time t;
210  t.set(2003, dayOfYear*3600*24);
211  TiXmlElement::appendSingleAttributeElement(c, "IBK:Time", nullptr, std::string(), t.toDayMonthFormat());
212  }
213  }
214 
215  if (!m_weekEndDays.empty()) {
216  TiXmlElement * c = new TiXmlElement("WeekEndDays");
217  e->LinkEndChild(c);
218  // encode days
219  std::string days;
220  for (day_t t : m_weekEndDays)
221  days += std::string(",") + KeywordList::Keyword("Schedules::day_t", t);
222  TiXmlText * text = new TiXmlText( days.substr(1) ); // Mind: remove leading , from string
223  c->LinkEndChild(text);
224  }
225 
226  for (int i=0; i<NUM_F; ++i) {
227  if (!m_flags[i].name().empty())
228  TiXmlElement::appendSingleAttributeElement(e, "IBK:Flag", "name", m_flags[i].name(), m_flags[i].isEnabled() ? "true" : "false");
229  }
230 
231  // now write schedules
232  if (!m_scheduleGroups.empty()) {
233  TiXmlElement * c = new TiXmlElement("ScheduleGroups");
234  e->LinkEndChild(c);
235  for (const std::pair<const std::string, std::vector<Schedule> > & svec : m_scheduleGroups) {
236  // create tags like
237  // <ScheduleGroup objectList="objectListName">
238  // <Schedule>...</Schedule>
239  // </ScheduleGroup>
240  TiXmlElement * g = new TiXmlElement("ScheduleGroup");
241  c->LinkEndChild(g);
242  g->SetAttribute("objectList", svec.first);
243  for (const Schedule & s : svec.second)
244  s.writeXML(g);
245  }
246  }
247 
248  // now write schedules
249  if (!m_annualSchedules.empty()) {
250  TiXmlElement * c = new TiXmlElement("AnnualSchedules");
251  e->LinkEndChild(c);
252  for (const std::pair<const std::string, std::vector<NANDRAD::LinearSplineParameter> > & svec : m_annualSchedules) {
253  // create tags like
254  // <AnnualSchedule objectList="objectListName">
255  // ...
256  // </AnnualSchedule>
257  TiXmlElement * g = new TiXmlElement("AnnualSchedule");
258  c->LinkEndChild(g);
259  g->SetAttribute("objectList", svec.first);
260  for (const NANDRAD::LinearSplineParameter & s : svec.second)
261  s.writeXML(g);
262  }
263  }
264 
265  return nullptr;
266 }

278 void Schedules::generateLinearSpline(const std::string & objectListName, const std::string & parameterName,
279  IBK::LinearSpline & spline, DailyCycle::interpolation_t & interpolationType) const
280 {
281  FUNCID(Schedules::generateLinearSpline);
282 
283  /*
284  - loop over all days (d=0,1,...,364)
285  - determine day type:
286  dayOfWeek = (startDayOffset + d) % 7 (modulo 7)
287  - look up daily cycle:
288  - find schedule where d in range:
289  - process daytypes in order Holidays, CurrentDayType, Weekdays/Weekends, AllDays
290  - if parameter is found in any of these days, take daily course and add to spline for this day,
291  - if parameter not found, skip and search through next schedule
292  */
293 
294  interpolationType = DailyCycle::NUM_IT; // initialize with invalid interpolation mode first, will be set and compared below
295 
296  // TIME_SHIFT defines the minimum gap introduced between two values in the spline,
297  // to avoid extreme gradients and thus slowdown of simulation.
298  // Note: if someone started a new interval just shortly (< TIME_SHIFT) before
299  // end of the day, this will generate a non-monotonic time vector which
300  // will blow up in our face once we generate the linear spline below.
301  // Thankfully, we catch that and can inform the user about this parameter
302  // problem.
303  const double TIME_SHIFT = 120; // in [s]
304 
305 
306  std::vector<double> tp;
307  IBK::UnitVector vals;
308  // loop over all days
309  for (unsigned int d=0; d<365; ++d) {
310  // compute date type
311  day_t dayOfWeek = (day_t)((d + m_firstDayOfYear) % 7);
312 
313  // also determine if this day is a holiday
314  bool isHoliday = (m_holidays.find(d) != m_holidays.end());
315 
316  // get vector of schedules for given object list
317  std::map<std::string, std::vector<Schedule> >::const_iterator it = m_scheduleGroups.find(objectListName);
318  IBK_ASSERT(it != m_scheduleGroups.end());
319  const std::vector<Schedule> & schedules = it->second;
320 
321  std::vector<const Schedule *> scheduleCandidates[NANDRAD::Schedule::NUM_ST];
322 
323  // now search through schedules and build a list of schedules sorted according to priority (i.e. daytype)
324  for (const Schedule & sched : schedules) {
325  // check for correct interval
326  if (!sched.containsDay(d))
327  continue; // outside scheduled date range - skip
328 
329  // look at schedule day type and see if our day fits
330  bool keep = false;
331  switch (sched.m_type) {
332  // AllDays are always kept
333  case NANDRAD::Schedule::ST_ALLDAYS :
334  keep = true;
335  break;
336 
337  // holidays are only kept if this is a holiday schedule
338  case NANDRAD::Schedule::ST_HOLIDAY :
339  if (isHoliday)
340  keep = true;
341  break;
342 
343  case NANDRAD::Schedule::ST_WEEKDAY :
344  // weekday schedules are only kept, if daytype is a weekday
345  if (dayOfWeek >= NANDRAD::Schedules::SD_MONDAY &&
346  dayOfWeek <= NANDRAD::Schedules::SD_FRIDAY)
347  {
348  keep = true;
349  }
350  break;
351 
352  case NANDRAD::Schedule::ST_WEEKEND :
353  // weekend schedules are only kept, if daytype is a weekend
354  if (dayOfWeek >= NANDRAD::Schedules::SD_SATURDAY &&
355  dayOfWeek <= NANDRAD::Schedules::SD_SUNDAY)
356  {
357  keep = true;
358  }
359  break;
360 
361  default :
362  // otherwise we only keep the schedule if this is a "our" day
363  if (sched.m_type == (int)dayOfWeek+NANDRAD::Schedule::ST_MONDAY) // mind the shift in enum numbers
364  keep = true;
365  }
366 
367  if (!keep)
368  continue; // schedule was filtered out, skip it
369 
370  // check if schedule provides the requested parameter
371  // Note: this is a slow operation, so we do it last
372  if (sched.m_parameters.find(parameterName) == sched.m_parameters.end())
373  continue; // parameter not provided
374 
375  // remember based on date type
376  scheduleCandidates[sched.m_type].push_back(&sched);
377  }
378 
379  // now we have a set of schedules to take our parameter from
380  // each of the lists should have at max one parameter - otherwise we have an ambiguity
381 
382  // check all lists types for duplicate schedules
383  for (unsigned int schedDayType=0; schedDayType<NANDRAD::Schedule::NUM_ST; ++schedDayType) {
384  IBK::Time dString(2003, (d + m_firstDayOfYear)*IBK::SECONDS_PER_DAY);
385  if (scheduleCandidates[schedDayType].size() > 1) {
386  // we allow at max. 2 schedules, but only if exactly one of the
387  // two is a whole year schedule
388  // so, we basically count the number of whole year schedules
389  // and bail out if we have more than one, otherwise we subtract
390  // the number of all year schedules from the number of schedules
391  // and check, that 0 or 1 remain
392  unsigned int allYearCount = 0;
393  for (const NANDRAD::Schedule * schedPtr : scheduleCandidates[schedDayType]) {
394  if (schedPtr->isWholeYearSchedule())
395  ++allYearCount;
396  }
397  if (allYearCount > 1)
398  throw IBK::Exception(IBK::FormatString("Ambiguous schedule parameter definition for "
399  "parameter '%1'. Multiple whole year schedules match this day.")
400  .arg(parameterName), FUNC_ID);
401  if (scheduleCandidates[schedDayType].size() - allYearCount > 1)
402  throw IBK::Exception(IBK::FormatString("Ambiguous schedule parameter definition for parameter '%1' at day '%2'. %3 "
403  "schedules of day type '%4' match this day.")
404  .arg(parameterName).arg(dString.toDayMonthFormat())
405  .arg(scheduleCandidates[schedDayType].size())
406  .arg(NANDRAD::KeywordList::Keyword("Schedule::type_t", (int)schedDayType)), FUNC_ID);
407  }
408  }
409 
410  // search backwards so that we find the highest priority schedule
411  bool dataAdded = false;
412  for (int i=NANDRAD::Schedule::NUM_ST-1; i>=0; --i) {
413  if (scheduleCandidates[i].empty()) continue; // skip day types without schedule
414  const NANDRAD::Schedule * sched = scheduleCandidates[i].front();
415  // priority handling for the case that we have both an whole year schedule and a limited schedule
416  if (scheduleCandidates[i].size() > 1) {
417  if (sched->isWholeYearSchedule())
418  sched = scheduleCandidates[i].back();
419  }
420  // find daily cycle that provides the parameter
421  std::map<std::string, NANDRAD::DailyCycle *>::const_iterator paraIT = sched->m_parameters.find(parameterName);
422  IBK_ASSERT(paraIT != sched->m_parameters.end());
423 
424  const NANDRAD::DailyCycle * dc = paraIT->second;
425  // retrieve daily cycle parameter data
426  const std::vector<NANDRAD::DailyCycle::valueData_t>::const_iterator valueDataIT =
427  std::find(dc->m_valueData.begin(), dc->m_valueData.end(), parameterName);
428  IBK_ASSERT(valueDataIT != dc->m_valueData.end());
429  const NANDRAD::DailyCycle::valueData_t & valData = *valueDataIT;
430 
431  // initialize interpolation type and check that interpolation type matches
432  // (must be the same in all daily cycles)
433  if (interpolationType == DailyCycle::NUM_IT)
434  interpolationType = dc->m_interpolation;
435  else if (interpolationType != dc->m_interpolation) {
436  throw IBK::Exception(IBK::FormatString("Mismatching interpolation types for parameter '%1' in "
437  "different DailyCycles.").arg(parameterName), FUNC_ID);
438  }
439 
440  // initialize unit and check that unit is always the same
441  if (vals.m_unit.id() == 0) {
442  vals.m_unit = valData.m_unit;
443  }
444  else {
445  if (vals.m_unit != valData.m_unit)
446  throw IBK::Exception(IBK::FormatString("Parameter '%1' was defined with unit '%2' and unit '%3' "
447  "in different DailyCycles. Please use the same unit for "
448  "the same parameter!")
449  .arg(parameterName).arg(vals.m_unit.name()).arg(valData.m_unit.name()), FUNC_ID);
450  }
451 
452 
453  // now finally append the daily spline to the global spline
454  for (unsigned int h=0; h<dc->m_timePoints.size(); ++h) {
455  double tpsec = dc->m_timePoints[h]*3600 + d*IBK::SECONDS_PER_DAY;
456  double val = (*valData.m_valueVec)[h];
457  if (tp.empty()) {
458  tp.push_back(tpsec);
459  vals.m_data.push_back(val);
460  }
461  else {
462  // different handling for constant and linearly interpolated values
463  if (interpolationType == NANDRAD::DailyCycle::IT_CONSTANT) {
464  // here we always add two points, one that ends the last interval and
465  // one that starts the new interval
466  tp.push_back(tpsec-TIME_SHIFT);
467  vals.m_data.push_back(vals.m_data.back());
468  tp.push_back(tpsec);
469  vals.m_data.push_back(val);
470  }
471  else {
472  // check if same tp is already in the data vector, and by same we mean
473  // dt = 0.1 s
474 
475  const double TIME_DELTA = 0.1;
476  double tplast = tp.back();
477  if (IBK::near_equal(tplast, tpsec, TIME_DELTA)) {
478  // instead of adding the same time point, move the last time point backwards
479  // Note: since we checked validity of all daily cycles already,
480  // we must have already at least 2 time points in the vector.
481  IBK_ASSERT(tp.size() >= 2);
482  // check if we can safely move the time point prior to
483  tp.back() -= TIME_SHIFT;
484  }
485  }
486  }
487  }
488  dataAdded = true; // we have added data for the day
489  break; // break the loop, since we have found our parameter set
490  } // loop to process all cycles
491 
492  // throw an exception if a parameter definition is missing for a day
493  if (!dataAdded) {
494  throw IBK::Exception(IBK::FormatString("Couldn't find valid DailyCycle parameter data at day %1 "
495  "for parameter '%2'.").arg(d).arg(parameterName), FUNC_ID);
496  }
497 
498  } // loop over all days of the year
499 
500  // in case of constant extrapolation, re-add the last time point close to the end of the interval
501  // so that we can have cyclic use without error in the last interval
502  if (interpolationType == NANDRAD::DailyCycle::IT_CONSTANT) {
503  tp.push_back(365*24*3600-TIME_SHIFT);
504  vals.m_data.push_back(vals.m_data.back());
505  }
506 
507  // convert the values to base SI unit and finally generate the spline
508  vals.convert(IBK::Unit(vals.m_unit.base_id())); // does not throw
509  // now generate spline
510  spline.setValues(tp, vals.m_data); // does not throw
511 
512  /// \todo remove redundant values from spline, i.e when you have {0, 0; 5,0; 10,0} you can drop the middle point
513  /// {0, 0; 10,0} without data loss
514  std::string errmsg;
515  bool success = spline.makeSpline(errmsg); // does not throw
516  if (!success)
517  throw IBK::Exception(IBK::FormatString("%1\nError creating spline for parameter '%2'. Possibly "
518  "a time interval ended too close to the end point of the interval. "
519  "Do not add time points < 24 h and less then < 2 minutes before the end of the "
520  "interval.").arg(errmsg).arg(parameterName), FUNC_ID);
521 }
