43 void Schedule::prepareCalculation() {
44  FUNCID(Schedule::prepareCalculation);
45  // loop over all daily cycles and initialize them
46  for (unsigned int i=0; i<m_dailyCycles.size(); ++i) {
47  NANDRAD::DailyCycle & dc = m_dailyCycles[i];
48  try {
49  dc.prepareCalculation();
50  }
51  catch (IBK::Exception & ex) {
52  throw IBK::Exception(ex, IBK::FormatString("Error initializing DailyCycle #%1.").arg(i), FUNC_ID);
53  }
54  for (const DailyCycle::valueData_t & paraData : dc.m_valueData) {
55  if (m_parameters.find(paraData.m_name) != m_parameters.end())
56  throw IBK::Exception(IBK::FormatString("Multiple DailyCycle blocks define the same parameter '%1'. This is not allowed!")
57  .arg(paraData.m_name), FUNC_ID);
58  m_parameters[paraData.m_name] = &dc; // persistent pointer guaranteed - we do not modifiy the m_dailyCycles vector anylonger
59  }
60  }
61 }
