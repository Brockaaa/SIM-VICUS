30 void Interface::updateComment(const std::vector<Zone> & zones) {
31 
32  if (m_zoneId == 0)
33  m_comment = "Interface to outside";
34  else {
35  // lookup zone and its display name\n"
36  std::vector<Zone>::const_iterator it = std::find(zones.begin(), zones.end(), m_zoneId);
37  if (it != zones.end()) {
38  if (!it->m_displayName.empty())
39  m_comment = IBK::FormatString("Interface to '%1'").arg(it->m_displayName).str();
40  else
41  m_comment = IBK::FormatString("Interface to anonymous zone with id #%1").arg(m_zoneId).str();
42  }
43  else {
44  m_comment = IBK::FormatString("ERROR: Zone with id #%1 does not exist.").arg(m_zoneId).str();
45  }
46  }
47 }

50 bool Interface::haveBCParameters() const {
51  if (m_id == INVALID_ID) return false;
52  if (m_heatConduction.m_modelType != InterfaceHeatConduction::NUM_MT) return true;
53  if (m_solarAbsorption.m_modelType != InterfaceSolarAbsorption::NUM_MT) return true;
54  if (m_longWaveEmission.m_modelType != InterfaceLongWaveEmission::NUM_MT) return true;
55  if (m_vaporDiffusion.m_modelType != InterfaceVaporDiffusion::NUM_MT) return true;
56  if (m_airFlow.m_modelType != InterfaceAirFlow::NUM_MT) return true;
57  return false;
58 }
