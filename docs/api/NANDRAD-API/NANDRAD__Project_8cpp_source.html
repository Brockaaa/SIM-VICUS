37 void Project::readXML(const IBK::Path & filename) {
38  FUNCID(Project::readXML);
39 
40  TiXmlDocument doc;
41  IBK::Path filenamePath(filename);
42  TiXmlElement * xmlElem = openXMLFile(m_placeholders, filenamePath, "NandradProject", doc);
43  if (!xmlElem)
44  return; // empty project, this means we are using only defaults
45 
46  // we read our subsections from this handle
47  TiXmlHandle xmlRoot = TiXmlHandle(xmlElem);
48 
49  // Directory Placeholders
50  xmlElem = xmlRoot.FirstChild("DirectoryPlaceholders").Element();
51  if (xmlElem) {
52  readDirectoryPlaceholdersXML(xmlElem);
53  // check for duplicate sections
54  xmlElem = xmlRoot.Child("DirectoryPlaceholders", 1).Element();
55  if (xmlElem != nullptr) {
56  throw IBK::Exception(IBK::FormatString("Duplicate section 'DirectoryPlaceholders'."), FUNC_ID);
57  }
58  }
59 
60  // add the project directory to the placeholders map
61  m_placeholders[IBK::PLACEHOLDER_PROJECT_DIR] = filenamePath.parentPath();
62 
63  try {
64  xmlElem = xmlRoot.FirstChild("Project").Element();
65  if (xmlElem) {
66  readXMLPrivate(xmlElem);
67  }
68  }
69  catch (IBK::Exception & ex) {
70  throw IBK::Exception(ex, IBK::FormatString("Error reading project '%1'.").arg(filename), FUNC_ID);
71  }
72 
73  /// \todo check uniqueness of all IDs in all separate id spaces
74  /// implement this via template algorithm, since all id-holding data types have a "find by ID"
75  /// comparison operator and are stored in vectors
76 }

79 void Project::writeXML(const IBK::Path & filename) const {
80  TiXmlDocument doc;
81  TiXmlDeclaration * decl = new TiXmlDeclaration( "1.0", "UTF-8", "" );
82  doc.LinkEndChild( decl );
83 
84  TiXmlElement * root = new TiXmlElement( "NandradProject" );
85  doc.LinkEndChild(root);
86 
87  root->SetAttribute("fileVersion", VERSION);
88 
89  // update interface-zone comment
90  for (ConstructionInstance & con : const_cast<std::vector<ConstructionInstance>&>(m_constructionInstances)) {
91  con.m_interfaceA.updateComment(m_zones);
92  con.m_interfaceB.updateComment(m_zones);
93  }
94 
95  writeDirectoryPlaceholdersXML(root);
96  writeXMLPrivate(root);
97 
98  doc.SaveFile( filename.c_str() );
99 }

109 void Project::readDirectoryPlaceholdersXML(const TiXmlElement * element) {
110 
111  // loop over all elements in this XML element
112  for (const TiXmlElement * e=element->FirstChildElement(); e; e = e->NextSiblingElement()) {
113 
114  // get element name
115  std::string name = e->Value();
116  // handle known elements
117 
118  if (name == "Placeholder") {
119  // search for attribute with given name
120  const TiXmlAttribute* attrib = TiXmlAttribute::attributeByName(e, "name");
121  if (attrib == nullptr) {
122  IBK::IBK_Message(IBK::FormatString(
123  "Missing '%1' attribute in Placeholder element.").arg("name"), IBK::MSG_WARNING);
124  continue;
125  }
126  m_placeholders[attrib->Value()] = std::string(e->GetText());
127  }
128  else {
129  IBK::IBK_Message(IBK::FormatString(
130  "Unknown element '%1' in DirectoryPlaceholders section.").arg(name), IBK::MSG_WARNING);
131  }
132 
133  }

137 void Project::writeDirectoryPlaceholdersXML(TiXmlElement * parent) const {
138 
139  if (m_placeholders.empty() ||
140  (m_placeholders.size() == 1 && m_placeholders.begin()->first == IBK::PLACEHOLDER_PROJECT_DIR))
141  return;
142 
143  TiXmlComment::addComment(parent,
144  "DirectoryPlaceholders section defines strings to be substituted with directories");
145 
146  TiXmlElement * e1 = new TiXmlElement( "DirectoryPlaceholders" );
147  parent->LinkEndChild( e1 );
148 
149  for (std::map<std::string, IBK::Path>::const_iterator it = m_placeholders.begin();
150  it != m_placeholders.end(); ++it)
151  {
152  if (it->first != IBK::PLACEHOLDER_PROJECT_DIR)
153  TiXmlElement::appendSingleAttributeElement(e1, "Placeholder",
154  "name", it->first,
155  it->second.str());
156  }
157 
158  TiXmlComment::addSeparatorComment(parent);
159 }

162 void Project::mergeSameConstructions() {
163  FUNCID(Project::mergeSameConstructions);
164 
165  IBK::IBK_Message("Merging redundant construction instances\n", IBK::MSG_PROGRESS, FUNC_ID, IBK::VL_STANDARD);
166  IBK_MSG_INDENT;
167 
168 #if 0
169 
170  // general concept:
171  // - all modifications are done in temporary memory first, and then applied collectively
172  // to project data structure
173 
174  // algorithm steps
175 
176  // 1. first determine which construction may not be merged due to other models referencing
177  // the construction (e.g. for internal loads, or when the zone has view factors).
178  // Store these construction IDs in a set, since they will be probably much less than the
179  // original construction list, and we need a fast search for such construction IDs later on.
180 
181  // 2. create a vector with "merge" entries for all constructions (index = -1 > not merged; index >= 0 -> construction index)
182 
183  // 3. process all construction (outer loop) -> current construction
184  // skip all constructions that are in the "excluded" set
185  // skip all constructions that are marked "merged"
186  // skip constructions with near zero area (they are ignored in the calculation anyways) - DISCUSS THREASHOLD!
187  // (alternatively, try to merge those constructions, since then outputs will be available for these constructions as well
188  // due to reverse mapping)
189  // process all constructions with index > current construction
190  // skip all constructions that are in the "excluded" set
191  // skip all constructions that are marked "merged"
192  // compare constructions based on:
193  // - construction type
194  // - boundary conditions
195  // - connected zones (mind, that construction may be flipped)
196  // if constructions are the same, record possible merge:
197  // - label construction to be mergable with current construction (store index of current construction)
198  // - mark current construction to be merged with self (store index of current construction in position of current construction)
199 
200  // 4. create new construction instances for merged constructions:
201  // - process "merged" record vector and for each entry "marked as self":
202  // - create a new construction instance as copy
203  // - store index to new construction in "merged" vector in place of original construction
204  // - process "merged" record vector and for each entry "marked as merged":
205  // - if mergable-construction index is in the range of the new construction instances, simply add an entry in the merge-info table
206  // - for other constructions, look up index in place of mergabe construction (this will now point to a newly created instance)
207  // -> add own cross section area to area of construction
208  // -> add own embedded objects to the list of embedded objects in the merged construction
209  // -> add merge entry
210  // -> change "merge" marker to hold the index of the merged new construction instance
211  // - copy all construction instances that are not merged into separate vector
212 
213  // Now the merge-entry vector holds for each merged constructionn the index of the associated newly create construction instance.
214 
215  // 5. update embedded object references
216  // process all embedded objects
217  // look up the construction index from referenced ID
218  // check merge-entry vector if construction was merged
219  // if so, record a "embedded object modified" info and adjust the construction instance ID in the embedded object
220 
221  // the algorithm works on indexed constructions, so create a vector with all construction instances
222  std::vector<ConstructionInstance> conInstances;
223  unsigned int numConInstances = m_constructionInstances.size();
224  conInstances.reserve(numConInstances); // avoid re-allocs during copy
225  for (const std::pair<unsigned int, ConstructionInstance> & c : m_constructionInstances)
226  conInstances.push_back(c.second);
227 
228 
229  // create a set of "excluded" construction instances
230 
231  std::set<unsigned int> excludedConstructionsIDs;
232  const char * const ModelsThatReferenceConstructionInstances[] = {
233  "ConstructionInternalSourceModel", "UnderfloorHeatingModel"
234  };
235  // loop over all models
236  for (const Model & m : m_models.m_models) { // C++11 range loop here (I'm lazy)
237  for (const char * modelID : ModelsThatReferenceConstructionInstances) {
238  if (m.m_modelIdName == modelID) {
239  // retrieve matching object list for all model feedbacks
240  for (const ImplicitModelFeedback & mf : m.m_implicitModelFeedbacks) {
241  const std::string & ol = mf.m_objectList;
242  std::map<std::string, ObjectList>::const_iterator it = m_objectLists.m_objectLists.find(ol);
243  if (it != m_objectLists.m_objectLists.end()) { // should be an assert, but we haven't checked the object structure yet, will be done during model init
244  // process all construction instances and remember those that are in the given ID range
245  for (const ConstructionInstance & c : conInstances) {
246  if (it->second.m_filterID.contains(c.m_id))
247  excludedConstructionsIDs.insert(c.m_id);
248  }
249  }
250  }
251  // add construction IDs of all referenced constructions to 'excludedConstructionsIDs'
252  }
253  }
254  }
255  // loop over all construction instances and their interfaces and associated zones: if the zone has view factors, the
256  // construction instance may not be merged
257  for (unsigned int i=0; i<numConInstances; ++i) {
258  // exclude construction with FMU export or import
259  if (!conInstances[i].m_FMUExportReferences.empty() ||
260  !conInstances[i].m_FMUImportReferences.empty()) {
261 
262  excludedConstructionsIDs.insert(conInstances[i].m_id);
263  continue;
264  }
265  // process all interfaces
266  for (const Interface & iface : conInstances[i].m_interfaces) {
267  if (iface.m_zoneId == 0) {
268  // check if outside has shading factors
269  if (!m_location.m_shadingFactorFileName.str().empty()) {
270  excludedConstructionsIDs.insert(conInstances[i].m_id);
271  break;
272  }
273  }
274  std::map< unsigned int, Zone>::const_iterator it = m_zones.find(iface.m_zoneId);
275  if (it == m_zones.end()) continue; // invalid zone ID -> the error will be triggered later again
276  // check if zone has view factors
277  if (!it->second.m_viewFactors.empty())
278  excludedConstructionsIDs.insert(conInstances[i].m_id);
279  }
280  }
281 
282  const char * const ModelsWithViewFactors[] = {
283  "HeatingPointSourceModel", "LightingPointSourceModel"
284  };
285  // search for point soucrs with view factors
286  // loop over all models
287  for (const Model & m : m_models.m_models) { // C++11 range loop here (I'm lazy)
288  for (const char * modelID : ModelsWithViewFactors) {
289  if (m.m_modelIdName == modelID) {
290  // retrieve matching object list for all model feedbacks
291 
292  std::map <unsigned int, double> viewFactors;
293  try {
294  extractViewFactorsFromModel(m, viewFactors);
295  }
296  catch (IBK::Exception &ex) {
297  throw IBK::Exception(ex, IBK::FormatString("Error checking #%1 with id #%2!")
298  .arg(modelID).arg(m.m_id), FUNC_ID);
299  }
300  // process all construction instances that neighbor interfaces from view factors or embedded objects
301  for (const ConstructionInstance & c : conInstances) {
302  const std::vector<Interface> &ifaces = c.m_interfaces;
303  const std::vector<EmbeddedObject> &objects = c.m_embeddedObjects;
304  // check all interfaces
305  for (const Interface & i : ifaces) {
306  if (viewFactors.find(i.m_id) != viewFactors.end()) {
307  excludedConstructionsIDs.insert(c.m_id);
308  break;
309  }
310  }
311  // check all embedded objects
312  for (const EmbeddedObject & o : objects) {
313  if (viewFactors.find(o.m_id) != viewFactors.end()) {
314  excludedConstructionsIDs.insert(c.m_id);
315  break;
316  }
317  }
318  }
319  // add construction IDs of all referenced constructions to 'excludedConstructionsIDs'
320  }
321  }
322  }
323 
324  // create vector with merge markers
325  std::vector<int> canBeMergedTo(numConInstances, -1); // initialized with -1 - not merged
326 
327 
328  // process all construction instances
329  for (unsigned int i=0; i<numConInstances; ++i) {
330  const ConstructionInstance & currentConInstance = conInstances[i]; // readability improvement
331  // skip construction instance, if already merged
332  if (canBeMergedTo[i] != -1)
333  continue;
334  // skip construction instance, if in excluded
335  if (excludedConstructionsIDs.find(currentConInstance.m_id) != excludedConstructionsIDs.end())
336  continue;
337 
338  // now process all subsequent construction instances
339  for (unsigned int j=i+1; j<numConInstances; ++j) {
340  const ConstructionInstance & c = conInstances[j]; // readability improvement
341  // skip construction instance, if already merged
342  if (canBeMergedTo[j] != -1)
343  continue;
344  // skip construction instance, if in excluded
345  if (excludedConstructionsIDs.find(c.m_id) != excludedConstructionsIDs.end())
346  continue;
347 
348  // now we compare currentConInstance with c
349  if (c.behavesLike(currentConInstance)) {
350  // can merge constructions, record merge-self marker
351  canBeMergedTo[i] = i;
352  // record merge-other marker
353  canBeMergedTo[j] = i;
354  }
355  }
356  }

358  struct t_mergeRecord {
359  int IDMergedFrom;
360  int IDMergedTo;
361  double originalCrossSection;
362  };
363  std::vector<t_mergeRecord> mergeRecord;
364 
365  // now process all possible merge constructions and create new copies at the end
366  for (unsigned int i=0; i<numConInstances; ++i) {
367  if (canBeMergedTo[i] == (int)i) {
368  canBeMergedTo[i] = conInstances.size(); // remember index of newly created construction instance
369  // create new instance
370  conInstances.push_back(conInstances[i]);
371  ConstructionInstance & c = conInstances.back();
372  // set new ID
373  c.m_id += 660000000; // TODO : improve this! - ID must not be become larger than 4,294,967,295
374  // record merge
375  t_mergeRecord r;
376  r.IDMergedTo = c.m_id;
377  r.IDMergedFrom = conInstances[i].m_id;
378  r.originalCrossSection = conInstances[i].m_para[ConstructionInstance::CP_AREA].value;
379  mergeRecord.push_back(r);
380  IBK::IBK_Message(IBK::FormatString("Merging construction id %1 -> id %2\n").arg(r.IDMergedFrom).arg(r.IDMergedTo), IBK::MSG_PROGRESS, FUNC_ID, IBK::VL_INFO);
381  }
382  }
383 
384  std::vector<ConstructionInstance> remainingOriginalConstructions;
385  remainingOriginalConstructions.reserve(numConInstances);
386 
387  // now do the actual merging
388  for (unsigned int i=0; i<numConInstances; ++i) {
389  // skip all constructions that cannot be merged
390  if (canBeMergedTo[i] == -1) {
391  remainingOriginalConstructions.push_back(conInstances[i]);
392  continue;
393  }
394  // skip constructions that were already merged (actually copied to new construction instances)
395  if (canBeMergedTo[i] >= (int)numConInstances)
396  continue;
397 
398  // lookup merge-to construction
399  int mergePartnerIndex = canBeMergedTo[i];
400  int newConInstanceIndex = canBeMergedTo[mergePartnerIndex];
401 
402  ConstructionInstance & c = conInstances[newConInstanceIndex];
403  // add own cross section area
404  c.m_para[ConstructionInstance::CP_AREA].value += conInstances[i].m_para[ConstructionInstance::CP_AREA].value;
405  // add own embedded objects
406  c.m_embeddedObjects.insert(c.m_embeddedObjects.end(), conInstances[i].m_embeddedObjects.begin(), conInstances[i].m_embeddedObjects.end());
407 
408  // record merge
409  t_mergeRecord r;
410  r.IDMergedTo = c.m_id;
411  r.IDMergedFrom = conInstances[i].m_id;
412  r.originalCrossSection = conInstances[i].m_para[ConstructionInstance::CP_AREA].value;
413  IBK::IBK_Message(IBK::FormatString("Merging construction id %1 -> id %2\n").arg(r.IDMergedFrom).arg(r.IDMergedTo), IBK::MSG_PROGRESS, FUNC_ID, IBK::VL_INFO);
414  mergeRecord.push_back(r);
415  }
416 
417  // now finally replace original constructionInstances
418  m_constructionInstances.clear();
419 
420  // first original construction instances (not merged)
421  for (const ConstructionInstance & c : remainingOriginalConstructions) {
422  m_constructionInstances[c.m_id] = c;
423  }
424  // now the newly created construction instances
425  for (unsigned int i=numConInstances; i<conInstances.size(); ++i) {
426  const ConstructionInstance & c = conInstances[i];
427  m_constructionInstances[c.m_id] = c;
428  }
429  if (numConInstances != conInstances.size())
430  IBK::IBK_Message( IBK::FormatString("Construction instance count reduced from %1 to %2\n").arg(numConInstances).arg(m_constructionInstances.size()),
431  IBK::MSG_PROGRESS, FUNC_ID, IBK::VL_STANDARD);
432 
433 #endif
