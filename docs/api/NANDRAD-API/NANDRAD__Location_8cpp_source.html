137 void Location::writeXML(TiXmlElement * parent) const {
138 
139  TiXmlComment::addComment(parent,
140  "Location of the building.");
141  TiXmlElement * e = new TiXmlElement("Location");
142  parent->LinkEndChild(e);
143 
144  // write parameters
145  for(unsigned int i = 0; i < NUM_LP; ++i) {
146  if(m_para[i].name.empty()) continue;
147  if(detailedOutput)
148  TiXmlComment::addComment(e,KeywordList::Description("Location::para_t",i));
149  TiXmlElement::appendIBKParameterElement(e,
150  m_para[i].name, m_para[i].IO_unit.name(), m_para[i].get_value());
151  }
152  // climate reference
153  if ( !m_climateFileName.str().empty() ) {
154 
155  if(detailedOutput)
156  TiXmlComment::addComment(e,"Reference to climate data set.");
157 
158  TiXmlElement * climateRef = new TiXmlElement("ClimateReference");
159  e->LinkEndChild( climateRef );
160 
161  if(!m_climateFileDisplayName.empty())
162  climateRef->SetAttribute("displayName", m_climateFileDisplayName);
163  TiXmlText * text = new TiXmlText( m_climateFileName.str() );
164  climateRef->LinkEndChild( text );
165 
166  }
167  // shading reference
168  if (!m_shadingFactorFileName.str().empty()) {
169 
170  if (detailedOutput)
171  TiXmlComment::addComment(e, "Reference to shading factor file.");
172 
173  TiXmlElement * shadingRef = new TiXmlElement("ShadingFactorReference");
174  e->LinkEndChild(shadingRef);
175 
176  if (!m_shadingFactorFileDisplayName.empty())
177  shadingRef->SetAttribute("displayName", m_shadingFactorFileDisplayName);
178  TiXmlText * text = new TiXmlText(m_shadingFactorFileName.str());
179  shadingRef->LinkEndChild(text);
180  }
181  // sensors
182  for (unsigned int i = 0; i < m_sensors.size(); ++i) {
183  // write sensor output
184  m_sensors[i].writeXML(e);
