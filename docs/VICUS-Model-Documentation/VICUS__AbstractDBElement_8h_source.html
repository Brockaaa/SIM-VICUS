39/*! Abstract base class of all database elements, needed to have generic DB table model implementations.
40 ID and display name are mandatory elements for each db element.
41*/
42class AbstractDBElement {
43public:
44 enum ComparisonResult {
45 /*! All parameters are identical (Caution: rounding errors!), except maybe the ID. */
46 Equal,
47 /*! All critical parameters are identical, yet some metadata (name etc.) may be different. */
48 OnlyMetaDataDiffers,
49 /*! Objects are completely different (at least one critical parameter differs). */
50 Different
51 };
52
53 /*! Default constructor. */
54 AbstractDBElement() = default;
55
56 /*! Initializing constructor. */
57 AbstractDBElement(unsigned int id, const IBK::MultiLanguageString & name) :
58 m_id(id),
59 m_displayName(name)
60 {}
61
62 /*! Virtual destructor, definition is in VICUS_Material.cpp. */
63 virtual ~AbstractDBElement();
64
65 virtual void readXML(const TiXmlElement * element) = 0;
66 virtual TiXmlElement * writeXML(TiXmlElement * parent) const = 0;
67
68 /*! Comparison of database element by content, without ID. */
69 virtual ComparisonResult equal(const AbstractDBElement * other) const = 0;
70
71 /*! Collects all pointers to child elements */
72 void collectChildren(std::set<const AbstractDBElement *> & allChildrenRefs) const;
73
74 /*! Collects all pointers to child elements which are local */
75 void collectLocalChildren(std::set<AbstractDBElement *> & localChildrenRefs) const;
76
77 /*! Source name for display purposes. */
78 const QString sourceName() const;
79
80 // *** Properties to be read/write in XML via code generator ***
81
82 /*! Unique ID of this DB element. */
83 unsigned int m_id = INVALID_ID;
84
85 /*! The description text for this DB element. */
86 IBK::MultiLanguageString m_displayName;
87
88 /*! False-color for this DB element (optional, may not be necessary for all DB elements). */
89 QColor m_color;
90
91
92 // *** Run time variables ***
93
94 /*! If true, this is a built-in DB element and cannot be modified/removed. */
95 bool m_builtIn = false;
96
97 /*! Determines wether this element is kept local in the current project or should be stored into the userDB (false). */
98 bool m_local = true;
99
100 /*! Pointer to elements which are referenced by this element */
101 std::set<AbstractDBElement*> m_childrenRefs;
102
103 /*! If true, this element is referenced in the current project */
104 mutable bool m_isReferenced = false;
105
106 /*! Holds error string in order to give users a tooltip in db dialog. */
107 mutable std::string m_errorMsg;
108
109};
110
111} // namespace VICUS
