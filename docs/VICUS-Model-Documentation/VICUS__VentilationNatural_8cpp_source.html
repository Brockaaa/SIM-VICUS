32bool VentilationNatural::isValid(const Database<Schedule> &scheduleDB) const {
33
34 std::string err;
35 if ( m_id == INVALID_ID )
36 return false;
37
38 if ( m_idSchedule == INVALID_ID )
39 return false;
40 else {
41 // check if schedule ID is existing and valid
42 const Schedule * sched = scheduleDB[m_idSchedule];
43 if (sched == nullptr)
44 return false;
45 if (!sched->isValid(err, true))
46 return false;
47 }
48
49 try {
50 // check whether a parameter with the correct unit has been set
51 m_para[P_AirChangeRate].checkedValue(VICUS::KeywordList::Keyword("VentilationNatural::para_t", P_AirChangeRate),
52 "1/h", "1/h", 0, true, 100, true, nullptr);
53 } catch (...) {
54 return false;
55 }
56
57 return true;
58}

60AbstractDBElement::ComparisonResult VentilationNatural::equal(const AbstractDBElement *other) const {
61 const VentilationNatural * otherVent = dynamic_cast<const VentilationNatural*>(other);
62 if (otherVent == nullptr)
63 return Different;
64
65 //first check critical data
66
67 //check parameters
68 for(unsigned int i=0; i<NUM_P; ++i){
69 if(m_para[i] != otherVent->m_para[i])
70 return Different;
71 }
72 if(m_idSchedule != otherVent->m_idSchedule)
73 return Different;
74
75 //check meta data
76
77 if(m_displayName != otherVent->m_displayName ||
78 m_color != otherVent->m_color ||
79 m_dataSource != otherVent->m_dataSource ||
80 m_notes != otherVent->m_notes)
81 return OnlyMetaDataDiffers;
82
83 return Equal;
84}
