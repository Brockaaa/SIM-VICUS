318bool IdealSurfaceHeatingCoolingModelGenerator::calculateSupplyTemperature(const std::vector<double> &supplySetpoints,const std::vector<double> &outdoorSetpoints,
319 const std::vector<double> &outdoorTemperatureSpline, std::vector<double> &supplyTemperature){
320 if(supplySetpoints.size() != outdoorSetpoints.size() && supplySetpoints.size() == 4)
321 return false;
322
323
324 double lowerSupplyHeatLimit = std::min<double>(supplySetpoints[0], supplySetpoints[1]);
325 double upperSupplyHeatLimit = std::max<double>(supplySetpoints[0], supplySetpoints[1]);
326 double lowerSupplyCoolLimit = std::min<double>(supplySetpoints[2], supplySetpoints[3]);
327
328 double lowerOutHeatLimit = std::min<double>(outdoorSetpoints[0], outdoorSetpoints[1]);
329 double upperOutHeatLimit = std::max<double>(outdoorSetpoints[0], outdoorSetpoints[1]);
330 double lowerOutCoolLimit = std::min<double>(outdoorSetpoints[2], outdoorSetpoints[3]);
331 double upperOutCoolLimit = std::max<double>(outdoorSetpoints[2], outdoorSetpoints[3]);
332
333 if(upperOutHeatLimit > lowerOutCoolLimit)
334 return false;
335
336 supplyTemperature.clear();
337
338 double deltaX = outdoorSetpoints[0] - outdoorSetpoints[1];
339 double mHeat = 0;
340 if(deltaX != 0.)
341 mHeat = (supplySetpoints[0] - supplySetpoints[1]) / deltaX;
342 double nHeat = supplySetpoints[0] - mHeat * outdoorSetpoints[0];
343
344 deltaX = outdoorSetpoints[2] - outdoorSetpoints[3];
345 double mCool = 0;
346 if(deltaX != 0.)
347 mCool = (supplySetpoints[2] - supplySetpoints[3]) / deltaX;
348 double nCool = supplySetpoints[2] - mCool * outdoorSetpoints[2];
349
350 for(unsigned int i=0; i<outdoorTemperatureSpline.size(); ++i){
351 double tOut = outdoorTemperatureSpline[i];
352 double tSupply;
353 if(tOut <= lowerOutHeatLimit)
354 tSupply = upperSupplyHeatLimit;
355 else if(tOut > lowerOutHeatLimit && tOut <= upperOutHeatLimit)
356 tSupply = mHeat * tOut + nHeat;
357 else if(tOut > upperOutHeatLimit && tOut < lowerOutCoolLimit)
358 tSupply = lowerSupplyHeatLimit;
359 else if(tOut >= lowerOutCoolLimit && tOut < upperOutCoolLimit)
360 tSupply = mCool * tOut + nCool;
361 else
362 tSupply = lowerSupplyCoolLimit;
363 supplyTemperature.push_back(tSupply);
364 }
365 return true;
366}

512bool Project::generateShadingFactorsFile(const std::map<unsigned int, unsigned int> &surfaceIdsVicusToNandrad,
513 const IBK::Path & projectFilePath, IBK::Path & shadingFactorFilePath) const
514{
515 FUNCID(Project::generateShadingFactorsFile);
516 IBK::Path basePath(projectFilePath.withoutExtension() + "_shadingFactors");
517
518 // search for available shading file, and if there are several matching shading files, issue an error
519 bool found = false;
520 IBK::Path vicusShadingFilePath;
521 for (const char * const ext : {"tsv", "d6o", "d6b"}) {
522 IBK::Path fullPath(basePath.str() + "." + ext);
523 if (fullPath.exists()) {
524 if (found) {
525 IBK::IBK_Message(IBK::FormatString("Found shading factor file '%1', "
526 "but another matching file '%2' exists already!")
527 .arg(fullPath).arg(vicusShadingFilePath), IBK::MSG_ERROR);
528 return false;
529 }
530 found = true;
531 vicusShadingFilePath = fullPath;
532 }
533 }
534
535 if (!found) {
536 IBK::IBK_Message(IBK::FormatString("No shading file exported, expected a file '%1'[.tsv, .d6o, .d6b]. Running simulation without shading factors.")
537 .arg(basePath), IBK::MSG_PROGRESS, FUNC_ID);
538 return true;
539 }
540 std::string ext = vicusShadingFilePath.extension();
541 // compose path to NANDRAD shading factor file
542 shadingFactorFilePath = vicusShadingFilePath.withoutExtension() + "Nandrad." + ext;
543
544 // we only need to convert the shading factors file, if the VICUS file has a newer time stamp
545 // Note: for additional security, we could store the original file's hash in the DataIO container and compare that as well
546 if (shadingFactorFilePath.exists() && vicusShadingFilePath.lastWriteTime() < shadingFactorFilePath.lastWriteTime()) {
547 IBK::IBK_Message(IBK::FormatString("NANDRAD shading factor file '%1' newer than original VICUS shading factor file '%2'. Shading file generation skipped.\n")
548 .arg(shadingFactorFilePath).arg(vicusShadingFilePath), IBK::MSG_PROGRESS);
549 return true;
550 }
551
552 // depending on the file extensions, we first parse the data into memory and then dump it back out into a suitable
553 // NANDRAD input file
554
555 IBK::IBK_Message(IBK::FormatString("Reading shading factor file '%1'\n").arg(vicusShadingFilePath), IBK::MSG_PROGRESS);
556 if (ext == "tsv") {
557 IBK::CSVReader csv;
558 try {
559 csv.read(vicusShadingFilePath, false, true); // headerOnly=false, extractUnits=true
560 // general error handling
561 if (csv.m_nColumns == 0 || csv.m_nRows == 0)
562 throw IBK::Exception("Missing data in shading factor file.", FUNC_ID);
563 if (csv.m_captions[0] != "Time")
564 throw IBK::Exception("Invalid caption of time column.", FUNC_ID);
565 IBK::Unit u;
566 try {
567 u = IBK::Unit(csv.m_units[0]);
568 }
569 catch (...) { /* do nothing */ }
570 if (u.base_id() != IBK_UNIT_ID_SECONDS)
571 throw IBK::Exception("Invalid unit of time column, a valid time unit is required.", FUNC_ID);
572 } catch (IBK::Exception & ex) {
573 ex.writeMsgStackToError();
574 IBK::IBK_Message(IBK::FormatString("Error reading shading factors file '%1'.").arg(vicusShadingFilePath), IBK::MSG_ERROR);
575 return false;
576 }
577
578 // we now create mapping vector, that relates new column index to original column index in the csv file
579 // and also we populate the nums vector; this is needed because the original shading factors file
580 // contains additional columns like "Azimuth" and "Altitude", which we need to skip
581 std::vector<unsigned int> colIndices;
582
583 // process all columns, but skip the mandatory time column
584 bool haveError = false;
585 std::string headerLine = "Time [" + csv.m_units[0] + "]";
586 colIndices.push_back(0); // we always keep column 0 with the time
587 for (unsigned int i=1; i<csv.m_captions.size(); ++i) {
588 std::string cap = csv.m_captions[i];
589 // skip Aziumuth and Altitude
590 if (cap == "Azimuth" || cap == "Altitude")
591 continue;
592 unsigned int id = VICUS::INVALID_ID;
593 std::string displayName;
594 try {
595 // when reading tsv-files, the headers have format "<id>" or "<id> '<display name>'"
596 std::string::size_type pos = cap.find_first_of(" '");
597 if (pos != std::string::npos)
598 cap = cap.substr(0, pos);
599 id = IBK::string2val<unsigned int>(cap); // might throw
600 // lookup displayname of referenced VICUS component instances
601 const VICUS::Surface * surf = surfaceByID(id);
602 const VICUS::SubSurface * subSurf = subSurfaceByID(id);
603 if (surf == nullptr && subSurf == nullptr) // can happen, if shading file is out of date and still references an invalid (now missing) surface ID
604 throw IBK::Exception("Invalid surface/sub-surface ID", FUNC_ID);
605 if (surf != nullptr)
606 displayName = surf->m_displayName.toStdString();
607 else
608 displayName = subSurf->m_displayName.toStdString();
609 } catch (...) {
610 IBK::IBK_Message( IBK::FormatString(" Invalid surface ID '%1' in shading factor file.").arg(cap), IBK::MSG_ERROR);
611 haveError = true;
612 continue;
613 }
614 // lookup id in id-map
615 std::map<unsigned int, unsigned int>::const_iterator nandradSurfIt = surfaceIdsVicusToNandrad.find(id);
616 if (nandradSurfIt == surfaceIdsVicusToNandrad.end()) {
617 IBK::IBK_Message( IBK::FormatString(" Invalid/unknown surface ID '%1' in shading factor file.").arg(id), IBK::MSG_ERROR);
618 haveError = true;
619 continue;
620 }
621 colIndices.push_back(i);
622
623 // store NANDRAD ID in file header line followed by display name; NANDRAD parses the headers and expects first an ID number in the header label
624 headerLine += "\t" + IBK::val2string(nandradSurfIt->second) + " '" + displayName + "' [" + csv.m_units[i] + "]";
625 }
626
627 // error check
628 if (haveError) {
629 // if we got errors we can not assign shading factors
630 IBK::IBK_Message(IBK::FormatString("There are invalid column headers!"), IBK::MSG_ERROR, FUNC_ID, IBK::VL_STANDARD);
631 return false;
632 }
633
634 // write file
635 std::ofstream out;
636 if (!IBK::open_ofstream(out, shadingFactorFilePath) ) {
637 IBK::IBK_Message(IBK::FormatString("Error writing shading file '%1'.").arg(shadingFactorFilePath), IBK::MSG_ERROR, FUNC_ID);
638 return false;
639 }
640
641 out << headerLine << '\n';
642
643 for (unsigned int row=0; row<csv.m_nRows; ++row) {
644 for (unsigned int col=0; col<colIndices.size(); ++col) {
645 if (col != 0)
646 out << "\t";
647 unsigned int originalColIndex = colIndices[col];
648 out << csv.m_values[row][originalColIndex];
649 }
650 out << "\n";
651 }
652
653 out.close();
654 }
655 else {
656 // read original DataIO file
657 DATAIO::DataIO shadingFile;
658 try {
659 shadingFile.read(vicusShadingFilePath, false);
660 // force conversion from strings to doubles
661 for (unsigned int i=0; i<shadingFile.m_timepoints.size(); ++i)
662 shadingFile.data(i);
663 } catch (IBK::Exception & ex) {
664 ex.writeMsgStackToError();
665 IBK::IBK_Message(IBK::FormatString("Error reading shading factors file '%1'.").arg(vicusShadingFilePath), IBK::MSG_ERROR);
666 return false;
667 }
668
669 // we only need to process the m_nums line, substitute the IDs and also generate the quantities line
670 std::string newQuantities;
671 std::vector<unsigned int> newNums;
672 for (unsigned int i=0; i<shadingFile.m_nums.size(); ++i) {
673 unsigned int id = shadingFile.m_nums[i];
674 std::map<unsigned int, unsigned int>::const_iterator nandradSurfIt = surfaceIdsVicusToNandrad.find(id);
675 if (nandradSurfIt == surfaceIdsVicusToNandrad.end()) {
676 IBK::IBK_Message( IBK::FormatString("Invalid/unknown surface ID '%1' in shading factor file.").arg(id), IBK::MSG_ERROR);
677 return false;
678 }
679 const VICUS::Surface * surf = surfaceByID(id);
680 const VICUS::SubSurface * subSurf = subSurfaceByID(id);
681 if (surf == nullptr && subSurf == nullptr) { // can happen, if shading file is out of date and still references an invalid (now missing) surface ID
682 IBK::IBK_Message( IBK::FormatString("Invalid/unknown surface ID '%1' in shading factor file.").arg(id), IBK::MSG_ERROR);
683 return false;
684 }
685 std::string displayName;
686 if (surf != nullptr)
687 displayName = surf->m_displayName.toStdString();
688 else
689 displayName = subSurf->m_displayName.toStdString();
690 // now substitute the ID and compose the quantity string
691 if (i != 0)
692 newQuantities += " | ";
693 if (!displayName.empty())
694 newQuantities += displayName;
695 else
696 newQuantities += IBK::val2string(nandradSurfIt->second);
697 newNums.push_back(nandradSurfIt->second);
698 }
699
700 shadingFile.m_type = DATAIO::DataIO::T_REFERENCE;
701 shadingFile.m_valueUnit = "---"; // unitless (convertible to 1, %)
702 shadingFile.m_projectFileName = projectFilePath.str();
703 shadingFile.m_quantityKeyword = "ShadingFactor";
704 shadingFile.m_spaceType = DATAIO::DataIO::ST_SINGLE;
705 shadingFile.m_timeType = DATAIO::DataIO::TT_NONE;
706 // shadingFile.m_startYear = 2007; // TODO : shouldn't we insert the correct start year here?
707 shadingFile.m_filename = shadingFactorFilePath;
708 shadingFile.m_nums = newNums;
709 shadingFile.m_quantity = newQuantities;
710
711 // now write DataIO file
712 shadingFile.write();
713 }
714 IBK::IBK_Message(IBK::FormatString("Created shading factors data file '%1' in NANDRAD format\n").arg(shadingFactorFilePath), IBK::MSG_PROGRESS, FUNC_ID);
715
716 return true;
717}
