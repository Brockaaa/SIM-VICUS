41const Object * Object::findChild(unsigned int ID) const {
42 if (m_id == ID)
43 return this;
44 // search all children
45 for (Object * o : m_children) {
46 const Object * ob = o->findChild(ID);
47 if (ob != nullptr)
48 return ob;
49 }
50 return nullptr; // not found
51}
