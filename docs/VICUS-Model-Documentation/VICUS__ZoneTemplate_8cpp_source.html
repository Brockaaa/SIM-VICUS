38bool ZoneTemplate::isValid(const Database<InternalLoad> & intLoadDB,
39 const Database<ZoneControlThermostat> & thermostatDB,
40 const Database<Schedule> &schedulesDB,
41 const Database<Infiltration> & infiltraionDB,
42 const Database<VentilationNatural> &ventilationDB,
43 const Database<ZoneIdealHeatingCooling> &idealHeatingCoolingDB,
44 const Database<ZoneControlShading> &ctrlShadingDB,
45 const Database<ZoneControlNaturalVentilation> &ctrlNatVentDB) const
46{
47
48 if (m_id == INVALID_ID) {
49 m_errorMsg = "ID of zone template is invalid.";
50 return false;
51 }
52
53 // test all referenced sub-template types
54 unsigned int count = 0;
55 for (unsigned int i=0; i<NUM_ST; ++i) {
56 unsigned int id = m_idReferences[i];
57 if (id == INVALID_ID)
58 continue;
59
60 ++count;
61
62 switch ((SubTemplateType)i) {
63 case ZoneTemplate::ST_IntLoadPerson:
64 case ZoneTemplate::ST_IntLoadLighting:
65 case ZoneTemplate::ST_IntLoadOther:
66 case ZoneTemplate::ST_IntLoadEquipment: {
67 const InternalLoad *intLoad = intLoadDB[id];
68 if (intLoad == nullptr) {
69 m_errorMsg = "Internal load with ID '" + std::to_string(id) + "' does not exist.";
70 return false;
71 }
72 if (!intLoad->isValid(schedulesDB)) {
73 m_errorMsg = IBK::FormatString("Internal load '%1' is invalid.").arg(intLoad->m_displayName).str();
74 return false;
75 }
76 }
77 break;
78
79 case ZoneTemplate::ST_ControlThermostat: {
80 const ZoneControlThermostat *thermo = thermostatDB[id];
81 if (thermo == nullptr) {
82 IBK::FormatString("Zone control thermostat with ID '%1' does not exist.").arg(id).str();
83 return false;
84 }
85 if (!thermo->isValid(schedulesDB)) {
86 m_errorMsg = IBK::FormatString("Zone control thermostat '%1' is invalid.").arg(thermo->m_displayName).str();
87 return false;
88 }
89 }
90 break;
91
92 case ZoneTemplate::ST_Infiltration: {
93 const Infiltration *inf = infiltraionDB[id];
94 if (inf == nullptr) {
95 IBK::FormatString("Infiltration model with ID '%1' does not exist.").arg(id).str();
96 return false;
97 }
98 if (!inf->isValid()) {
99 m_errorMsg = IBK::FormatString("Infiltration model '%1' is invalid.").arg(inf->m_displayName).str();
100 return false;
101 }
102 }
103 break;
104
105 case ZoneTemplate::ST_VentilationNatural: {
106 const VentilationNatural *venti = ventilationDB[id];
107 if (venti == nullptr) {
108 IBK::FormatString("Natural ventilation model with ID '%1' does not exist.").arg(id).str();
109 return false;
110 }
111 if (!venti->isValid(schedulesDB)) {
112 m_errorMsg = IBK::FormatString("Natural ventilation model '%1' is invalid.").arg(venti->m_displayName).str();
113 return false;
114 }
115 }
116 break;
117
118 case ZoneTemplate::ST_IdealHeatingCooling: {
119 const ZoneIdealHeatingCooling *ideal = idealHeatingCoolingDB[id];
120 if (ideal == nullptr) {
121 IBK::FormatString("Ideal heating model with ID '%1' does not exist.").arg(id).str();
122 return false;
123 }
124 if (!ideal->isValid()) {
125 m_errorMsg = IBK::FormatString("Ideal heating model '%1' is invalid.").arg(ideal->m_displayName).str();
126 return false;
127 }
128 // ideal heating/cooling always requires a valid thermostat
129 const ZoneControlThermostat *thermo = thermostatDB[m_idReferences[ST_ControlThermostat]];
130 if (thermo == nullptr) {
131 IBK::FormatString("Zone control thermostat with ID '%1' does not exist.").arg(m_idReferences[ST_ControlThermostat]).str();
132 return false;
133 }
134 if (!thermo->isValid(schedulesDB)) {
135 m_errorMsg = IBK::FormatString("Zone control thermostat '%1' is invalid.").arg(thermo->m_displayName).str();
136 return false;
137 }
138 }
139 break;
140
141 case ZoneTemplate::ST_ControlVentilationNatural: {
142 const ZoneControlNaturalVentilation *ctrl = ctrlNatVentDB[id];
143 if(ctrl == nullptr) {
144 IBK::FormatString("Zone natural ventilation control model with ID '%1' does not exist.").arg(id).str();
145 return false;
146 }
147 if (!ctrl->isValid(schedulesDB)) {
148 m_errorMsg = IBK::FormatString("Zone natural ventilation control model '%1' is invalid.").arg(ctrl->m_displayName).str();
149 return false;
150 }
151 // Zone natural ventilation control always requires a valid ventilation model
152 const VentilationNatural *vent = ventilationDB[m_idReferences[ST_VentilationNatural]];
153 if (vent == nullptr) {
154 IBK::FormatString("Ventilation model with ID '%1' does not exist.").arg(m_idReferences[ST_ControlThermostat]).str();
155 return false;
156 }
157 if (!vent->isValid(schedulesDB)) {
158 m_errorMsg = IBK::FormatString("Ventilation '%1' is invalid.").arg(vent->m_displayName).str();
159 return false;
160 }
161 }
162 break;
163
164 case ZoneTemplate::ST_ControlShading: {
165 const ZoneControlShading *ctrl = ctrlShadingDB[id];
166 if(ctrl == nullptr) {
167 IBK::FormatString("Shading control model with ID '%1' does not exist.").arg(id).str();
168 return false;
169 }
170 if (!ctrl->isValid()) {
171 m_errorMsg = IBK::FormatString("Shading control model '%1' is invalid.").arg(ctrl->m_displayName).str();
172 return false;
173 }
174 }
175 break;
176
177 case ZoneTemplate::NUM_ST: ; // just to make compiler happy
178 }
179 }
180 if (count == 0)
181 return false; // must have at least one subtype
182
183 return true;
184}

197ZoneTemplate::SubTemplateType ZoneTemplate::usedReference(unsigned int index) const {
198 int count = 0;
199 int i=0;
200 // Example : index = 0 and we have the first id set
201 // loop 1: i = 0, count = 1 -> return i=0
202 // Example 2: index = 1 and we have the first and third id set:
203 // loop 1: i = 0, count = 1
204 // loop 2: i = 1, count = 1
205 // loop 3: i = 2, count = 2 -> return i=1
206 for (; i<NUM_ST; ++i) {
207 // increase count for each used id reference
208 if (m_idReferences[i] != VICUS::INVALID_ID)
209 ++count;
210 if (count > (int)index)
211 break;
212 }
213 return (ZoneTemplate::SubTemplateType)i; // if index > number of used references, we return NUM_ST here
214}

217AbstractDBElement::ComparisonResult ZoneTemplate::equal(const AbstractDBElement *other) const {
218 const ZoneTemplate * otherEPD = dynamic_cast<const ZoneTemplate*>(other);
219 if (otherEPD == nullptr)
220 return Different;
221
222 // check parameters
223 for (unsigned int i=0; i<NUM_ST; ++i){
224 if (m_idReferences[i] != otherEPD->m_idReferences[i])
225 return Different;
226 }
227
228 // check meta data
229 if (m_displayName != otherEPD->m_displayName ||
230 m_dataSource != otherEPD->m_dataSource ||
231 m_notes != otherEPD->m_notes)
232 return OnlyMetaDataDiffers;
233
234 return Equal;
235}
