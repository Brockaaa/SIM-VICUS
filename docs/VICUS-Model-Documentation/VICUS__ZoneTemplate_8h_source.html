48/*! Container class that references one or more zone control/sub-templates that define zone behavior/controls. */
49class ZoneTemplate : public AbstractDBElement {
50public:
51
52 /*! Types used to identify individual sub-templates.
53 \note Do not change the order!
54 First have a look to Project::generateBuildingProjectData()
55 */
56 enum SubTemplateType {
57 ST_IntLoadPerson, // Keyword: IntLoadPerson <#FFBB55> 'Person loads'
58 ST_IntLoadEquipment, // Keyword: IntLoadEquipment <#AA2222> 'Equipment loads'
59 ST_IntLoadLighting, // Keyword: IntLoadLighting <#FFEECC> 'Lighting loads'
60 ST_IntLoadOther, // Keyword: IntLoadOther <#602222> 'Other internal loads'
61 ST_ControlThermostat, // Keyword: ControlThermostat <#E00010> 'Thermostat control'
62 ST_ControlShading, // Keyword: ControlShading <#A2F0F4> 'Shading control'
63 ST_ControlVentilationNatural, // Keyword: ControlVentilationNatural <#00A000> 'Natural ventilation control'
64 ST_Infiltration, // Keyword: Infiltration <#A0B0FF> 'Infiltration loads'
65 ST_VentilationNatural, // Keyword: NaturalVentilation <#22EE22> 'Natural ventilation loads
66 ST_IdealHeatingCooling, // Keyword: IdealHeatingCooling <#B08000> 'Heating/cooling loads'
67 NUM_ST
68 };

93 \return Returns type of corresponding id reference or NUM_ST, if index is larger than the number of non-empty
