33void NetworkNode::collectConnectedEdges(std::set<const NetworkNode *> & connectedNodes,
34 std::set<const NetworkEdge *> & connectedEdge) const {
35 // store ourselves as connected
36 connectedNodes.insert(this);
37 // now ask connected elements to collect their nodes
38 for (const NetworkEdge * e : m_edges) {
39 // only process edges that are not yet collected
40 if (connectedEdge.find(e) == connectedEdge.end())
41 e->collectConnectedNodes(connectedNodes, connectedEdge);
42 }
43}

46void NetworkNode::setInletOutletNode(std::set<const NetworkNode *> &visitedNodes, std::vector<const NetworkEdge *> &orderedEdges) const {
47 // store ourselves as connected
48 visitedNodes.insert(this);
49 // now ask connected elements to collect their nodes
50 for (NetworkEdge * e : m_edges) {
51 // only process edges that are not yet collected
52 if (std::find(orderedEdges.begin(), orderedEdges.end(), e) == orderedEdges.end()){
53 e->m_idNodeInlet = m_id;
54 e->m_idNodeOutlet = e->neighbourNode(m_id);
55 e->setInletOutletNode(visitedNodes, orderedEdges);
56 }
57 }

61void NetworkNode::updateIsDeadEnd(){
62 // buildings or sources are never dead ends
63 if (m_type == NT_SubStation || m_type == NT_Source){
64 m_isDeadEnd = false;
65 return;
66 }
67 // check all neighbouring nodes. If only 1 or less are not dead ends: this is a dead end
68 unsigned numNonDeadEndNeighbours = 0;
69 for (NetworkEdge *e: m_edges){
70 if (!e->neighbourNode(this)->m_isDeadEnd)
71 ++numNonDeadEndNeighbours;
72 }
73 m_isDeadEnd = numNonDeadEndNeighbours <= 1;
74}

87const NetworkNode * NetworkNode::findNextNonRedundantNode(std::set<unsigned> & redundantNodes, double & distance, const NetworkEdge *edgeToVisit) const
88{
89 distance += edgeToVisit->length();
90 NetworkNode * nextNode = edgeToVisit->neighbourNode(this);
91 if (!nextNode->isRedundant())
92 return nextNode;
93 redundantNodes.insert(nextNode->m_id);
94 return nextNode->findNextNonRedundantNode(redundantNodes, distance, nextNode->neighborEdge(edgeToVisit));
95}

98void NetworkNode::findRedundantNodes(std::set<unsigned> & redundantNodes, std::set<const NetworkEdge *> & visitedEdges) const
99{
100 // redundant nodes have exactly 2 connected edges
101 if (this->isRedundant()){
102 redundantNodes.insert(this->m_id);
103 }
104 for (const NetworkEdge * e: m_edges){
105 // remember visited edges and dont visit them again
106 if (visitedEdges.find(e) == visitedEdges.end()){
107 visitedEdges.insert(e);
108 if (e->nodeId1() == this->m_id)
109 e->m_node2->findRedundantNodes(redundantNodes, visitedEdges);
110 else
111 e->m_node1->findRedundantNodes(redundantNodes, visitedEdges);
112 }
113 }
114}

117bool NetworkNode::findPathToSource(std::set<NetworkEdge*> &path, std::set<NetworkEdge*> &visitedEdges, std::set<unsigned> &visitedNodes){
118
119 if (visitedNodes.find(m_id) != visitedNodes.end())
120 return false;
121 visitedNodes.insert(m_id);
122
123 // if reached end of graph
124 if (m_edges.size()==1 && visitedEdges.find(m_edges[0]) != visitedEdges.end()){
125 if (m_type==NodeType::NT_Source){
126 return true;
127 }
128 return false;
129 }
130
131 for (NetworkEdge *e: m_edges){
132 if (visitedEdges.find(e) == visitedEdges.end()){
133 visitedEdges.insert(e);
134 if (e->neighbourNode(this)->findPathToSource(path, visitedEdges, visitedNodes)){
135 path.insert(e);
136 return true;
137 }
138 }
139 }
140 return false;
141}

144void NetworkNode::updateNeighbourDistances() const{
145 // calculate alternative distance to neighbour.
146 // If it is shorter than current one: set it as its current distance
147 for (const NetworkEdge *e :m_edges){
148 NetworkNode * neighbour = e->neighbourNode(this);
149 double alternativeDistance = m_distanceToStart + e->length();
150 if (alternativeDistance < neighbour->m_distanceToStart){
151 neighbour->m_distanceToStart = alternativeDistance;
152 neighbour->m_predecessor = this;
153 }
154 }
155}

158void NetworkNode::pathToNull(std::vector<NetworkEdge*> &path) const{
159 if (m_predecessor == nullptr)
160 return;
161 for (NetworkEdge * e: m_edges){
162 if (e->neighbourNode(this) == m_predecessor){
163 path.push_back(e);
164 e->neighbourNode(this)->pathToNull(path);

169double NetworkNode::adjacentHeatingDemand(std::set<NetworkEdge *> visitedEdges){
170 for (NetworkEdge *e: m_edges){
171 if (visitedEdges.find(e)==visitedEdges.end()){
172 visitedEdges.insert(e);
173 if (e->m_nominalHeatingDemand>0)
174 return e->m_nominalHeatingDemand;
175 e->neighbourNode(this)->adjacentHeatingDemand(visitedEdges);
176 }
177 }
178 return 0;
179}
