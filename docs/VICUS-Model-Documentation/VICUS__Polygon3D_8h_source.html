35/*! Class Polygon3D stores a planar polygon in 3D space.
36 This class merely wraps IBKMK::Polygon3D and provides read/write functionality.
37*/
38class Polygon3D : public IBKMK::Polygon3D {
39 VICUS_READWRITE_PRIVATE
40public:
41
42 // *** PUBLIC MEMBER FUNCTIONS ***
43
44 Polygon3D() = default;
45
46 /*! Conversion constructor */
47 explicit Polygon3D(const IBKMK::Polygon3D & poly3D) :
48 IBKMK::Polygon3D(poly3D)
49 {
50 }
51
52 /*! Initializing constructor.
53 Vertexes a, b and c must be given in counter-clockwise order, so that (b-a) x (c-a) yields the normal vector of the plane.
54 If t is Polygon2D::T_Rectangle, vertex c actually corresponds to vertex d of the rectangle, and vertex c is computed
55 internally.
56 */
57 Polygon3D(IBKMK::Polygon2D::type_t t, const IBKMK::Vector3D & a, const IBKMK::Vector3D & b, const IBKMK::Vector3D & c) :
58 IBKMK::Polygon3D(t, a, b, c)
59 {}
60
61 /*! Constructs a polygon from 2D polygon with normal vector, xaxis and offset. */
62 Polygon3D(const IBKMK::Polygon2D & p2d, const IBKMK::Vector3D & offset,
63 const IBKMK::Vector3D & normal, const IBKMK::Vector3D & xAxis) :
64 IBKMK::Polygon3D(p2d, offset, normal, xAxis)
65 {}
66
67 /*! Constructs a polygon from a 3D polyline (which might be invalid in any number of ways).
68 The normal vector will be deduced from rotation direction of the polygon, and the x-axis vector will be the vector
69 from first to second vertex at a suitable (automatically selected) vertex of the polygon.
70
71 \note Once all collinear points have been removed the offset point will be the first vertex of the polygon. Use offset()
72 to retrieve the offset.
73 */
74 Polygon3D(const std::vector<IBKMK::Vector3D> & vertexes) :
75 IBKMK::Polygon3D(vertexes)
76 {}
77
78 void readXML(const TiXmlElement * element);
79 TiXmlElement * writeXML(TiXmlElement * parent) const;
80
81 VICUS_COMP(Polygon3D)
82};
