97unsigned int Network::addNode(unsigned int preferedId, const IBKMK::Vector3D &v, const NetworkNode::NodeType type, const bool consistentCoordinates) {
98
99 // if there is an existing node with identical coordinates, return its id and dont add a new one
100 if (consistentCoordinates){
101 for (NetworkNode &n: m_nodes){
102 if (n.m_position.distanceTo(v) < GeometricResolution){
103 if(n.m_type != type){
104 n.m_type = type;
105 }
106 return n.m_id;
107 }
108 }
109 }
110 // else add new node
111 m_nodes.push_back(NetworkNode(preferedId, type, v));
112 updateNodeEdgeConnectionPointers();
113 return preferedId;
114}

139void Network::updateNodeEdgeConnectionPointers() {
140 FUNCID(Network::updateNodeEdgeConnectionPointers);
141
142 // resolve all node and edge pointers
143 // check for valid node IDs and throws exceptions, if any check fails
144
145 // first clear edge pointers in all nodes
146 for (NetworkNode & n : m_nodes)
147 n.m_edges.clear();
148
149 // loop over all edges
150 for (NetworkEdge & e : m_edges) {
151 // an edge must have 2 valid node IDs
152 if (e.nodeId1() == VICUS::INVALID_ID || e.nodeId2() == VICUS::INVALID_ID)
153 throw IBK::Exception(IBK::FormatString("Network edge #%1 does not have to valid node IDs.").arg(e.m_id), FUNC_ID);
154
155 // an edge must not reference the same node twice
156 if (e.nodeId1() == e.nodeId2())
157 throw IBK::Exception(IBK::FormatString("Network edge #%1 referenced the same node #%2 on both sides.")
158 .arg(e.m_id).arg(e.nodeId1()), FUNC_ID);
159
160 // store pointers to connected nodes
161 try {
162 e.m_node1 = nodeById(e.nodeId1());
163 } catch (...) {
164 throw IBK::Exception(IBK::FormatString("Network edge #%1 references invalid node #%2.")
165 .arg(e.m_id).arg(e.nodeId1()), FUNC_ID);
166 }
167 try {
168 e.m_node2 = nodeById(e.nodeId2());
169 } catch (...) {
170 throw IBK::Exception(IBK::FormatString("Network edge #%1 references invalid node #%2.")
171 .arg(e.m_id).arg(e.nodeId2()), FUNC_ID);
172 }
173
174 // now also store pointer to this edge into connected nodes
175 e.m_node1->m_edges.push_back(&e);
176 e.m_node2->m_edges.push_back(&e);
177 }
178
179 // finally, also update all VICUS::Object data members
180 m_children.clear();
181 for (NetworkEdge & e : m_edges) {
182 e.m_parent = this;
183 m_children.push_back(&e);
184 }
185 for (NetworkNode & n : m_nodes) {
186 n.m_parent = this;
187 m_children.push_back(&n);
188 }
189}

192void Network::updateVisualizationRadius(const VICUS::Database<VICUS::NetworkPipe> & pipeDB) {
193
194 // process all edges and update their display radius
195 for (VICUS::NetworkEdge & e : m_edges) {
196 double radius = 0.5;
197 if (e.m_idPipe != VICUS::INVALID_ID){
198 const VICUS::NetworkPipe * pipe = pipeDB[e.m_idPipe];
199 if (pipe != nullptr)
200 radius *= pipe->m_para[VICUS::NetworkPipe::P_DiameterOutside].value * m_scaleEdges;
201 }
202 e.m_visualizationRadius = radius;
203 }
204
205 // process all nodes and update their display radius, default radius = 1 cm
206 for (const VICUS::NetworkNode & no : m_nodes) {
207 double radius = 1 * m_scaleNodes / 100;
208 switch (no.m_type) {
209 case NetworkNode::NT_SubStation: {
210 // scale node by heating demand - 1 mm / 1000 W; 4800 W -> 48 * 0.01 = radius = 0.48
211 if (no.m_maxHeatingDemand.value > 0)
212 radius *= no.m_maxHeatingDemand.value / 1000;
213 } break;
214 case NetworkNode::NT_Source:
215 case NetworkNode::NT_Mixer: {
216 // if we have connected pipes, compute max radius of adjacent pipes (our node should be larger than the pipes)
217 for (const VICUS::NetworkEdge * edge: no.m_edges)
218 radius = std::max(radius, edge->m_visualizationRadius*1.2); // enlarge by 20 % over edge diameter
219 } break;
220 default:;
221 }
222 // store values
223 no.m_visualizationRadius = radius;
224 }
225}

314bool Network::checkConnectedGraph() const {
315 if (m_edges.size()==0 || m_nodes.size()==0)
316 return false;
317
318 std::set<const NetworkNode*> connectedNodes;
319 std::set<const NetworkEdge*> connectedEdge;
320
321 // start by any node
322 const NetworkEdge * start = &m_edges[0];
323
324 // ask edge to check its nodes
325 start->collectConnectedNodes(connectedNodes, connectedEdge);
326
327 return (connectedEdge.size() && m_edges.size() && connectedNodes.size() == m_nodes.size());
328}

332void Network::generateIntersections(unsigned int nextUnusedId, std::vector<unsigned int> &addedNodes, std::vector<unsigned int> &addedEdges){
333
334 bool foundIntersection = true;
335
336 while (foundIntersection) {
337 foundIntersection = false;
338 for (unsigned i1=0; i1<m_edges.size(); ++i1) {
339 for (unsigned i2=i1+1; i2<m_edges.size(); ++i2) {
340
341 // calculate intersection
342 NetworkLine l1 = NetworkLine(m_edges[i1]);
343 NetworkLine l2 = NetworkLine(m_edges[i2]);
344 IBKMK::Vector3D ps;
345 l1.intersection(l2, ps);
346
347 // if it is within both lines: add node and edges, adapt exisiting nodes
348 if (l1.containsPoint(ps) && l2.containsPoint(ps)){
349 unsigned nInter = addNode(++nextUnusedId, ps, NetworkNode::NT_Mixer);
350 addedNodes.push_back(nInter);
351 addEdge(++nextUnusedId, nInter, m_edges[i1].nodeId1(), true, m_edges[i1].m_idPipe);
352 addedEdges.push_back(nextUnusedId);
353 addEdge(++nextUnusedId, nInter, m_edges[i2].nodeId1(), true, m_edges[i2].m_idPipe);
354 addedEdges.push_back(nextUnusedId);
355 m_edges[i1].changeNode1(nodeById(nInter));
356 m_edges[i2].changeNode1(nodeById(nInter));
357 updateNodeEdgeConnectionPointers();
358 // after we have found and add an intersection: leave both for loops and start again
359 foundIntersection = true;
360 continue;
361 }
362 }
363
364 if (foundIntersection)
365 continue;
366 }
367 }

371void Network::connectBuildings(unsigned int nextUnusedId, const bool extendSupplyPipes) {
372
373 int idNextBuilding = nextUnconnectedBuilding();
374
375 while (idNextBuilding >= 0) {
376
377 unsigned int idBuilding = (unsigned int)idNextBuilding;
378
379 // find closest supply edge
380 double distMin = std::numeric_limits<double>::max();
381 unsigned int idxEdgeMin = std::numeric_limits<unsigned int>::max();
382 for (unsigned idx=0; idx<m_edges.size(); ++idx){
383 if (!m_edges[idx].m_supply)
384 continue;
385 double dist = NetworkLine(m_edges[idx]).distanceToPoint(nodeById(idBuilding)->m_position);
386 if (dist<distMin){
387 distMin = dist;
388 idxEdgeMin = idx;
389 }
390 }
391
392 // no supply edge found
393 if (idxEdgeMin == std::numeric_limits<unsigned int>::max())
394 return;
395
396 // calculate branch node
397 IBKMK::Vector3D pBranch;
398 unsigned idBranch;
399 double lineFactor = 0;
400 IBKMK::Vector3D a1 = m_edges[idxEdgeMin].m_node1->m_position;
401 IBKMK::Vector3D a2 = m_edges[idxEdgeMin].m_node2->m_position;
402 IBKMK::Vector3D b = a2 - a1;
403 IBKMK::lineToPointDistance(a1, b, nodeById(idBuilding)->m_position, lineFactor, pBranch);
404
405 // check if branch point is "outside" edge, i.e. if it is not between both end points or less than 1 m away from them
406 if (lineFactor * b.magnitude() < 1 || lineFactor * b.magnitude() > b.magnitude() - 1) {
407 // now we look which end point of edge is closer to it
408 double dist1 = (pBranch - a1).magnitude();
409 double dist2 = (pBranch - a2).magnitude();
410 idBranch = (dist1 < dist2) ? m_edges[idxEdgeMin].nodeId1() : m_edges[idxEdgeMin].nodeId2();
411 // special case: the selected end point is not a mixer. We create a branch node 1 m away from this point
412 if (nodeById(idBranch)->m_type != NetworkNode::NT_Mixer) {
413 if (dist1 < dist2)
414 pBranch = a1 + b * (1.0/b.magnitude());
415 else
416 pBranch = a2 - b * (1.0/b.magnitude());
417 idBranch = addNode(++nextUnusedId, pBranch, NetworkNode::NT_Mixer);
418 addEdge(++nextUnusedId, m_edges[idxEdgeMin].nodeId1(), idBranch, true, m_edges[idxEdgeMin].m_idPipe);
419 m_edges[idxEdgeMin].changeNode1(nodeById(idBranch));
420 updateNodeEdgeConnectionPointers();
421 }
422 // if pipe should be extended, change coordinates of branch node
423 if (extendSupplyPipes) {
424 nodeById(idBranch)->m_position = pBranch;
425 for (NetworkEdge *e: nodeById(idBranch)->m_edges)
426 e->setLengthFromCoordinates();
427 }
428 }
429
430 // branch node is "inside" edge: split edge
431 else {
432 idBranch = addNode(++nextUnusedId, pBranch, NetworkNode::NT_Mixer);
433 addEdge(++nextUnusedId, m_edges[idxEdgeMin].nodeId1(), idBranch, true, m_edges[idxEdgeMin].m_idPipe);
434 m_edges[idxEdgeMin].changeNode1(nodeById(idBranch));
435 updateNodeEdgeConnectionPointers();
436 }
437
438 // finally connect building to branch node
439 addEdge(++nextUnusedId, idBranch, idBuilding, false, m_edges[idxEdgeMin].m_idPipe);
440
441 idNextBuilding = nextUnconnectedBuilding();
442 }
443}

455void Network::cleanDeadEnds(){
456 for (unsigned i=0; i<m_nodes.size(); ++i){
457 for (const NetworkNode &n: m_nodes){
458 // look if this is a dead end
459 nodeById(n.m_id)->updateIsDeadEnd();
460 if (n.m_isDeadEnd){
461 // if so, remove all edges and the node itself
462 for (NetworkEdge *e: n.m_edges)
463 m_edges.erase(m_edges.begin() + indexOfEdge(e->nodeId1(), e->nodeId2()) );
464 m_nodes.erase(m_nodes.begin() + indexOfNode(n.m_id));
465 updateNodeEdgeConnectionPointers();
466 break;
467 }
468 }
469 }

474void Network::cleanRedundantEdges(unsigned int nextUnusedId, Network & cleanNetwork) const{
475
476 IBK_ASSERT(m_edges.size()>0);
477 std::set<unsigned> proccessedNodes;
478
479 for (const NetworkEdge &edge: m_edges){
480
481 if (edge.m_node1->isRedundant() || edge.m_node2->isRedundant()){
482
483 // proccess redundant nodes only once
484 NetworkNode * redundantNode = (edge.m_node1->isRedundant()) ? edge.m_node1: edge.m_node2;
485 if (proccessedNodes.find(redundantNode->m_id) != proccessedNodes.end())
486 continue;
487 proccessedNodes.insert(redundantNode->m_id);
488
489 // get previous node and next non-redundant node
490 NetworkNode * previousNode = edge.neighbourNode(redundantNode);
491 NetworkEdge * nextEdge = redundantNode->neighborEdge(&edge);
492 std::set<unsigned> redundantNodes;
493 double totalLength = edge.length();
494 const NetworkNode * nextNode = redundantNode->findNextNonRedundantNode(redundantNodes, totalLength, nextEdge);
495 for (const unsigned nId: redundantNodes)
496 proccessedNodes.insert(nId);
497
498 // add nodes and reduced edge to new network
499 unsigned id1 = cleanNetwork.addNode(++nextUnusedId, *previousNode);
500 unsigned id2 = cleanNetwork.addNode(++nextUnusedId, *nextNode);
501 cleanNetwork.addEdge(NetworkEdge(++nextUnusedId, id1, id2, edge.m_supply, totalLength, edge.m_idPipe));
502 }
503 else{
504 unsigned id1 = cleanNetwork.addNode(++nextUnusedId, *edge.m_node1);
505 unsigned id2 = cleanNetwork.addNode(++nextUnusedId, *edge.m_node2);
506 cleanNetwork.addEdge(NetworkEdge(++nextUnusedId, id1, id2, edge.m_supply, edge.length(), edge.m_idPipe));
507 }
508 }

512void Network::removeShortEdges(const double &thresholdLength) {
513 FUNCID(Network::removeShortEdges);
514
515 updateNodeEdgeConnectionPointers();
516
517 // check if there is a source node
518 std::vector<NetworkNode> sources;
519 findSourceNodes(sources);
520 if (sources.size() < 1)
521 throw IBK::Exception("Network has no source node. Set one node to type source.", FUNC_ID);
522
523
524 // we iterate as many times as there are edges
525 bool hasChanged = true;
526 std::set<const VICUS::NetworkNode *> dummyNodeSet;
527 std::vector<const VICUS::NetworkEdge *> orderedEdges;
528 unsigned int size = m_edges.size();
529 for (unsigned int count=0; count<size; ++count){
530
531 // if the network has changed since last iteration:
532 // we put edges in a deterministic order. They are ordered according to their distance from the source node
533 // And even more important: We need to put them in an order so that one node of each edge has already occured in
534 // one of the previuos edges within the vector
535 if (hasChanged){
536 orderedEdges.clear();
537 dummyNodeSet.clear();
538 nodeById(sources[0].m_id)->setInletOutletNode(dummyNodeSet, orderedEdges);
539 hasChanged = false;
540 }
541
542 // we store the node ids that we have already processed
543 std::set<unsigned int> processedIds;
544 processedIds.insert(orderedEdges[0]->nodeId1());
545
546 // now go through the ordered edges
547 for (const NetworkEdge *e: orderedEdges){
548
549 // determine which of both nodeIds has already been processed (=exId)
550 // and which is of both is new (=newId)
551 unsigned int newId;
552 if (processedIds.find(e->nodeId1()) != processedIds.end())
553 newId = e->nodeId2();
554 else
555 newId = e->nodeId1();
556 processedIds.insert(newId);
557 unsigned int exId = e->neighbourNode(newId);
558
559 // if the length of this edge is below threshold and the new node is not a building: we want to
560 // - modify all edges connected to the new node of this edge and connect them to the existing node
561 // - remove the according edge and the new node
562 // - leave the loop and start again so that we can assess the next edges with there adjusted lengths
563 if (e->length() < thresholdLength && nodeById(newId)->m_type == NetworkNode::NT_Mixer){
564
565 for (NetworkEdge *adjacentEdge: nodeById(newId)->m_edges){
566 if (adjacentEdge->nodeId1() == newId)
567 adjacentEdge->changeNode1(nodeById(exId));
568 else
569 adjacentEdge->changeNode2(nodeById(exId));
570 }
571
572 m_nodes.erase(m_nodes.begin() + indexOfNode(newId));
573 m_edges.erase(m_edges.begin() + indexOfEdge(exId, exId));
574
575 // now update the pointers and leave the loop, start again from the beginning
576 updateNodeEdgeConnectionPointers();
577 hasChanged = true;
578 break;
579 }
580 }
581 }

586void Network::findShortestPathForBuildings(std::map<unsigned int, std::vector<NetworkEdge *> > &minPathMap) const{
587 FUNCID(Network::findShortestPathForBuildings);
588
589 // check for source
590 std::vector<NetworkNode> sources;
591 findSourceNodes(sources);
592 if (sources.size() < 1)
593 throw IBK::Exception("Network has no source node. Set one node to type source.", FUNC_ID);
594
595 // iterate over all buildings
596 minPathMap.clear();
597 for (const NetworkNode &node: m_nodes) {
598
599 if (node.m_type != NetworkNode::NT_SubStation)
600 continue;
601
602 // there must be a defined maximum heating demand
603 if (node.m_maxHeatingDemand.value <= 0)
604 throw IBK::Exception(IBK::FormatString("Maximum heating demand of node '%1' must be >0").arg(node.m_id), FUNC_ID);
605
606 // for each source find the shortest path to current node. Finally select the shortest of these paths
607 std::vector<NetworkEdge * > minPath;
608 double minPathLength = std::numeric_limits<double>::max();
609 for (const NetworkNode &source: sources){
610 std::vector<NetworkEdge * > path;
611 // shortest path between source and node
612 dijkstraShortestPathToSource(node, source, path);
613 double pathLength = 0;
614 // length of this path
615 for (NetworkEdge *edge: path)
616 pathLength += edge->length();
617 // minumum length of paths for each source to current building
618 if (pathLength < minPathLength){
619 minPathLength = pathLength;
620 minPath = path;
621 }
622 }
623 // finally store the shortest path
624 minPathMap[node.m_id] = minPath;
625 }
626}

629void Network::sizePipeDimensions(const NetworkFluid *fluid, std::vector<const VICUS::NetworkPipe*> & availablePipes) {
630FUNCID(Network::sizePipeDimensions);
631
632 updateNodeEdgeConnectionPointers();
633
634 // check pipe database
635 if (availablePipes.empty())
636 throw IBK::Exception(IBK::FormatString("The pipe database of network '%1' is empty."
637 "Please add pipes to this network").arg(m_id), FUNC_ID);
638
639 // check parameters
640 for (unsigned int n = 0; n < NUM_P; ++n){
641 try {
642 m_para[VICUS::Network::P_MaxPressureLoss].checkedValue("MaxPressureLoss", "Pa/m", "Pa/m",
643 std::numeric_limits<double>::lowest(), true,
644 std::numeric_limits<double>::max(), true, nullptr);
645 m_para[VICUS::Network::P_TemperatureSetpoint].checkedValue("TemperatureSetpoint", "C", "C",
646 std::numeric_limits<double>::lowest(), true,
647 std::numeric_limits<double>::max(), true, nullptr);
648 m_para[VICUS::Network::P_TemperatureDifference].checkedValue("TemperatureDifference", "K", "K",
649 std::numeric_limits<double>::lowest(), true,
650 std::numeric_limits<double>::max(), true, nullptr);
651 } catch (IBK::Exception &ex) {
652 throw IBK::Exception(ex, "Error in sizing pipes algorithm!", FUNC_ID);
653 }
654 }
655
656 // set all edges heating demand = 0
657 for (NetworkEdge &edge: m_edges)
658 edge.m_nominalHeatingDemand = 0;
659
660 // find shortest path for each building node to closest source node
661 std::map<unsigned int, std::vector<NetworkEdge *> > shortestPaths;
662 findShortestPathForBuildings(shortestPaths);
663
664 // now for each building node: go along shortest path and add the nodes heating demand to each edge along that path
665 for (auto it = shortestPaths.begin(); it != shortestPaths.end(); ++it){
666 NetworkNode *building = nodeById(it->first); // get pointer to building node
667 std::vector<NetworkEdge *> &shortestPath = it->second; // for readability
668 for (NetworkEdge * edge: shortestPath)
669 edge->m_nominalHeatingDemand += building->m_maxHeatingDemand.value;
670 }
671
672 // in case there is a pipe which is not part of any path (e.g. in circular grid): assign the adjacent heating demand
673 for (NetworkEdge &e: m_edges){
674 if (e.m_nominalHeatingDemand <= 0){
675 std::set<NetworkEdge *> edges1, edges2;
676 e.m_nominalHeatingDemand = 0.5 * ( e.m_node1->adjacentHeatingDemand(edges1)
677 + e.m_node2->adjacentHeatingDemand(edges2) );
678 }
679 }
680
681 // determine pipe with largest inner diameter
682 const NetworkPipe * largestPipe = availablePipes[0];
683 for (const NetworkPipe * pipe: availablePipes) {
684 if (pipe->diameterInside() > largestPipe->diameterInside())
685 largestPipe = pipe;
686 }
687
688 double rho = fluid->m_para[NetworkFluid::P_Density].value;
689 double kinvis = fluid->m_kinematicViscosity.m_values.value(m_para[P_TemperatureSetpoint].get_value("C"));
690 double cp = fluid->m_para[NetworkFluid::P_HeatCapacity].value;
691
692 // for each edge: find the smallest pipe from DB that has a pressure loss below deltapMax
693 double deltaPMax = m_para[P_MaxPressureLoss].get_value("Pa/m");
694 for (NetworkEdge &e: m_edges) {
695 const NetworkPipe * currentPipe = nullptr; // here we store the candidate for the pipe to use
696
697 for (const NetworkPipe * pipe : availablePipes){
698 e.m_nominalMassFlow = e.m_nominalHeatingDemand / (m_para[P_TemperatureDifference].get_value("K") * cp);
699 double di = pipe->diameterInside();
700 double area = PI/4 * di * di;
701 double vel = e.m_nominalMassFlow / (rho * area);
702 double re = IBK::ReynoldsNumber(vel, kinvis, di);
703 // pressure loss per length (Pa/m)
704 double zeta = 1.0 / di * IBK::FrictionFactorSwamee(re, di, m_para[VICUS::NetworkPipe::P_DiameterOutside].value);
705 double dp = zeta * rho/2 * vel * vel;
706 // select smallest possible pipe
707 if (dp < deltaPMax){
708 // if still unset, use this pipe
709 if (currentPipe == nullptr)
710 currentPipe = pipe;
711 // otherwise compare inside diameter and only take if smaller
712 else if (pipe->diameterInside() < currentPipe->diameterInside())
713 currentPipe = pipe;
714 }
715 }
716
717 // if we found a pipe, store its id
718 if (currentPipe != nullptr)
719 e.m_idPipe = currentPipe->m_id;
720 // otherwise store ID of biggest pipe
721 else
722 e.m_idPipe = largestPipe->m_id;
723 }
724
725}

728void Network::calcTemperatureChangeIndicator(const NetworkFluid & fluid, const Database<NetworkPipe> &pipeDB,
729 std::map<unsigned int, std::vector<NetworkEdge *> > &shortestPaths) const{
730 FUNCID(Network::calcTemperatureChangeIndicator);
731
732 // check for source
733 std::vector<NetworkNode> sources;
734 findSourceNodes(sources);
735 if (sources.size() < 1)
736 throw IBK::Exception("Network has no source node. Set one node to type source.", FUNC_ID);
737
738 // set all edges heating demand = 0
739 for (const NetworkEdge &edge: m_edges)
740 const_cast<NetworkEdge&>(edge).m_nominalHeatingDemand = 0;
741
742 // find shortest path for each building node to closest source node
743 shortestPaths.clear();
744 findShortestPathForBuildings(shortestPaths);
745
746 // now for each building node: go along shortest path and add the nodes heating demand to each edge along that path
747 for (auto it = shortestPaths.begin(); it != shortestPaths.end(); ++it){
748 const NetworkNode *building = nodeById(it->first); // get pointer to building node
749 std::vector<NetworkEdge *> &shortestPath = it->second; // for readability
750 for (NetworkEdge * edge: shortestPath)
751 edge->m_nominalHeatingDemand += building->m_maxHeatingDemand.value;
752 }
753
754 // in case there is a pipe which is not part of any path (e.g. in circular grid): assign the adjacent heating demand
755 for (const NetworkEdge &e: m_edges){
756 if (e.m_nominalHeatingDemand <= 0){
757 std::set<NetworkEdge *> edges1, edges2;
758 const_cast<NetworkEdge&>(e).m_nominalHeatingDemand = 0.5 * ( e.m_node1->adjacentHeatingDemand(edges1)
759 + e.m_node2->adjacentHeatingDemand(edges2) );
760 }
761 }
762
763 // fluid properties
764 const double &rho = fluid.m_para[NetworkFluid::P_Density].value;
765 const double &kinvis = fluid.m_kinematicViscosity.m_values.value(m_para[P_TemperatureSetpoint].get_value("C"));
766 const double &cp = fluid.m_para[NetworkFluid::P_HeatCapacity].value;
767 const double &lambda = fluid.m_para[NetworkFluid::P_Conductivity].value;
768
769 // calculate temperature change indicator for each edge
770 for (const NetworkEdge &e: m_edges){
771
772 const NetworkPipe *pipe = pipeDB[e.m_idPipe];
773 Q_ASSERT(pipe != nullptr);
774 const_cast<NetworkEdge&>(e).m_nominalMassFlow = e.m_nominalHeatingDemand / (m_para[P_TemperatureDifference].get_value("K") * cp);
775
776 double di = pipe->diameterInside();
777 double area = PI/4 * di * di;
778 double vel = e.m_nominalMassFlow / (rho * area);
779 double re = IBK::ReynoldsNumber(vel, kinvis, di);
780 double prandtl = IBK::PrandtlNumber(kinvis, cp, lambda, rho);
781 double nusselt = IBK::NusseltNumber(re, prandtl, e.length(), pipe->diameterInside());
782 double innerHeatTransferCoefficient = nusselt * lambda / pipe->diameterInside();
783 double UAValue = e.length() /
784 (
785 1.0 / ( innerHeatTransferCoefficient * pipe->diameterInside() * PI)
786 + 1.0 / pipe->UValue()
787 );
788
789 // is dimensionless or [K/K] for interpretation
790 const_cast<NetworkEdge&>(e).m_tempChangeIndicator = UAValue / (e.m_nominalMassFlow * cp);
791 }
792}

803void Network::dijkstraShortestPathToSource(const NetworkNode &startNode, const NetworkNode &endNode,
804 std::vector<NetworkEdge*> &pathEndToStart) const {
805
806 // init: all nodes have infinte distance to start node and no predecessor
807 for (const NetworkNode &n: m_nodes){
808 n.m_distanceToStart = std::numeric_limits<double>::max();
809 n.m_predecessor = nullptr;
810 }
811 startNode.m_distanceToStart = 0;
812 std::set<unsigned> visitedNodes;
813
814 // go through all not-visited nodes
815 while (visitedNodes.size() <= m_nodes.size()){
816 // find node with currently smallest distance to start, which has not yet been visited:
817 double minDistance = std::numeric_limits<double>::max();
818 const NetworkNode *nMin = nullptr;
819 for (const NetworkNode &n: m_nodes){
820 if (visitedNodes.find(n.m_id) == visitedNodes.end() && n.m_distanceToStart < minDistance){
821 minDistance = n.m_distanceToStart;
822 nMin = &n;
823 }
824 }
825 IBK_ASSERT(nMin != nullptr);
826 // if endNode reached: return path
827 if (nMin->m_id == endNode.m_id){
828 nMin->pathToNull(pathEndToStart);
829 return;
830 }
831 // update distance from start to neighbours of nMin
832 visitedNodes.insert(nMin->m_id);
833 nMin->updateNeighbourDistances();
834 }
835}
