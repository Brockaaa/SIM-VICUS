30bool Construction::isValid(const VICUS::Database<Material> & materials) const {
31 for (unsigned int i=0; i<m_materialLayers.size(); ++i) {
32 if (!m_materialLayers[i].isValid(materials)) {
33 // NOTE: isValid() returns false also if material does not exist, which can happen for newly constructed constructions
34 const Material * mat = materials[m_materialLayers[i].m_idMaterial];
35 if (mat == nullptr)
36 m_errorMsg = "No material referenced in this layer, yet";
37 else
38 m_errorMsg = "Material '" + mat->m_displayName.string("de", true) + "' is not valid.";
39 return false; // error, invalid layer thickness
40 }
41 }
42 return true;
43}

46bool Construction::calculateUValue(double & UValue, const VICUS::Database<Material> & materials, double ri, double re) const {
47 // simple calculation
48 double R = ri + re;
49 for (unsigned int i=0; i<m_materialLayers.size(); ++i) {
50 const Material * mat = materials[m_materialLayers[i].m_idMaterial];
51 if (mat == nullptr)
52 return false; // error, material with this ID is not found
53 double lambda, thickness;
54 if ( (lambda = mat->m_para[Material::P_Conductivity].value) <= 0)
55 return false; // error, invalid lambda
56 if ( (thickness = m_materialLayers[i].m_thickness.value) <= 0)
57 return false; // error, invalid layer thickness
58 R += thickness/lambda;
59 }
60 UValue = 1/(R+1e-10);
61 return true;
62}

65AbstractDBElement::ComparisonResult Construction::equal(const AbstractDBElement *other) const{
66 const Construction * otherConstr = dynamic_cast<const Construction*>(other);
67 if (otherConstr == nullptr)
68 return Different;
69
70 if (m_materialLayers != otherConstr->m_materialLayers)
71 return Different;
72
73 if ( m_displayName != otherConstr->m_displayName ||
74 m_dataSource != otherConstr->m_dataSource ||
75 m_notes != otherConstr->m_notes ||
76 m_usageType != otherConstr->m_usageType ||
77 m_insulationKind != otherConstr->m_insulationKind ||
78 m_materialKind != otherConstr->m_materialKind)
79 return OnlyMetaDataDiffers;
80
81 return Equal;
82}
