40/*! Contains definition of a single grid plane and related functionality.
41
42 A grid plane is defined by the offset (always in the middle), its normal vector and
43 local X vector, that defines the orientation.
44
45 The dimension of the grid plane is defined by width (total width) and the major grid spacing.
46 The width is automatically adjusted to hold an even number number of grid spacings.
47 For example, if the grid is defined with a width of 125 m, and grid spacing is 10 m, than
48 the final grid will extend from -70..70 covering a total of 140 m.
49
50 The minor grid spacing is computed automatically to be 1/10 of the major grid.
51*/
52class GridPlane {
53 VICUS_READWRITE_PRIVATE
54public:
55 // *** PUBLIC MEMBER FUNCTIONS ***
56
57 GridPlane() = default;
58
59 /*! Initializing constructor, generated grid will be visible and active by default. */
60 GridPlane(const IBKMK::Vector3D & offset, const IBKMK::Vector3D & normal, const IBKMK::Vector3D & localX,
61 const QColor majorGridColor, double width, double majorGridSpacing);
62
63 void readXML(const TiXmlElement * element);
64 TiXmlElement * writeXML(TiXmlElement * parent) const;
65
66 VICUS_COMP(GridPlane)
67
68 /*! This function checks for valid normal and localX vectors and updates the cached localY vector.
69 Throws an exception in case of invalid vectors.
70 */
71 void updateLocalY();
72
73 /*! Computes intersection point of a line defined by point p and direction with the plane,
74 and also returns the nearest grid snap point.
75 Throws an exception if plane is invalid (null vector).
76 \param p Start point of the line
77 \param direction Direction vector of the line
78 \param t Here the scale factor for the line to hit the plane is stored (0...1)
79 \param intersectionPoint Will hold the point where line hits the surface
80 \return Returns true, if an intersection was found, otherwise false. If plane is inactive, function
81 always returns false;
82 */
83 bool intersectsLine(const IBKMK::Vector3D & p, const IBKMK::Vector3D & direction,
84 double & t, IBKMK::Vector3D & intersectionPoint) const;
85
86 /*! Determines the snap point nearest to the intersection point (which must lie on the plane).
87 \param intersectionPoint Point on the plane
88 \param snapPoint Will hold the snap point on grid nearest to the intersection point
89 */
90 void closestSnapPoint(const IBKMK::Vector3D & intersectionPoint, IBKMK::Vector3D & snapPoint) const;
91
92 /*! Some descriptive name of the grid. */
93 QString m_name; // XML:A
94 /*! True, if grid should be shown. */
95 bool m_isVisible = true; // XML:A
96 /*! True, if grid should be clickable. */
97 bool m_isActive = true; // XML:A
98 /*! Offset vector of grid plane */
99 IBKMK::Vector3D m_offset; // XML:E
100 /*! Normal vector of grid plane, must have magnitude 1 for intersectsLine() function to work. */
101 IBKMK::Vector3D m_normal; // XML:E
102 /*! Local X vector of grid plane, must be perpendicular to m_normal; indicates the orientation of the grid. */
103 IBKMK::Vector3D m_localX; // XML:E
104 /*! Major grid color (minor grid color is automatically adjusted, either brighter or dimmer depending on current theme) */
105 QColor m_color; // XML:A
106 /*! Grid width in [m]. */
107 double m_width = 100; // XML:A
108 /*! Major grid spacing in [m]. */
109 double m_spacing = 10; // XML:A
110
111private:
112
113 // *** RUNTIME VARIABLES ***
114
115 // The variables below are used in the snapping algorithm and are cached to avoid calculation
116 // They are recomputed whenever the grid is prepared for rendering (see Vic3D::GridObject::create()).
117
118 /*! Local Y-axis. */
119 IBKMK::Vector3D m_localY;
120 /*! Minimum/Maximum local x/y coordinate of grid. */
121 double m_gridExtends;
122 /*! Number of major grid lines (always an odd number). */
123 unsigned int m_nGridLines = 0;
124
125
126 friend class Vic3D::GridObject;
127};
