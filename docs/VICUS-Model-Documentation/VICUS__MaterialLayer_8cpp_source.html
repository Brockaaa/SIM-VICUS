30bool MaterialLayer::isValid(const VICUS::Database<VICUS::Material> & materials) const {
31 const Material * mat = materials[m_idMaterial];
32 if (mat == nullptr) {
33 m_errorMsg = "Material with id '" + std::to_string(m_idMaterial) +"' does not exist.";
34 return false; // error, material with this ID is not found
35 }
36
37 if (!mat->isValid(false)) {
38 m_errorMsg = "Material '" + mat->m_displayName.string("de", true) + "' is not valid.";
39 return false;
40 }
41
42 try {
43 m_thickness.checkedValue("Thickness", "m", "m", 0, false, std::numeric_limits<double>::max(), false,
44 "Thickness > 0 is required.");
45 } catch (IBK::Exception & ex) {
46 ex.writeMsgStackToError();
47 m_errorMsg = "Material '" + mat->m_displayName.string("de", true) + "' has following error: " + ex.what();
48 return false;
49 }
50 return true;
51}
