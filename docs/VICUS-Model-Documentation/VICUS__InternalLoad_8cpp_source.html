32bool InternalLoad::isValid(const Database<Schedule> &scheduleDB) const {
33 std::string err;
34 switch (m_category) {
35 case VICUS::InternalLoad::IC_Person: {
36
37 try {
38 switch(m_personCountMethod) {
39 case InternalLoad::PCM_PersonPerArea: {
40 m_para[P_PersonPerArea].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_PersonPerArea),
41 "Person/m2", "Person/m2", 0, true, 100000, true, nullptr);
42 }
43 break;
44 case InternalLoad::PCM_AreaPerPerson: {
45 m_para[P_AreaPerPerson].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_AreaPerPerson),
46 "m2/Person", "m2/Person", 0, false, 100000, true, nullptr);
47
48 }
49 break;
50 case InternalLoad::PCM_PersonCount: {
51 m_para[P_PersonCount].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_PersonCount),
52 "-", "-", 0, true, 100000, true, nullptr);
53 }
54 break;
55 case InternalLoad::NUM_PCM:
56 return false;
57 }
58
59 } catch (IBK::Exception &ex) {
60 m_errorMsg = ex.what();
61 return false;
62 }
63
64 if (m_idActivitySchedule == INVALID_ID) {
65 m_errorMsg = "Activity schedule is not set.";
66 return false;
67 }
68 if (m_idOccupancySchedule == INVALID_ID) {
69 m_errorMsg = "Occupancy schedule is not set.";
70 return false;
71 }
72
73 // check schedules occ and act
74 // check if schedule ID is existing and valid
75 const Schedule * actSched = scheduleDB[m_idActivitySchedule];
76 if (actSched == nullptr) {
77 m_errorMsg = "Activity schedule with id '" + std::to_string(m_idActivitySchedule) + "' does not exist.";
78 return false;
79 }
80 if (!actSched->isValid(err, true)) {
81 m_errorMsg = "Activity schedule '" + actSched->m_displayName.string("de", true) + "' is invalid.";
82 return false;
83 }
84
85 // check schedule moist rate
86 if (m_idMoistureProductionRatePerAreaSchedule != INVALID_ID) {
87 const Schedule *moistSched = scheduleDB[m_idMoistureProductionRatePerAreaSchedule];
88 if (moistSched == nullptr) {
89 m_errorMsg = "Moisture rate schedule with id '" + std::to_string(m_idMoistureProductionRatePerAreaSchedule) + "' does not exist.";
90 return false;
91 }
92 if (!moistSched->isValid(err, true)) {
93 m_errorMsg = "Moisture rate schedule '" + moistSched->m_displayName.string("en", true) + "' is invalid.";
94 return false;
95 }
96 }
97
98 const Schedule * occSched = scheduleDB[m_idOccupancySchedule];
99 if (occSched == nullptr) {
100 m_errorMsg = "Occupancy schedule with id '" + std::to_string(m_idActivitySchedule) + "' does not exist.";
101 return false;
102 }
103 if (!occSched->isValid(err, true)) {
104 m_errorMsg = "Occupancy schedule '" + actSched->m_displayName.string("de", true) + "' is invalid.";
105 return false;
106 }
107 }
108 break;
109
110 case VICUS::InternalLoad::IC_ElectricEquiment:
111 case VICUS::InternalLoad::IC_Lighting:
112 case VICUS::InternalLoad::IC_Other: {
113 switch (m_powerMethod) {
114 case VICUS::InternalLoad::PM_PowerPerArea: {
115 try {
116 m_para[P_PowerPerArea].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_PowerPerArea),
117 "W/m2", "W/m2", 0, true, 1000, true, nullptr);
118 } catch (IBK::Exception &ex) {
119 m_errorMsg = ex.what();
120 return false;
121 }
122 }
123 break;
124 case VICUS::InternalLoad::PM_Power: {
125 try {
126 m_para[P_Power].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_Power),
127 "W", "W", 0, true, 100000, true, nullptr);
128 } catch (IBK::Exception &ex) {
129 m_errorMsg = ex.what();
130 return false;
131 }
132 }
133 break;
134 case VICUS::InternalLoad::NUM_PM: {
135 m_errorMsg = "Internal loads type is not defined.";
136 return false;
137 }
138 }
139
140 if (m_idPowerManagementSchedule == INVALID_ID) {
141 m_errorMsg = "Power management schedule is not set.";
142 return false;
143 }
144
145 // check if schedule ID is existing and valid
146 const Schedule * powerSched = scheduleDB[m_idPowerManagementSchedule];
147 if (powerSched == nullptr) {
148 m_errorMsg = "Power management schedule with id '" + std::to_string(m_idPowerManagementSchedule) + "' does not exist.";
149 return false;
150 }
151 if (!powerSched->isValid(err, true)) {
152 m_errorMsg = "Power management '" + powerSched->m_displayName.string("de", true) + "' is invalid.";
153 return false;
154 }
155 }
156 break;
157
158 case NUM_MC: {
159 m_errorMsg = "Internal load category is not set.";
160 return false;
161 }
162 }
163
164 try {
165
166 m_para[P_ConvectiveHeatFactor].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_ConvectiveHeatFactor),
167 "---", "---", 0, true, 1, true, nullptr);
168 if (m_category == IC_ElectricEquiment) {
169 m_para[P_LossHeatFactor].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_LossHeatFactor),
170 "---", "---", 0, true, 1, true, nullptr);
171 m_para[P_LatentHeatFactor].checkedValue(KeywordList::Keyword("InternalLoad::para_t", P_LatentHeatFactor),
172 "---", "---", 0, true, 1, true, nullptr);
173 }
174 } catch (IBK::Exception &ex) {
175 m_errorMsg = ex.what();
176 return false;
177 }
178
179 return true;
180}

183AbstractDBElement::ComparisonResult InternalLoad::equal(const AbstractDBElement *other) const {
184 const InternalLoad * otherIntLoad = dynamic_cast<const InternalLoad*>(other);
185 if (otherIntLoad == nullptr)
186 return Different;
187
188 for (unsigned int i=0; i<NUM_P; ++i){
189 if (m_para[i] != otherIntLoad->m_para[i])
190 return Different;
191 }
192 if ( m_category != otherIntLoad->m_category ||
193 m_personCountMethod != otherIntLoad->m_personCountMethod||
194 m_powerMethod != otherIntLoad->m_powerMethod ||
195 m_idOccupancySchedule != otherIntLoad->m_idOccupancySchedule ||
196 m_idActivitySchedule != otherIntLoad->m_idActivitySchedule ||
197 m_idMoistureProductionRatePerAreaSchedule != otherIntLoad->m_idMoistureProductionRatePerAreaSchedule ||
198 m_idPowerManagementSchedule != otherIntLoad->m_idPowerManagementSchedule)
199 return Different;
200
201 //check meta data
202
203 if (m_displayName != otherIntLoad->m_displayName)
204 return OnlyMetaDataDiffers;
205
206 return Equal;
207}
