43/*! Stores data for a single material layer in a construction. */
44class MaterialLayer {
45public:
46
47 // *** PUBLIC MEMBER FUNCTIONS ***
48
49 /*! Default c'tor. */
50 MaterialLayer() {}
51
52 /*! Simple Constructor with thickness in [m] and material id. */
53 MaterialLayer(double thickness, unsigned int id):
54 m_idMaterial(id),
55 m_thickness(IBK::Parameter("Thickness", thickness, "m"))
56 {}
57
58 /*! Simple Constructor with thickness and material id. */
59 MaterialLayer(IBK::Parameter thickness, unsigned int id):
60 m_idMaterial(id),
61 m_thickness(thickness)
62 {}
63
64
65 VICUS_READWRITE
66
67 /*! Checks if all referenced materials exist and if their parameters are valid. */
68 bool isValid(const VICUS::Database<VICUS::Material> & materials) const;
69
70 /*! Inequality operator. */
71 bool operator!=(const MaterialLayer & other) const {
72 return (m_idMaterial != other.m_idMaterial ||
73 m_thickness != other.m_thickness);
74 }
75 /*! Equality operator. */
76 bool operator==(const MaterialLayer & other) const { return !operator!=(other); }
77
78 // *** PUBLIC MEMBER VARIABLES ***
79
80 /*! Unique ID of material. */
81 unsigned int m_idMaterial = INVALID_ID; // XML:A:required
82
83 /*! Thickness of the material layer. */
84 IBK::Parameter m_thickness; // XML:E:required
85
86 /*! Holds error string in order to give users a tooltip in db dialog. */
87 mutable std::string m_errorMsg;
88};
