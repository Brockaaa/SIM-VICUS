30bool BoundaryCondition::isValid(const Database<Schedule> & scheduleDB) const {
31 if (m_id == VICUS::INVALID_ID)
32 return false;
33
34 if(!m_heatConduction.isValid(scheduleDB)) {
35 m_errorMsg = "Heat conduction is not valid.";
36 return false;
37 }
38
39 try {
40 m_longWaveEmission.checkParameters();
41 // TODO : add vapor diffusion/air flow once needed
42 } catch (...) {
43 m_errorMsg = "Long wave emission is not valid.";
44 return false;
45 }
46 try {
47 m_solarAbsorption.checkParameters();
48 } catch (...) {
49 m_errorMsg = "Solar absorption is not valid.";
50 return false;
51 }
52
53 return true;
54}

64QString BoundaryCondition::htmlDescription(const VICUS::Database<Schedule> & scheduleDB) const {
65 QString html = "<html><body>";
66
67 if (!isValid(scheduleDB))
68 html += tr("<p><span style=\" color:#a40000;\">Invalid parameter definition found.</span></p>");
69
70 html += tr("<p><b>Parameters:</b></p><ul>");
71 if (m_heatConduction.m_modelType != VICUS::InterfaceHeatConduction::NUM_MT) {
72 QString heatCondInfo;
73 switch (m_heatConduction.m_modelType) {
74 case VICUS::InterfaceHeatConduction::MT_Constant:
75 heatCondInfo = tr("Constant, heat transfer coefficient = %1 W/m2K").arg(m_heatConduction.m_para[NANDRAD::InterfaceHeatConduction::P_HeatTransferCoefficient].get_value("W/m2K"));
76 break;
77 case VICUS::InterfaceHeatConduction::NUM_MT: break;
78 }
79 html += tr("<li><i>Heat conduction</i><br>%1</li>").arg(heatCondInfo);
80 }
81 if (m_solarAbsorption.m_modelType != NANDRAD::InterfaceSolarAbsorption::NUM_MT) {
82 QString info;
83 switch (m_solarAbsorption.m_modelType) {
84 case NANDRAD::InterfaceSolarAbsorption::NUM_MT: break;
85 case NANDRAD::InterfaceSolarAbsorption::MT_Constant:
86 info = tr("Constant, adsorption coeff. = %1").arg(m_solarAbsorption.m_para[NANDRAD::InterfaceSolarAbsorption::P_AbsorptionCoefficient].value);
87 break;
88 }
89 html += tr("<li><i>Solar adsorption</i><br>%1</li>").arg(info);
90 }
91 if (m_longWaveEmission.m_modelType != NANDRAD::InterfaceLongWaveEmission::NUM_MT) {
92 QString info;
93 switch (m_longWaveEmission.m_modelType) {
94 case NANDRAD::InterfaceLongWaveEmission::NUM_MT: break;
95 case NANDRAD::InterfaceLongWaveEmission::MT_Constant:
96 info = tr("Constant, emissivity = %1").arg(m_longWaveEmission.m_para[NANDRAD::InterfaceLongWaveEmission::P_Emissivity].value);
97 break;
98 }
99 html += tr("<li><i>Long wave radiation exchange</i><br>%1</li>").arg(info);
100 }
101
102 html += "</body></html>";
103 return html;
104}

107AbstractDBElement::ComparisonResult BoundaryCondition::equal(const AbstractDBElement *other) const {
108 const BoundaryCondition * otherBC = dynamic_cast<const BoundaryCondition*>(other);
109 if (otherBC == nullptr)
110 return Different;
111
112 // check parameters
113
114 if (m_heatConduction != otherBC->m_heatConduction ||
115 m_longWaveEmission != otherBC->m_longWaveEmission ||
116 m_solarAbsorption != otherBC->m_solarAbsorption ||
117 m_vaporDiffusion != otherBC->m_vaporDiffusion ||
118 m_airFlow != otherBC->m_airFlow)
119 return Different;
120
121 // check meta data
122
123 if (m_displayName != otherBC->m_displayName)
124 return OnlyMetaDataDiffers;
125
126 return Equal;
127}
