39bool SupplySystem::isValid(const Database<SubNetwork> &subNetworkDB,
40 const Database<NetworkComponent> &compDB,
41 const Database<NetworkController> &ctrlDB,
42 const Database<Schedule> &scheduleDB) const
43{
44 FUNCID(SupplySystem::isValid);
45 try {
46
47 // check parameters for stand alone mode
48 if(m_supplyType == ST_StandAlone) {
49 m_para[P_MaximumMassFlux].checkedValue("MaximumMassFlux", "kg/s", "kg/s", 0.0, false,
50 std::numeric_limits<double>::max(), false,
51 "Maximum mass flux must be > 0");
52 m_para[P_SupplyTemperature].checkedValue("SupplyTemperature", "C", "C", 0.0, false,
53 std::numeric_limits<double>::max(), false,
54 "Supply temperature must above water freezing point.");
55 return true;
56 }
57
58 // check parameters for database mode
59 if(m_supplyType == ST_DatabaseFMU) {
60 if(m_idSupplyFMU == VICUS::INVALID_ID) {
61 throw IBK::Exception("Database FMU is not set!", FUNC_ID);
62 }
63
64 m_para[P_MaximumMassFluxFMU].checkedValue("MaximumMassFluxFMU", "kg/s", "kg/s", 0.0, false,
65 std::numeric_limits<double>::max(), false,
66 "Maximum mass flux must be > 0");
67 m_para[P_HeatingPowerFMU].checkedValue("HeatingPowerFMU", "kW", "kW", 0.0, false,
68 std::numeric_limits<double>::max(), false,
69 "Maximum heating power flux must be > 0");
70 }
71
72 // check parameters for user defined mode
73 if(m_supplyType == ST_UserDefinedFMU) {
74 if(m_supplyFMUPath.isEmpty() ) {
75 throw IBK::Exception("Path to user defined FMU is not set!", FUNC_ID);
76 }
77
78 QFileInfo fileInfo(m_supplyFMUPath);
79 if( !fileInfo.exists() || fileInfo.suffix() != "fmu") {
80 throw IBK::Exception(QString("Invalid path %1 to user defined FMU!").arg(m_supplyFMUPath).toStdString(),
81 FUNC_ID);
82 }
83
84 m_para[P_MaximumMassFluxFMU].checkedValue("MaximumMassFluxFMU", "kg/s", "kg/s", 0.0, false,
85 std::numeric_limits<double>::max(), false,
86 "Maximum mass flux must be > 0");
87 m_para[P_HeatingPowerFMU].checkedValue("HeatingPowerFMU", "kW", "kW", 0.0, false,
88 std::numeric_limits<double>::max(), false,
89 "Maximum heating power flux must be > 0");
90 return true;
91 }
92
93 // check sub network
94 if (m_supplyType == ST_SubNetwork) {
95 m_subNetwork = subNetworkDB[m_idSubNetwork];
96 if (m_subNetwork != nullptr) {
97 if (m_subNetwork->isValid(compDB, ctrlDB, scheduleDB))
98 return true;
99 }
100 }
101
102 return false;
103
104 } catch (IBK::Exception & ex) {
105 ex.writeMsgStackToError();
106 m_errorMsg = ex.what();
107 return false;
108 }
109}
