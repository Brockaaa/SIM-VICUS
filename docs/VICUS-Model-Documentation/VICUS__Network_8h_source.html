67 /*! Defines which model in Nandrad shall be used */
68 enum PipeModel {
69 PM_SimplePipe, // Keyword: SimplePipe 'Pipe with a single fluid volume and with heat exchange'
70 PM_DynamicPipe, // Keyword: DynamicPipe 'Pipe with a discretized fluid volume and heat exchange'
71 NUM_PM
72 };
73
74 /*! The various types (equations) of the hydraulic component. */
75 enum ModelType {
76 MT_HydraulicNetwork, // Keyword: HydraulicNetwork 'Only Hydraulic calculation with constant temperature'
77 MT_ThermalHydraulicNetwork, // Keyword: ThermalHydraulicNetwork 'Thermo-hydraulic calculation'
78 NUM_MT
79 };
80
81 /*! Defines wether this network represents a Single or DoublePipe Network (where each edges represents 2 pipes) */
82 enum NetworkType {
83 NET_SinglePipe, // Keyword: SinglePipe
84 NET_DoublePipe, // Keyword: DoublePipe
85 NUM_NET
86 };
87
88 /*! The parameters */
89 enum para_t {
90 P_TemperatureSetpoint, // Keyword: TemperatureSetpoint [C] 'Temperature for pipe dimensioning algorithm'
91 P_TemperatureDifference, // Keyword: TemperatureDifference [K] 'Temperature difference for pipe dimensioning algorithm'
92 P_MaxPressureLoss, // Keyword: MaxPressureLoss [Pa/m] 'Maximum pressure loss for pipe dimensioning algorithm'
93 P_ReferencePressure, // Keyword: ReferencePressure [Pa] 'Reference pressure applied to reference element'
94 P_DefaultFluidTemperature, // Keyword: DefaultFluidTemperature [C] 'Fluid temperature for hydraulic calculation, else initial temperature'
95 P_InitialFluidTemperature, // Keyword: InitialFluidTemperature [C] 'Initial Fluid temperature for thermo-hydraulic calculation'
96 P_MaxPipeDiscretization, // Keyword: MaxPipeDiscretization [m] 'Maximum discretization step for dynamic pipe model'
97 NUM_P
98 };

114 /*! add node to network based on coordinates and type and return the node id.
115 * When considerCoordinates==true and the given coordinates exist already in the network: return the id of this existing node
116 ALWAYS use this function if you add nodes with coordinates that where calculated based on already existing coordinates */
117 unsigned int addNode(unsigned int preferedId, const IBKMK::Vector3D &v, const NetworkNode::NodeType type, const bool considerCoordinates=true);

119 /*! generate all intersections between edges in the network, hence connects all edges which intersect each other */
120 void generateIntersections(unsigned int nextUnusedId, std::vector<unsigned int> & addedNodes, std::vector<unsigned int> & addedEdges);
121
122 /*! Should be called whenever m_nodes or m_edges has been modified.
123 Checks for correct node/edge IDs and throws IBK::Exceptions, if invalid IDs are found.
124 */

126
127 /*! Checks that all edges and nodes are connected with each other (i.e. single graph network). */
128 bool checkConnectedGraph() const;
129
130 /*! iterates through all building nodes, finds closest supply edge and connects the building node to the network */
131 void connectBuildings(unsigned int nextUnusedId, const bool extendSupplyPipes);
132
133 /*! returns the first id in m_nodes, which is an unconnected building */
134 int nextUnconnectedBuilding() const;
135
136 /*! remove all nodes and their adjacent edges when the respective node is a "dead end",
137 meaning that none of the neighboring nodes is a building or source */
138 void cleanDeadEnds();
139
140 /*! stores a copy of the network without any redundant edges */
141 void cleanRedundantEdges(unsigned int nextUnusedId, Network & cleanNetwork) const;
142
143 /*! iteratively removes edges which have a length below thresholdLength in [m] */
144 void removeShortEdges(const double &thresholdLength);
145
146 /*! For each building node: Find shortest path to the closest source node and store the pointers to the edges
147 * along that path. The result is a map with keys being the ids of the building nodes */
148 void findShortestPathForBuildings(std::map<unsigned int, std::vector<NetworkEdge *> > &minPathMap) const;
149
150 /*! calculate pipe dimensions using a maximum pressure loss per length and fixed temperature difference
151 * the mass flow rate of each pipe will be calculated based on the heatDemand of connected consumer loads (e.g. buildings)
152 */
153 void sizePipeDimensions(const NetworkFluid *fluid, std::vector<const NetworkPipe *> & availablePipes);
154
155 /*! Calculate the temperature change indicator for each edge. Therefore the massflow under nominal conditions for
156 each edge is calculated using the shortest Path algorithm. Valid pipeIds must be specified in advance for
157 each edge.
158 */
159 void calcTemperatureChangeIndicator(const NetworkFluid &fluid, const Database<NetworkPipe> &pipeDB,
160 std::map<unsigned int, std::vector<NetworkEdge *> > &shortestPaths) const;
161
162 void findSourceNodes(std::vector<NetworkNode> &sources) const;
163

166 void writeNetworkNodesCSV(const IBK::Path &file) const;
167
168 void writePathCSV(const IBK::Path &file, const NetworkNode & nodeById, const std::vector<NetworkEdge *> &path) const;
169
170 void writeBuildingsCSV(const IBK::Path &file) const;
171
172 /*! find shortest Path from given startNode (e.g. a building) to Node with type source
173 * using dijkstra-algorithm, implemented according to Wikipedia and return path as vector of edges
174 */
175 void dijkstraShortestPathToSource(const NetworkNode &startNode, const NetworkNode &endNode,
176 std::vector<NetworkEdge*> &pathEndToStart) const;

179 void updateExtends();

184 void setOrigin(const IBKMK::Vector3D &origin);

207 /*! returns the number of nodes with type NT_Building */

213 void updateVisualizationRadius(const VICUS::Database<VICUS::NetworkPipe> & pipeDB);
214
215 /*! sets default colors */
216 void setDefaultColors() const;
217
218 /*! sets the network object as well as all edges and nodes visible */
219 void setVisible(bool visible);
220
221 /*! Identifies node by its id (in m_nodes vector) and returns according pointer.

223 */
224 NetworkNode *nodeById(unsigned int id);
225
226 /*! Identifies node by its id (in m_nodes vector) and returns according const pointer.

228 */
229 const NetworkNode *nodeById(unsigned int id) const;

231 /*! Identifies node by its id and returns according index in m_nodes vector.
