10bool SubNetwork::isValid(const Database<NetworkComponent> &compDB,
11 const Database<NetworkController> &ctrlDB,
12 const Database<Schedule> &scheduleDB) const
13{
14 if (m_elements.empty())
15 return false;
16
17 for (const NetworkElement &e: m_elements) {
18 // check if we have valid ids
19 if (e.m_id == INVALID_ID ||
20 e.m_componentId == INVALID_ID ||
21 e.m_inletNodeId == INVALID_ID ||
22 e.m_outletNodeId == INVALID_ID)
23 return false;
24
25 // check if the component exists in DB
26 if (compDB[e.m_componentId] == nullptr)
27 return false;
28 if (!compDB[e.m_componentId]->isValid(scheduleDB))
29 return false;
30
31 // if controller id exists, it must also reference a valid controller in DB
32 if (e.m_controlElementId != INVALID_ID){
33 if (ctrlDB[e.m_controlElementId] == nullptr)
34 return false;
35 if (!ctrlDB[e.m_controlElementId]->isValid(scheduleDB))
36 return false;
37 }
38 }
39
40 return true;
41}
