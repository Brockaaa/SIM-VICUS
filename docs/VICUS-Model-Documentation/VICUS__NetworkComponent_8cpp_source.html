34bool NetworkComponent::isValid(const Database<Schedule> &scheduleDB) const {
35 FUNCID(NetworkComponent::isValid);
36
37 NANDRAD::HydraulicNetworkComponent::ModelType nandradModelType = nandradNetworkComponentModelType(m_modelType);
38
39 // check standard parameter
40 std::vector<unsigned int> paraVec = NANDRAD::HydraulicNetworkComponent::requiredParameter(nandradModelType, 1);
41 for (unsigned int i: paraVec){
42 try {
43 NANDRAD::HydraulicNetworkComponent::checkModelParameter(m_para[i], i);
44 } catch (...) {
45 m_errorMsg = "Standard parameters of component are invalid.";
46 return false;
47 }
48 }
49
50 // check additional parameter
51 std::vector<unsigned int> paraVecAdd = additionalRequiredParameter(m_modelType);
52 for (unsigned int i: paraVecAdd){
53 try {
54 checkAdditionalParameter(m_para[i], i);
55 } catch (...) {
56 m_errorMsg = "Aditional parameters of component are invalid.";
57 return false;
58 }
59 }
60
61 // check integer parameter
62 std::vector<unsigned int> paraVecInt = requiredIntParameter(m_modelType);
63 for (unsigned int i: paraVecInt){
64 try {
65 checkIntParameter(m_intPara[i], i);
66 } catch (...) {
67 m_errorMsg = "Integer parameters of component are invalid.";
68 return false;
69 }
70 }
71
72 // check if there is the correct number of schedules and given schedules really exist
73 std::vector<std::string> reqSchedules = NANDRAD::HydraulicNetworkComponent::requiredScheduleNames(nandradModelType);
74 if (reqSchedules.size() != m_scheduleIds.size())
75 return false;
76 for (unsigned int id: m_scheduleIds){
77 const Schedule *sched = scheduleDB[id];
78 if (sched == nullptr) {
79 m_errorMsg = "Schedule is not set properly.";
80 return false;
81 }
82 // TODO: implement sched.isValid() ?
83 // problem: does currently not work properly for annual schedules
84 }
85
86 // check if required schedules are given
87 if (reqSchedules.size() != m_scheduleIds.size()) {
88 m_errorMsg = "Required schedules are not properly set.";
89 return false;
90 }
91
92 // pipe properties
93 if (hasPipeProperties(m_modelType) && m_pipePropertiesId == INVALID_ID) {
94 m_errorMsg = "Pipe properties are not set.";
95 return false;
96 }
97
98
99 try {
100 // check data table heat pumps
101 if (m_modelType == MT_HeatPumpOnOffSourceSide ) { //|| m_modelType == MT_HeatPumpOnOffSourceSideWithBuffer) {
102 if (m_polynomCoefficients.m_values.at("QdotCondensator").size() != 6)
103 throw IBK::Exception(IBK::FormatString("'%1' requires polynom coefficient parameter 'QdotCondensator' with exactly 6 values.")
104 .arg(KeywordList::Keyword("HydraulicNetworkComponent::ModelType", m_modelType)), FUNC_ID);
105 if (m_polynomCoefficients.m_values.at("ElectricalPower").size() != 6)
106 throw IBK::Exception(IBK::FormatString("'%1' requires polynom coefficient parameter 'ElectricalPower' with exactly 6 values.")
107 .arg(KeywordList::Keyword("HydraulicNetworkComponent::ModelType", m_modelType)), FUNC_ID);
108 }
109 if (m_modelType == MT_HeatPumpVariableSourceSide) {
110 if (m_polynomCoefficients.m_values.at("COP").size() != 6)
111 throw IBK::Exception(IBK::FormatString("'%1' requires polynom coefficient parameter 'COP' with exactly 6 values.")
112 .arg(KeywordList::Keyword("HydraulicNetworkComponent::ModelType", m_modelType)), FUNC_ID);
113 }
114
115 // check data table pumps:
116 // if there is a data table, it should have correct entries
117 // if correct entries exist, they should have correct number of parameters
118 if (m_modelType == MT_ConstantPressurePump || m_modelType == MT_ControlledPump || m_modelType == MT_VariablePressurePump) {
119 if (!m_polynomCoefficients.m_values.empty()) {
120 if (m_polynomCoefficients.m_values.find("MaximumElectricalPower") == m_polynomCoefficients.m_values.end() ||
121 m_polynomCoefficients.m_values.find("MaximumPressureHead") == m_polynomCoefficients.m_values.end() )
122 throw IBK::Exception(IBK::FormatString("'%1' data table should contain entries 'MaximumElectricalPower' and 'MaximumPressureHead'.")
123 .arg(KeywordList::Keyword("HydraulicNetworkComponent::ModelType", m_modelType)), FUNC_ID);
124 // data table should have correct number of parameters
125 if (m_polynomCoefficients.m_values.at("MaximumElectricalPower").empty() && m_polynomCoefficients.m_values.at("MaximumElectricalPower").size() != 3 )
126 throw IBK::Exception(IBK::FormatString("'%1' polynom coefficient parameter 'MaximumElectricalPower' should have exactly 3 values (quadratic polynom).")
127 .arg(KeywordList::Keyword("HydraulicNetworkComponent::ModelType", m_modelType)), FUNC_ID);
128 if (m_polynomCoefficients.m_values.at("MaximumPressureHead").empty() && m_polynomCoefficients.m_values.at("MaximumPressureHead").size() != 3 )
129 throw IBK::Exception(IBK::FormatString("'%1' polynom coefficient parameter 'MaximumPressureHead' should have exactly 3 values (quadratic polynom).")
130 .arg(KeywordList::Keyword("HydraulicNetworkComponent::ModelType", m_modelType)), FUNC_ID);
131 }
132 }
133
134 // for MT_ConstantPressurePump and MT_VariablePressurePump, we enforce existance of complete parameter set (pump efficiency,
135 // maximum pressure head and maximum electric power) once one of the parameters or maximum pressure head and maximum electric power
136 // is given
137 if (m_modelType == MT_ConstantPressurePump || m_modelType == MT_VariablePressurePump) {
138 // general case
139 if(!m_para[NANDRAD::HydraulicNetworkComponent::P_PumpMaximumElectricalPower].name.empty() &&
140 m_para[NANDRAD::HydraulicNetworkComponent::P_MaximumPressureHead].empty())
141 throw IBK::Exception("Missing paramneter 'MaximumPressureHead'!", FUNC_ID);
142 if(!m_para[NANDRAD::HydraulicNetworkComponent::P_MaximumPressureHead].name.empty() &&
143 m_para[NANDRAD::HydraulicNetworkComponent::P_PumpMaximumElectricalPower].empty())
144 throw IBK::Exception("Missing paramneter 'PumpMaximumElectricalPower'!", FUNC_ID);
145 }
146
147 } catch (IBK::Exception &ex) {
148 m_errorMsg = ex.what();
149 return false;
150 } catch (std::exception &ex) {
151 m_errorMsg = ex.what();
152 return false;
153 }
154
155 return true;
156}

159AbstractDBElement::ComparisonResult NetworkComponent::equal(const AbstractDBElement *other) const {
160
161 const NetworkComponent * otherNetComp = dynamic_cast<const NetworkComponent*>(other);
162 if (otherNetComp == nullptr)
163 return Different;
164
165 //check parameters
166 for (unsigned int i=0; i<NetworkComponent::NUM_P; ++i){
167 if (m_para[i] != otherNetComp->m_para[i])
168 return Different;
169 }
170 for (unsigned int i=0; i<NetworkComponent::NUM_IP; ++i){
171 if (m_intPara[i] != otherNetComp->m_intPara[i])
172 return Different;
173 }
174
175 if (m_modelType != otherNetComp->m_modelType)
176 return Different;
177
178 // check data table
179 if (m_polynomCoefficients != otherNetComp->m_polynomCoefficients)
180 return Different;
181
182 // check schedule ids
183 if (m_scheduleIds.size() != otherNetComp->m_scheduleIds.size())
184 return Different;
185 for (unsigned int i=0; i<m_scheduleIds.size(); ++i){
186 if (m_scheduleIds[i] != otherNetComp->m_scheduleIds[i])
187 return Different;
188 }
189
190 //check meta data
191 if (m_displayName != otherNetComp->m_displayName ||
192 m_color != otherNetComp->m_color ||
193 m_dataSource != otherNetComp->m_dataSource ||
194 m_manufacturer != otherNetComp->m_manufacturer ||
195 m_notes != otherNetComp->m_notes)
196 return OnlyMetaDataDiffers;
197
198 return Equal;
199}

226std::vector<unsigned int> NetworkComponent::optionalParameter(const NetworkComponent::ModelType modelType) {
227 // we use switch for maintanance reasons
228 switch (modelType) {
229 case MT_ConstantPressurePump:
230 case MT_VariablePressurePump:
231 return {P_MaximumPressureHead, P_PumpMaximumElectricalPower, P_FractionOfMotorInefficienciesToFluidStream};

252void NetworkComponent::checkAdditionalParameter(const IBK::Parameter & para, const unsigned int numPara) {
253 const char * enumName = "NetworkComponent::para_t";
254 const char * name = KeywordList::Keyword(enumName, (int)numPara);
255 const char * unit = KeywordList::Unit(enumName, (int)numPara);
256 switch ((para_t)numPara) {
257 case P_PipeLength:
258 para.checkedValue(name, unit, unit, 0, false, std::numeric_limits<double>::max(), true, nullptr);
259 break;
260 default: break;
261 }
262}

264
265void NetworkComponent::checkIntParameter(const IBK::IntPara & para, const unsigned int numPara) {
266 FUNCID(NetworkComponent::checkIntParameter);
267 const char * enumName = "NetworkComponent::para_t";
268 const char * name = KeywordList::Keyword(enumName, (int)numPara);
269 switch (numPara) {
270 case IP_NumberParallelPipes:
271 case IP_NumberParallelElements:
272 {
273 if (para.value < 1)
274 throw IBK::Exception(IBK::FormatString("% must be > 1").arg(name), FUNC_ID);
275 } break;
276 case NUM_IP: break;
277 }
278
279}
