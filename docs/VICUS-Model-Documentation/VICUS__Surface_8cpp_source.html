36void Surface::initializeColorBasedOnInclination() {
37 // default color for walls
38 m_displayColor = QColor(200,200,140,255);
39 const double angleForWalls = 0.35; // only very vertical planes are treated as walls
40 // Floor
41 if (m_geometry.normal().m_z < -angleForWalls)
42 m_displayColor = QColor("#566094");
43 // Roof
44 else if (m_geometry.normal().m_z > angleForWalls)
45 m_displayColor = QColor(150,50,20,255);
46}

49void Surface::updateParents() {
50 m_children.clear();
51 for (SubSurface & sub : m_subSurfaces) {
52 m_children.push_back(&sub);
53 sub.m_parent = this;
54 }
55 for (Surface &childSurf : m_childSurfaces) {
56 m_children.push_back(&childSurf);
57 childSurf.m_parent = this;
58 childSurf.updateParents();
59 }
60}

63void Surface::readXML(const TiXmlElement * element) {
64 FUNCID(Surface::readXML);
65 // read 3D geometry
66 VICUS::Polygon3D poly3D;
67 const TiXmlElement * c = element->FirstChildElement();
68 while (c) {
69 const std::string & cName = c->ValueStr();
70 if (cName == "Polygon3D") {
71 try {
72 poly3D.readXML(c);
73 } catch (...) {
74 // invalid polygons are just skipped, we don't want to raise an error here!
75 IBK::IBK_Message( IBK::FormatString(XML_READ_ERROR).arg(element->Row()).arg(
76 IBK::FormatString("Error reading Polygon3D tag.") ), IBK::MSG_WARNING, FUNC_ID);
77 return;
78 }
79 // remove Polygon3D element from parent, to avoid getting spammed with "unknown Polygon3D" warning
80 const_cast<TiXmlElement *>(element)->RemoveChild(const_cast<TiXmlElement *>(c));
81 break;
82 }
83 c = c->NextSiblingElement();
84 }
85
86 readXMLPrivate(element);
87 // copy polygon to plane geometry
88 std::vector<PlaneGeometry::Hole> holes;
89 for(const SubSurface & s : m_subSurfaces)
90 holes.push_back(PlaneGeometry::Hole(s.m_id, s.m_polygon2D, false));
91
92 for(const Surface & s : m_childSurfaces) {
93 const IBKMK::Vector3D &offset = poly3D.offset();
94 const IBKMK::Vector3D &localX = poly3D.localX();
95 const IBKMK::Vector3D &localY = poly3D.localY();
96
97 const std::vector<IBKMK::Vector3D> &vertexes = s.polygon3D().vertexes();
98 std::vector<IBKMK::Vector2D> holePoints(vertexes.size());
99
100 for(unsigned int j=0; j<vertexes.size(); ++j) {
101 IBKMK::planeCoordinates(offset, localX, localY, vertexes[j], holePoints[j].m_x, holePoints[j].m_y);
102 }
103
104 holes.push_back(PlaneGeometry::Hole(s.m_id, holePoints, true) );
105 }
106 // if we didn't get a Polygon3D element, the next call will throw an exception
107 m_geometry.setGeometry( poly3D, holes);
108}

134void Surface::setChildAndSubSurfaces(const std::vector<SubSurface> & subSurfaces,
135 const std::vector<Surface> & childSurfaces) {
136 m_subSurfaces = subSurfaces;
137 m_childSurfaces = childSurfaces;
138 std::vector<PlaneGeometry::Hole> holes;
139 for (const SubSurface & s : m_subSurfaces)
140 holes.push_back(PlaneGeometry::Hole(s.m_id, s.m_polygon2D, false));
141 for (const Surface & s : m_childSurfaces) {
142
143 const IBKMK::Vector3D &offset = geometry().offset();
144 const IBKMK::Vector3D &localX = geometry().localX();
145 const IBKMK::Vector3D &localY = geometry().localY();
146
147 const std::vector<IBKMK::Vector3D> &vertexes = s.polygon3D().vertexes();
148 std::vector<IBKMK::Vector2D> holePoints(vertexes.size());
149
150 for(unsigned int j=0; j<vertexes.size(); ++j) {
151 IBKMK::planeCoordinates(offset, localX, localY, vertexes[j], holePoints[j].m_x, holePoints[j].m_y);
152 }
153
154 holes.push_back(PlaneGeometry::Hole(s.m_id, holePoints, true) );
155 }
156 m_geometry.setHoles(holes);
157}

159void Surface::flip() {
160 m_geometry.flip(); // the hole polygons have been adjusted here already
161 IBK_ASSERT(m_subSurfaces.size() == m_geometry.holes().size());
162 for (unsigned int i=0, count=m_subSurfaces.size(); i<count; ++i)
163 m_subSurfaces[i].m_polygon2D = m_geometry.holes()[i].m_holeGeometry;
164}

170double Surface::areaWithoutSubsurfaces() const {
171 double area = m_geometry.area(3);
172 for(const VICUS::SubSurface & ss : m_subSurfaces){
173 area -= ss.m_polygon2D.area(3);
174 }
175 return area;
176}
