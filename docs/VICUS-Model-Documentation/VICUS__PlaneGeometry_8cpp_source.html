63 // a valid polygon has at least 1 vertex and valid normal and localX vectors

66 m_dirty = true;
67 }

77bool PlaneGeometry::isValid() const {
78 if (!m_polygon.isValid())
79 return false;
80 if (m_dirty)
81 triangulate();
82
83 return !m_triangulationData.m_triangles.empty();

87double PlaneGeometry::inclination(int digits) const {
88 double inc = std::acos(normal().m_z) / IBK::DEG2RAD;
89
90 inc = std::round(inc*IBK::f_pow10(digits))/IBK::f_pow10(digits);
91 return inc;
92}

95double PlaneGeometry::orientation(int digits) const {
96 double val = 90 - std::atan2(normal().m_y, normal().m_x) / IBK::DEG2RAD;
97 if (val<0)
98 val += 360;
99
100 // round value to n digits
101 val = std::round(val*IBK::f_pow10(digits))/IBK::f_pow10(digits);
102
103 return val;
104}

106double PlaneGeometry::area(int digits) const {
107 double area = 0.0;
108 for(const VICUS::PlaneGeometry::Hole &h : m_holes) {
109 if(h.m_isChildSurface)
110 area -= h.m_holeGeometry.area(digits);
111 }
112 return area + m_polygon.polyline().area(digits);
113}

116void PlaneGeometry::triangulate() const {
117 m_triangulationData.clear();
118 m_triangulationDataWithoutHoles.clear();
119 m_holeTriangulationData.clear();
120 m_dirty = false;
121
122 // only continue, if the polygon itself is valid
123 if (!m_polygon.isValid())
124 return;
125
126 // Create a copy of the vertex error of the outer polygon (we are going to add vertexes of the holes later on)
127 // Note: do not access m_vertexes herea as we may regenerate the vertexes.
128 std::vector<IBKMK::Vector3D> vertexes = m_polygon.vertexes();
129
130 // We have special handling for type triangle and rectangle, but only if
131 // we have no holes. This will save some work for quite a lot of planes.
132 if (m_holes.empty() && m_polygon.type() != VICUS::Polygon2D::T_Polygon) {
133 switch (m_polygon.type()) {
134
135 case Polygon2D::T_Triangle :
136 m_triangulationData.m_triangles.push_back( IBKMK::Triangulation::triangle_t(0, 1, 2) );
137 break;
138
139 case Polygon2D::T_Rectangle :
140 m_triangulationData.m_triangles.push_back( IBKMK::Triangulation::triangle_t(0, 1, 2) );
141 m_triangulationData.m_triangles.push_back( IBKMK::Triangulation::triangle_t(2, 3, 0) );
142 break;
143
144 default: ;
145 }
146 m_triangulationData.m_vertexes.swap(vertexes);
147 m_triangulationData.m_normal = m_polygon.normal(); // cache normal for easy access
148
149 // no holes, copy data
150 m_triangulationDataWithoutHoles.m_vertexes = m_triangulationData.m_vertexes;
151 m_triangulationDataWithoutHoles.m_normal = m_polygon.normal();
152
153 return; // done
154 }
155
156
157 // now the generic code
158
159 const std::vector<IBKMK::Vector2D> &parentPoly = m_polygon.polyline().vertexes();
160 // process all holes and check if they are valid (i.e. their polygons do not intersect our polygons
161
162 // here we store the state of the sub-surface polygon: 0 - invalid, 1 - partially valid, 2 - completely valid
163 // polygons with status 1 are drawn but not used in triangulation of polygon3D
164 std::vector<unsigned int> polygonStatus(m_holes.size(), 0);
165 for (unsigned int i=0; i<m_holes.size(); ++i) {
166 const Polygon2D & p2 = m_holes[i].m_holeGeometry;
167 const std::vector<IBKMK::Vector2D> &subSurfacePoly = p2.vertexes();
168
169 // check if any of the holes are invalid
170 // TODO : Dirk, invalide Polygone einfach ignorieren und nicht zeichnen? Oder als "outline" zeichnen?
171 // Damit man diese "invaliden" Polygonfl√§chen noch anfassen kann und vielleicht zur√ºck in die
172 // Konstruktion schieben kann, m√ºsste man diese trotzdem triangulieren. Vorschlag:
173 // Unterscheiden zwischen komplett ung√ºltig (verwunden, nur 2 nicht-kollinieare Punkte, ...)
174 // und korrigierbar falsch ("au√üerhalb des Polygons" oder "√ºberlappend mit anderem Fenster").
175 // Letztere k√∂nnte man zun√§chst auch noch zeichnen, sollte dann aber den "invalid" Zustand irgendwie
176 // visualisiert bekommen (und beim NANDRAD -Export vielleicht als Fehler angezeigt bekommen?).
177
178
179 // skip invalid polygons alltogether
180 if (!p2.isValid()) {
181 polygonStatus[i] = 0;
182 continue;
183 }
184
185
186 // First check: test all vertexes of polygon;
187 // - if outside outer polygon, sub-surface is (partially) outside polygon
188
189 bool valid = true;
190 unsigned int pointCountHole = subSurfacePoly.size();
191 for (unsigned int j=0; j<pointCountHole; ++j) {
192 if (IBKMK::pointInPolygon(parentPoly, subSurfacePoly[j]) == -1) {
193 valid = false;
194 break;
195 }
196 }
197
198 if (!valid) {
199 polygonStatus[i] = 1;
200 continue;
201 }
202
203 // Second check: test intersections of all sub-surface polygon edges with edges of outer polygon
204 for (unsigned int j=0; j<pointCountHole; ++j){
205 const IBKMK::Vector2D &point1 = subSurfacePoly[j];
206 const IBKMK::Vector2D &point2 = subSurfacePoly[(j+1) % pointCountHole];
207 IBKMK::Vector2D intersectP;
208 if (IBKMK::intersectsLine2D(parentPoly, point1, point2, intersectP)) {
209 // if intersection point is close to first or second point of line, polygon is ok
210 const double EPS = 1e-4;
211 if (! (intersectP.distanceTo(point1) < EPS || intersectP.distanceTo(point2) < EPS) )
212 {
213 // polygon is also ok, if intersection point is "on" the outer polygon
214 if (IBKMK::pointInPolygon(parentPoly, subSurfacePoly[j]) != 0) {
215 valid = false;
216 break;
217 }
218 }
219 }
220
221 }
222 if (!valid) {
223 polygonStatus[i] = 1;
224 continue;
225 }
226
227 // TODO Dirk, check intersections with other sub-surfaces
228
229 // all checks done
230 polygonStatus[i] = 2;
231 }
232
233 // now populate global vertex vector and generate 2D polygon
234 // we have then two vectors, one with the 3D vertexes, and one with the 2D vertexes inside the plane
235
236 // first add the polygon points
237 std::vector<IBKMK::Vector2D> points = m_polygon.polyline().vertexes();
238
239 // Vector 'points' contains now the 2D coordinates for all 3D coordinates in 'vertexes' with same order.
240
241 // now generate the edges for this polygon
242 std::vector<std::pair<unsigned int, unsigned int> > edges;
243 for (unsigned int i=0; i<points.size(); ++i)
244 edges.push_back(std::make_pair(i, (i+1) % points.size()));
245
246
247 // we now have the points of the outer polygon in vector 'points' and the outer edges defined in 'edges'
248
249
250 // now process holes/windows
251 m_holeTriangulationData.resize(m_holes.size()); // create an (empty) data structure for each hole
252
253 // loop all holes/windows
254 for (unsigned int holeIdx = 0; holeIdx < polygonStatus.size(); ++holeIdx) {
255 // skip all invalid polygons; polygons partially or fully outside the outer polygon
256 // are still being triangulated (and show), but they points are not used for the triangulation
257 // of the outer polygon with holes
258 if (polygonStatus[holeIdx] == 0)
259 continue;
260
261 const Polygon2D & p2 = m_holes[holeIdx].m_holeGeometry; // polygon of currently processed hole
262
263 std::vector<unsigned int> vertexIndexes; // mapping table, relates hole index to global index in 'points' vector
264 // process all vertexes and compose vector with 2D and 3D coordinates of the hole polygon
265 std::vector<IBKMK::Vector2D> matchedHolePoints;
266 std::vector<IBKMK::Vector3D> matchedHolePoints3D;
267 const std::vector<IBKMK::Vector2D> & holePoints = p2.vertexes();
268 for (const IBKMK::Vector2D & v : holePoints) {
269 // for each vertex in window do:
270 // - check if vertex is already in vertex list, then re-use same vertex index,
271 // otherwise add vertex and get new index
272 unsigned int j=0;
273 for (; j<points.size(); ++j)
274 if (points[j].similar(v, 1e-6))
275 break;
276 // store index (either of existing vertex or next vertex to be added)
277 vertexIndexes.push_back(j);
278 if (j == points.size()) {
279 points.push_back(v);
280 // compute the matching 3D vertex and add to list of vertexes
281 IBKMK::Vector3D v3 = offset() + localX()*v.m_x + localY()*v.m_y;
282 vertexes.push_back(v3);
283 }
284 matchedHolePoints.push_back(points[j]);
285 matchedHolePoints3D.push_back(vertexes[j]);
286 }
287
288 // now the 'points' vector holds vertexes of the outer polygon and all the holes
289
290// IBK::IBK_Message("Edges\n", IBK::MSG_PROGRESS, "");
291 // add edges
292 std::vector<std::pair<unsigned int, unsigned int> > holeOnlyEdges;
293 for (unsigned int i=0, vertexCount = p2.vertexes().size(); i<vertexCount; ++i) {
294 unsigned int i1 = vertexIndexes[i];
295 unsigned int i2 = vertexIndexes[(i+1) % vertexCount];
296 // add edge to global edge list (for outer polygon), but only, if polygon is entirely valid
297 if (polygonStatus[holeIdx] == 2) {
298 edges.push_back(std::make_pair(i1, i2));
299 // IBK::IBK_Message(IBK::FormatString(" edges[%1] = (%2, %3)\n").arg(i).arg(i1).arg(i2), IBK::MSG_PROGRESS, "");
300 }
301
302 // add edge to vector with only hole edges (numbered from 0...vertexCount-1)
303 holeOnlyEdges.push_back(std::make_pair(i, (i+1) % vertexCount));
304 }
305
306 // Example: outer polygon is defined through vertexes 0...3
307 // hole is defined through vertexes 4...7
308 // 'points' holds vertexes 0...7
309 // 'edges' holds all edges (of outer polygon and inner polyons) and uses all vertexes
310 // 'holeOnlyEdges' holds only edges of hole: 0-1, 1-2, 2-3, 3-0
311 // 'vertexIndexes' maps hole-vertex-index to global index:
312 // vertexIndexes[0] = 4
313 // vertexIndexes[1] = 5
314 // vertexIndexes[2] = 6
315 // vertexIndexes[3] = 7
316
317 // now do the triangulation of hole alone
318 IBKMK::Triangulation triangu;
319
320 // Note: we give the global points vector and all edgs of all holes
321 // IBK::point2D<double> is a IBKMK::Vector2D
322 triangu.setPoints(reinterpret_cast< const std::vector<IBK::point2D<double> > & >(matchedHolePoints), holeOnlyEdges);
323
324 // and copy the triangle data and remap the points
325 for (auto tri : triangu.m_triangles) {
326 if (tri.isDegenerated()) // protect against -1 vertex indexes
327 continue;
328 m_holeTriangulationData[holeIdx].m_triangles.push_back(tri);
329 }
330 m_holeTriangulationData[holeIdx].m_vertexes.swap(matchedHolePoints3D);
331 m_holeTriangulationData[holeIdx].m_normal = m_polygon.normal(); // cache normal for easy access
332 }
333
334 // now generate the triangulation data for the entire surface, without subsurface polygons
335
336#if 0
337 for (auto p : points)
338 IBK::IBK_Message(IBK::FormatString(" points.push_back( IBK::point2D<double>(%1,%2));\n")
339 .arg(p.m_x).arg(p.m_y), IBK::MSG_PROGRESS, "");
340 for (auto e : edges)
341 IBK::IBK_Message(IBK::FormatString(" edges.push_back(std::make_pair(%1, %2));\n").arg(e.first).arg(e.second), IBK::MSG_PROGRESS, "");
342#endif
343
344 IBKMK::Triangulation triangu;
345 // Note: IBK::point2D<double> is a IBKMK::Vector2D
346 triangu.setPoints(reinterpret_cast< const std::vector<IBK::point2D<double> > & >(points), edges);
347
348// IBK::IBK_Message("Triangulation\n", IBK::MSG_PROGRESS, "");
349 for (auto tri : triangu.m_triangles) {
350 // skip degenerated triangles
351 if (tri.isDegenerated())
352 continue;
353 m_triangulationData.m_triangles.push_back(tri);
354// IBK::IBK_Message(IBK::FormatString("%1, %2, %3\n").arg(tri.i1).arg(tri.i2).arg(tri.i3), IBK::MSG_PROGRESS, "");
355 }
356
357 // store vertexes for triangles
358 m_triangulationData.m_vertexes.swap(vertexes);
359 m_triangulationData.m_normal = m_polygon.normal(); // cache normal for easy access
360
361 Q_ASSERT(m_holes.size() == m_holeTriangulationData.size());
362}

365bool PlaneGeometry::intersectsLine(const IBKMK::Vector3D & p1, const IBKMK::Vector3D & d, IBKMK::Vector3D & intersectionPoint,
366 double & dist, int & holeIndex, bool hitBackfacingPlanes, bool endlessPlane) const
367{
368 holeIndex = -1;
369 // We need to guard against invalid geometry
370 if (!isValid())
371 return false;
372
373 // first the normal test
374
375 double d_dot_normal = d.scalarProduct(normal());
376 double angle = d_dot_normal/d.magnitude();
377 // line parallel to plane? no intersection
378 if (angle < 1e-8 && angle > -1e-8)
379 return false;
380
381 // Condition 1: same direction of normal vectors?
382 if (!hitBackfacingPlanes && angle >= 0)
383 return false; // no intersection possible
384
385 const IBKMK::Vector3D & planeOffset = offset();
386 double t = (planeOffset - p1).scalarProduct(normal()) / d_dot_normal;
387
388 // Condition 2: outside viewing range?
389 if (t < 0 || t > 1)
390 return false;
391
392 // now determine location on plane
393 IBKMK::Vector3D x0 = p1 + t*d;
394
395 // plane is endless - return intersection point and normalized distance t (no hole checking here!)
396 if (endlessPlane) {
397 intersectionPoint = x0;
398 dist = t;
399 return true;
400 }
401
402 // test if intersection point is inside our plane
403 // we have a specialized variant for triangles and rectangles, but only if there are no holes
404 if (m_holes.empty() && m_polygon.type() != Polygon2D::T_Polygon) {
405
406 switch (m_polygon.type()) {
407 case Polygon2D::T_Triangle :
408 case Polygon2D::T_Rectangle : {
409
410 // we have three possible ways to get the intersection point, try them all until we succeed
411 const IBKMK::Vector3D & a = m_polygon.vertexes()[1] - m_polygon.vertexes()[0];
412 const IBKMK::Vector3D & b = m_polygon.vertexes().back() - m_polygon.vertexes()[0];
413 double x,y;
414 if (!IBKMK::planeCoordinates(planeOffset, a, b, x0, x, y, VICUS_PLANE_PROJECTION_TOLERANCE))
415 return false;
416
417 if (m_polygon.type() == Polygon2D::T_Triangle && x >= 0 && x+y <= 1 && y >= 0) {
418 intersectionPoint = x0;
419 dist = t;
420 return true;
421 }
422 else if (m_polygon.type() == Polygon2D::T_Rectangle && x >= 0 && x <= 1 && y >= 0 && y <= 1) {
423 intersectionPoint = x0;
424 dist = t;
425 return true;
426 }
427
428 } break;
429 default: ;
430 }
431 return false;
432 }
433
434 // compute 2D coordinates in local plane - if that fails, bail out (can happen when line is parallel to plane)
435 double x,y;
436 if (!IBKMK::planeCoordinates(planeOffset, localX(), localY(), x0, x, y, VICUS_PLANE_PROJECTION_TOLERANCE))
437 return false;
438
439 IBK::point2D<double> intersectionPoint2D(x,y);
440
441 // first check if we hit a hole
442 for (unsigned int j=0; j<m_holes.size(); ++j) {
443 // - if hole is valid, check if we are inside the polygon
444 if (m_holes[j].m_holeGeometry.isValid()) {
445 // run point in polygon algorithm
446 if (IBKMK::pointInPolygon(m_holes[j].m_holeGeometry.vertexes(), intersectionPoint2D) != -1) {
447 holeIndex = (int)j;
448 intersectionPoint = x0;
449 dist = t;
450 return true;
451 }
452 }
453 }
454
455 // otherwise we might still hit the rest of the polygon
456 if (IBKMK::pointInPolygon(polygon2D().vertexes(), intersectionPoint2D ) != -1) {
457 dist = t;
458 intersectionPoint = x0;
459 return true;
460 }
461 else
462 return false;
463}

487void PlaneGeometry::flip() {
488 // flip the polygon first
489 m_polygon.flip();
490 // since localX and localY axes have been swapped, we need to swap all hole coordinates as well
491 for (PlaneGeometry::Hole & h : m_holes) {
492 std::vector<IBKMK::Vector2D> vertexes2D = h.m_holeGeometry.vertexes();
493 for (IBKMK::Vector2D & v : vertexes2D)
494 std::swap(v.m_x, v.m_y);
495 h.m_holeGeometry.setVertexes(vertexes2D);
496 }
497}

499void PlaneGeometry::changeOrigin(unsigned int idx) {
500 const std::vector<IBKMK::Vector3D> &verts = m_polygon.vertexes();
501
502 if(verts.size()<3 || idx > verts.size()-1)
503 return;
504
505 std::vector<IBKMK::Vector3D> newVerts;
506 newVerts.insert(newVerts.begin(), verts.begin() + idx, verts.end());
507 newVerts.insert(newVerts.end(), verts.begin(), verts.begin() + idx);
508
509 m_polygon.setVertexes(newVerts);
510}
