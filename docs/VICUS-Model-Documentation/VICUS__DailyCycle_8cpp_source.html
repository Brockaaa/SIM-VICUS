31bool DailyCycle::isValid() const {
32
33 if (m_dayTypes.empty() || m_dayTypes.size()> 7+1) { //Mo, ..., Sun + Holiday
34 m_errorMsg = "Daily cycle day types are not valid.";
35 return false; // day types not valid
36 }
37
38 for (int dt : m_dayTypes) {
39 switch ((NANDRAD::Schedule::ScheduledDayType)dt) {
40 case NANDRAD::Schedule::ST_MONDAY:
41 case NANDRAD::Schedule::ST_TUESDAY:
42 case NANDRAD::Schedule::ST_WEDNESDAY:
43 case NANDRAD::Schedule::ST_THURSDAY:
44 case NANDRAD::Schedule::ST_FRIDAY:
45 case NANDRAD::Schedule::ST_SATURDAY:
46 case NANDRAD::Schedule::ST_SUNDAY:
47 case NANDRAD::Schedule::ST_HOLIDAY:
48 break;
49
50 default: {
51 m_errorMsg = "Daily cycle day types are not valid.";
52 return false; // day type is not valid
53 }
54 }
55 }
56
57 if (m_values.size() != m_timePoints.size() || m_values.empty()) {
58 m_errorMsg = "Daily cycle has empty values or size of timepoints and values do not match.";
59 return false; // empty values or size not same of values and timepoints
60 }
61
62 if (m_timePoints[0] != 0.0) {
63 m_errorMsg = "Daily cycle start point is not zero.";
64 return false; // start point is not 0
65 }
66
67 // checked time points vector and time point and value vectors have same size
68 return true;
69}

79DailyCycle DailyCycle::multiply(const DailyCycle &other) const {
80 FUNCID(DailyCycle::multiply);
81
82 DailyCycle dc;
83
84 if (!isValid() || !other.isValid())
85 throw IBK::Exception("Both daily cycles must be valid in order to be multiplied.", FUNC_ID);
86
87 // Only keep equal day types
88 for (unsigned int i=0; i<m_dayTypes.size(); ++i){
89 for (unsigned int j=0; j<other.m_dayTypes.size(); ++j){
90 if (m_dayTypes[i] == other.m_dayTypes[j])
91 dc.m_dayTypes.push_back(m_dayTypes[i]);
92 }
93 }
94 if (dc.m_dayTypes.empty())
95 return dc;
96
97 // make a copy of the other daily cycle to the new daily cycle
98 dc.m_timePoints = other.m_timePoints;
99 dc.m_values = other.m_values;
100
101 // compare timepoints
102 unsigned int j=0;
103 unsigned int i=0;
104 while (i<dc.m_timePoints.size()) {
105
106 double timePoint = m_timePoints[j];
107 double timePoint2 = dc.m_timePoints[i];
108 double val = dc.m_values[i];
109 if(IBK::nearly_equal<3>(timePoint, timePoint2)){
110 if(j+1 >= m_timePoints.size())
111 //fertig
112 break;
113 //multi
114 //dc.m_values[i] *= m_values[j];
115 ++j;
116 }
117 else if(timePoint > timePoint2){
118 bool haveNext = i+1<dc.m_timePoints.size();
119
120 //insert a time point from this daily cycle in the new daily cycle
121 if((haveNext && timePoint < dc.m_timePoints[i+1]) || !haveNext){
122 dc.m_timePoints.insert(dc.m_timePoints.begin()+i+1, timePoint);
123 dc.m_values.insert(dc.m_values.begin()+i+1, val);
124 }
125 ++i;
126 }
127 }
128
129 //multi
130 i=0;
131 j=0;
132 //now daily cycle (dc) holds all start times (interval start point)
133 //iterate dc time points and check for same or lower time points in dc
134 //if true multiply
135 //otherwise increment
136 while (i<dc.m_timePoints.size()){
137 double timePoint = m_timePoints[j];
138 double timePoint2 = dc.m_timePoints[i];
139
140 if(IBK::nearly_equal<3>(timePoint, timePoint2) ||
141 (timePoint2 > timePoint &&
142 j+1<m_timePoints.size() &&
143 timePoint2 < m_timePoints[j+1]) ||
144 j+1 >= m_timePoints.size()){
145
146 dc.m_values[i] *= m_values[j];
147 ++i;
148 }
149 else if(j+1<m_timePoints.size() &&
150 timePoint2 >= m_timePoints[j+1]){
151 ++j;
152 }
153 }
154
155 return dc;
156}

159DailyCycle DailyCycle::multiply(double factor) const {
160 FUNCID(DailyCycle::multiply);
161
162 if (!isValid())
163 throw IBK::Exception("Daily cycle must be valid in order to be multiplied.", FUNC_ID);
164
165 DailyCycle dc(*this);
166

168 dc.m_values[i] *= factor;

255DailyCycle DailyCycle::add(double val) const{
256 FUNCID(DailyCycle::add);
257
258 if (!isValid())
259 throw IBK::Exception("Daily cycle must be valid in order to add values to it.", FUNC_ID);
260
261 DailyCycle dc(*this);
262
263 for (unsigned int i=0; i<dc.m_values.size(); ++i)
264 dc.m_values[i] += val;
265
266 return dc;
