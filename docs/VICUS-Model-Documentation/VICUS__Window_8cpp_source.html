32AbstractDBElement::ComparisonResult Window::equal(const AbstractDBElement *other) const {
33 const Window * otherWin = dynamic_cast<const Window*>(other);
34 if (otherWin == nullptr)
35 return Different;
36
37 //first check critical data
38
39 //check parameters
40 if (m_idGlazingSystem != otherWin->m_idGlazingSystem ||
41 m_methodDivider!= otherWin->m_methodDivider||
42 m_methodFrame!= otherWin->m_methodFrame ||
43 m_frame != otherWin->m_frame ||
44 m_divider != otherWin->m_divider)
45 return Different;
46
47 for(unsigned int i=0; i<NUM_P; ++i){
48 if(m_para[i] != otherWin->m_para[i])
49 return Different;
50 }
51
52 //check meta data
53
54 if (m_displayName != otherWin->m_displayName ||
55 m_dataSource != otherWin->m_dataSource ||
56 m_notes != otherWin->m_notes ||
57 m_color != otherWin->m_color)
58 return OnlyMetaDataDiffers;
59
60 return Equal;
61}

64bool Window::isValid() const {
65 if(m_id == INVALID_ID) {
66 m_errorMsg = "Window does not exist.";
67 return false;
68 }
69
70 if(m_idGlazingSystem == INVALID_ID) {
71 m_errorMsg = "Glazing system of window does not exist.";
72 return false;
73 }
74
75 double fractionFrame = 0;
76 switch (m_methodFrame) {
77 case VICUS::Window::M_None:
78 break;
79 case VICUS::Window::M_Fraction:{
80 try {
81 fractionFrame = m_para[P_FrameFraction].checkedValue(KeywordList::Keyword("Window::para_t", P_FrameFraction),
82 "---", "---", 0, false, 1, false, nullptr);
83 if(!m_frame.isValid()) {
84 m_errorMsg = "Window frame fraction is not valid.";
85 return false;
86 }
87 } catch (IBK::Exception &ex) {
88 m_errorMsg = ex.what();
89 return false;
90 }
91
92 }
93 break;
94 case VICUS::Window::M_ConstantWidth:{
95 try {
96 m_para[P_FrameWidth].checkedValue(KeywordList::Keyword("Window::para_t", P_FrameWidth),
97 "m", "m", 0, false, 2, true, nullptr);
98 if(!m_frame.isValid()) {
99 m_errorMsg = "Window frame width is not valid.";
100 return false;
101 }
102 } catch (IBK::Exception &ex) {
103 m_errorMsg = ex.what();
104 return false;
105 }
106
107 }
108 break;
109 case VICUS::Window::NUM_M: {
110 m_errorMsg = "Window frame method is not set.";
111 return false;
112 }
113 }
114 switch (m_methodDivider) {
115 case VICUS::Window::M_None:
116
117 break;
118 case VICUS::Window::M_Fraction:{
119 try {
120
121 //also check the sum of the area fractions
122 // only possible if we have two fraction parts
123 double maxVal = 1 - fractionFrame;
124 m_para[P_DividerFraction].checkedValue(KeywordList::Keyword("Window::para_t", P_DividerFraction),
125 "---", "---", 0, false, maxVal, false, nullptr);
126 if(!m_divider.isValid()) {
127 m_errorMsg = "Window divider fraction is not valid.";
128 return false;
129 }
130
131 } catch (IBK::Exception &ex) {
132 m_errorMsg = "Window divider fraction: " + std::string(ex.what());
133 return false;
134 }
135 }
136 break;
137 case VICUS::Window::M_ConstantWidth:{
138 try {
139 m_para[P_DividerWidth].checkedValue(KeywordList::Keyword("Window::para_t", P_DividerWidth),
140 "m", "m", 0, false, 2, true, nullptr);
141 if(!m_divider.isValid()) {
142 m_errorMsg = "Window divider width is not valid.";
143 return false;
144 }
145 } catch (IBK::Exception &ex) {
146 m_errorMsg = "Window divider width: " + std::string(ex.what());
147 return false;
148 }
149 }
150 break;
151 case VICUS::Window::NUM_M:
152 return false;
153 }
154
155
156
157
158 return true;
159}

162bool Window::calculateUValue(double & UValue, const VICUS::Database<Material> & /*materials*/,
163 const VICUS::Database<WindowGlazingSystem> & glazingSystems,
164 double ri, double re) const
165{
166 UValue = -1;
167 if (m_idGlazingSystem == VICUS::INVALID_ID)
168 return false;
169
170 const WindowGlazingSystem *glSys = glazingSystems[m_idGlazingSystem];
171
172 if (glSys == nullptr)
173 return false;
174
175 UValue = 1/(1/glSys->uValue() + ri + re);
176
177 // TODO : include frame in U-value calculation
178 return true;
179}
