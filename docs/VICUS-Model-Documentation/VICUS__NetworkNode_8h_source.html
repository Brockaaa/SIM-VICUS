56 enum NodeType {
57 NT_SubStation, // Keyword: SubStation
58 NT_Mixer, // Keyword: Mixer
59 NT_Source, // Keyword: Source
60 NUM_NT
61 };

63 NetworkNode() = default;
64
65 NetworkNode(const unsigned id, const NodeType type, const IBKMK::Vector3D &v, const double heatDemand=0):
66 m_position(v),
67 m_type(type),
68 m_maxHeatingDemand("MaxHeatingDemand", heatDemand, "W")
69 {
70 m_id = id;
71 }

73 /*! check connectivity of graph trhough recursive search */
74 void collectConnectedEdges(std::set<const NetworkNode*> & connectedNodes, std::set<const NetworkEdge*> & connectedEdge) const;

76 void setInletOutletNode(std::set< const NetworkNode*> & visitedNodes, std::vector<const NetworkEdge*> & orderedEdges) const;
77
78 /*! updates m_isDeadEnd. If node has less than two neighbours which are not a deadEnd and node is not a building
79 * nor a source: m_isDeadEdnd = true */
80 void updateIsDeadEnd();
81
82 /*! Caution: for some applications this definition may needs to be more precise
83 * e.g. compare types of connected edges */
84 bool isRedundant() const{
85 return (m_edges.size()==2);
86 }
87
88 /*! Only callable if node has exactly two edges: return the edge which is not the given edge */
89 NetworkEdge * neighborEdge(const NetworkEdge * e) const;
90
91 /*! get a set of all redundant nodes in the graph */
92 void findRedundantNodes(std::set<unsigned> & redundantNodes, std::set<const NetworkEdge*> & visitedEdges) const;
93
94 /*! looking from this node in the direction of the given edgeToVisit: return the next node that is not redundant
95 * and the distance to this node */
96 const NetworkNode * findNextNonRedundantNode(std::set<unsigned> & redundantNodes, double & distance, const NetworkEdge* edgeToVisit) const;
97
98 /*! simple algorithm to find the path from this node to the node of type NT_SOURCE.
99 * The path is stored as a set of edges */
100 bool findPathToSource(std::set<NetworkEdge*> &path, std::set<NetworkEdge*> &visitedEdges, std::set<unsigned> &visitedNodes);
101
102 /*! used for dijkstra algorithm. Look at all neighbour nodes: if the m_distanceToStart of this node + the distance to the neighbour
103 * is shorter than the current m_distanceToStart of the neighbour, update it. This makes sure the neighbour nodes have assigned
104 * the currently smallest distance from start */
105 void updateNeighbourDistances() const;
106
107 /*! used for dijkstra algorithm. appends the edge which leads to the predecessor node to path and calls itself for the predecessor node
108 * until a node without predecessor is reached. this way the path from a building to the source can be created, if the predecessors have been set */
109 void pathToNull(std::vector<NetworkEdge * > & path) const;
110
111 /*! looks at all adjacent nodes to find a node which has a heating demand >0 and returns it. */
112 double adjacentHeatingDemand(std::set<NetworkEdge*> visitedEdges);

131 NANDRAD::HydraulicNetworkHeatExchange m_heatExchange; // XML:E

139 /*! The radius used for the visualization of this node in the 3D scene [m].

141 */
142 mutable double m_visualizationRadius;
143
144 /*! Color to be used for displaying (visible) nodes. */
145 mutable QColor m_color;
146
147 /*! Used in dijkstra algorithm. */
148 mutable double m_distanceToStart = (std::numeric_limits<double>::max)();
149 mutable const NetworkNode * m_predecessor = nullptr;
150
151 /*! Defines wether this node is a dead end. */
152 mutable bool m_isDeadEnd = false;
