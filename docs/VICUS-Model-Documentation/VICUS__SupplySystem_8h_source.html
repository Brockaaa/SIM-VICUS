50/*! Defines the base structure of a generic network:
51 A generic network may be generated for surface heating FMU export. We bundle
52 all heatings by a hydraulically balanced network and support FMI-coupling
53 at a single demand network branch.
54*/
55class SupplySystem : public AbstractDBElement {
56 Q_DECLARE_TR_FUNCTIONS(SupplySystem)
57public:
58
59 /*! Supplier supported by the generic network model. */
60 enum SupplyType {
61 ST_StandAlone, // Keyword: StandAlone 'Stand-alone mode with given mass flux and suppply temperature'
62 ST_SubNetwork, // Keyword: SubNetwork 'User defined sub network'
63 ST_DatabaseFMU, // Keyword: DatabaseFMU 'Supply FMU loaded from a database and parametrized by the user'
64 ST_UserDefinedFMU, // Keyword: UserDefinedFMU 'User defined supply FMU'
65 NUM_ST
66 };
67
68 /*! Network parameters.
69 */
70 enum para_t {
71 P_MaximumMassFlux, // Keyword: MaximumMassFlux [kg/s] 'Maximum mass flux into the network, needed for pump control'
72 P_SupplyTemperature, // Keyword: SupplyTemperature [C] 'Constant supply temeprature'
73 P_MaximumMassFluxFMU, // Keyword: MaximumMassFluxFMU [kg/s] 'Maximum mass flux towards the building.'
74 P_HeatingPowerFMU, // Keyword: HeatingPowerFMU [W] 'Procuder heating power'
75 NUM_P
76 };
