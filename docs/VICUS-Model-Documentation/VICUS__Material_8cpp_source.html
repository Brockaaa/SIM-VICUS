74AbstractDBElement::ComparisonResult Material::equal(const AbstractDBElement * other) const {
75 const Material * otherMaterial = dynamic_cast<const Material*>(other);
76 if (otherMaterial == nullptr)
77 return Different;
78
79 for (unsigned int i=0; i<NUM_P; ++i) {
80 if (m_para[i] != otherMaterial->m_para[i])
81 return Different;
82 }
83
84 if (m_idEpds != otherMaterial->m_idEpds)
85 return Different;
86
87 // check meta data
88
89 if (m_displayName != otherMaterial->m_displayName ||
90 m_category != otherMaterial->m_category ||
91 m_notes != otherMaterial->m_notes ||
92 m_dataSource != otherMaterial->m_dataSource ||
93 m_manufacturer != otherMaterial->m_manufacturer)
94 return OnlyMetaDataDiffers;
95
96 // Note: color differences do not count as difference
97
98 return Equal;
99}
