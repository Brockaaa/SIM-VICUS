51 /*! Model types supported by this model. */

58 /*! Defines what kind of zone is connected via this boundary condition.
59 If 'Constant' or 'Scheduled' are used when neighbouring zones are not calculated themselves
60 but their temperature is known. For ground zones, i.e. in case of soil contact to the construction
61 the heat exchange coeff. should be very high 1000 W/m2K to neglect the heat transfer resistance.
62 */
63 enum OtherZoneType {
64 OZ_Standard, // Keyword: Standard 'Active zone or outside'
65 OZ_Constant, // Keyword: Constant 'Zone/ground with constant temperature'
66 OZ_Scheduled, // Keyword: Scheduled 'Zone/ground with scheduled temperature'
67 NUM_OZ
68 };
69
70 /*! Parameters. */
71 enum para_t {
72 /*! Constant heat transfer coefficient [W/m2K] */
73 P_HeatTransferCoefficient, // Keyword: HeatTransferCoefficient [W/m2K] 'Convective heat transfer coefficient'
74 /*! Reference to a const temperature of neighboring zone (required for OZ_Constant). */
75 P_ConstTemperature, // Keyword: ConstTemperature [C] 'Constant temperature of other zone/ground'
76 NUM_P
77 };
