30bool Component::isValid(const VICUS::Database<Material> & materials, const VICUS::Database<Construction> & constructions,
31 const VICUS::Database<BoundaryCondition> & bcs, const VICUS::Database<Schedule> & scheduleDB) const
32{
33 try {
34
35 // referenced construction instance exists and is valid?
36 const Construction * con = constructions[m_idConstruction];
37 if (con == nullptr) {
38 m_errorMsg = "Construction is not set.";
39 return false;
40 }
41
42 if (!con->isValid(materials)) {
43 m_errorMsg = "Construction '" + con->m_displayName.string("de", true) + "' is not valid.";
44 return false;
45 }
46
47 const BoundaryCondition *bcA = bcs[m_idSideABoundaryCondition];
48 const BoundaryCondition *bcB = bcs[m_idSideBBoundaryCondition];
49
50 if (bcA == nullptr && bcB == nullptr) {
51 m_errorMsg = "Boundary Condition A and B is not set.";
52 return false;
53 }
54
55 if (bcA != nullptr && !bcA->isValid(scheduleDB)) {
56 m_errorMsg = "Boundary Condition A is not valid.";
57 return false;
58 }
59
60 if (bcB != nullptr && !bcB->isValid(scheduleDB)) {
61 m_errorMsg = "Boundary Condition B is not valid.";
62 return false;
63 }
64
65 if( ( bcA != nullptr && bcB == nullptr && bcA->hasSetpointTemperatureForZone() ) ||
66 ( bcB != nullptr && bcA == nullptr && bcB->hasSetpointTemperatureForZone() ) ||
67 ( bcA != nullptr && bcB != nullptr &&
68 bcA->hasSetpointTemperatureForZone() && bcB->hasSetpointTemperatureForZone() ) ) {
69 m_errorMsg = "";
70 return false;
71 }
72
73 if (m_activeLayerIndex != VICUS::INVALID_ID) {
74 if (m_activeLayerIndex >= con->m_materialLayers.size()) {
75 m_errorMsg = "Actice layer index '" + std::to_string(m_activeLayerIndex) +
76 "' is bigger then number of layers '" + std::to_string(con->m_materialLayers.size()) + "'";
77 return false;
78 }
79 }
80 return true;
81
82 } catch (IBK::Exception & ex) {
83 ex.writeMsgStackToError();
84 return false;
85 }
86}

89AbstractDBElement::ComparisonResult Component::equal(const AbstractDBElement *other) const {
90 const Component * otherComp = dynamic_cast<const Component*>(other);
91 if (otherComp == nullptr)
92 return Different;
93
94 if ( m_idConstruction != otherComp->m_idConstruction ||
95 m_idSideABoundaryCondition != otherComp->m_idSideABoundaryCondition||
96 m_idSideBBoundaryCondition != otherComp->m_idSideBBoundaryCondition ||
97 m_idSurfaceProperty != otherComp->m_idSurfaceProperty ||
98 m_activeLayerIndex != otherComp->m_activeLayerIndex)
99 return Different;
100
101 if ( m_displayName != otherComp->m_displayName ||
102 m_dataSource != otherComp->m_dataSource ||
103 m_manufacturer != otherComp->m_manufacturer ||
104 m_type != otherComp->m_type ||
105 m_notes != otherComp->m_notes)
106 return OnlyMetaDataDiffers;
107
108 return Equal;
109}
