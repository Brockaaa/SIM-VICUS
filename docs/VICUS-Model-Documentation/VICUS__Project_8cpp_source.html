219void Project::parseHeader(const IBK::Path & filename) {
220 FUNCID(Project::parseHeader);
221
222 std::ifstream inputStream;
223 if (!IBK::open_ifstream(inputStream, filename, std::ios_base::binary)) {
224 throw IBK::Exception( IBK::FormatString("Cannot open input file '%1' for reading").arg(filename.c_str()), FUNC_ID);
225 }
226 std::string line;
227 unsigned int i=12;
228 while (std::getline(inputStream, line) && --i > 0) {
229 // abort when <Project is in the line but not <ProjectInfo
230 if (line.find("<Project") != std::string::npos && line.find("<ProjectInfo") == std::string::npos)
231 break;
232 size_t pos = line.find("<Created>");
233 if (pos != std::string::npos) {
234 size_t pos2 = line.find("</Created>");
235 if (pos2 != std::string::npos)
236 m_projectInfo.m_created = line.substr(pos + 9, pos2 - pos - 9);
237 }
238 pos = line.find("<LastEdited>");
239 if (pos != std::string::npos) {
240 size_t pos2 = line.find("</LastEdited>");
241 if (pos2 != std::string::npos)
242 m_projectInfo.m_lastEdited = line.substr(pos + 12, pos2 - pos - 12);
243 }
244 pos = line.find("<Comment>");
245 if (pos != std::string::npos) {
246 size_t pos2 = line.find("</Comment>");
247 if (pos2 != std::string::npos)
248 m_projectInfo.m_comment = line.substr(pos + 9, pos2 - pos - 9);
249 }
250 }

276void Project::readXMLDocument(TiXmlElement * rootElement) {
277 FUNCID(Project::readXML);
278
279 // we read our subsections from this handle
280 TiXmlHandle xmlRoot = TiXmlHandle(rootElement);
281
282 // clear existing grid planes
283 m_viewSettings.m_gridPlanes.clear();
284
285 try {
286 TiXmlElement * xmlElem = xmlRoot.FirstChild("ProjectInfo").Element();
287 if (xmlElem) {
288 m_projectInfo.readXML(xmlElem);
289 }
290
291 // Directory Placeholders
292 xmlElem = xmlRoot.FirstChild( "DirectoryPlaceholders" ).Element();
293 if (xmlElem) {
294 readDirectoryPlaceholdersXML(xmlElem);
295 }
296
297 xmlElem = xmlRoot.FirstChild("Project").Element();
298 if (xmlElem) {
299 readXML(xmlElem);
300 }
301
302 // if we do not have a default grid, create it
303 if (m_viewSettings.m_gridPlanes.empty()) {
304 m_viewSettings.m_gridPlanes.push_back( VICUS::GridPlane(IBKMK::Vector3D(0,0,0), IBKMK::Vector3D(0,0,1),
305 IBKMK::Vector3D(1,0,0), QColor("white"), 200, 10 ) );
306 }
307
308 // update internal pointer-based links and alongside check for data consistency
309 updatePointers();
310
311 // update the colors
312 // if project has invalid colors nothing is drawn ...
313 for (VICUS::Building &b : m_buildings) {
314 for (VICUS::BuildingLevel & bl : b.m_buildingLevels) {
315 for (VICUS::Room &r : bl.m_rooms) {
316 for (VICUS::Surface &s : r.m_surfaces) {
317 if (!s.m_displayColor.isValid())
318 s.initializeColorBasedOnInclination();
319 s.m_color = s.m_displayColor;
320 for (const VICUS::SubSurface &sub : s.subSurfaces()) {
321 const_cast<VICUS::SubSurface &>(sub).updateColor();
322 }
323 }
324 }
325 }
326 }
327
328 // plain geometry surfaces will be silver
329 for (VICUS::Surface &s : m_plainGeometry.m_surfaces) {
330 if (s.m_color == QColor::Invalid) {
331 s.m_color = QColor("#C0C0C0");
332 }
333 }
334
335 // set default colors for network objects
336 for (VICUS::Network & net : m_geometricNetworks) {
337 // updateColor is a const-function, this is possible since
338 // the m_color property of edges and nodes is mutable
339 net.setDefaultColors();
340 }
341 }
342 catch (IBK::Exception & ex) {
343 throw IBK::Exception(ex, IBK::FormatString("Error reading project from text."), FUNC_ID);
344 }

371void Project::readDirectoryPlaceholdersXML(const TiXmlElement * element) {
372
373 // loop over all elements in this XML element
374 for (const TiXmlElement * e=element->FirstChildElement(); e; e = e->NextSiblingElement()) {
375 // get element name
376 std::string name = e->Value();
377 // handle known elements
378 if (name == "Placeholder") {
379 // search for attribute with given name
380 const TiXmlAttribute* attrib = TiXmlAttribute::attributeByName(e, "name");
381 if (attrib == nullptr) {
382 IBK::IBK_Message(IBK::FormatString(
383 "Missing '%1' attribute in Placeholder element.").arg("name"), IBK::MSG_WARNING);
384 continue;
385 }
386 m_placeholders[attrib->Value()] = e->GetText();
387 }
388 else {
389 IBK::IBK_Message(IBK::FormatString(
390 "Unknown element '%1' in DirectoryPlaceholders section.").arg(name), IBK::MSG_WARNING);
391 }
392 }
393}

396void Project::writeDirectoryPlaceholdersXML(TiXmlElement * parent) const {
397 const char * const FUNC_ID = "[Project::writeDirectoryPlaceholdersXML]";
398
399 // lookup all used path placeholders
400 std::set<std::string> usedPlaceholders;
401
402 // first glob filenames used anywhere in the project
403 std::vector<IBK::Path> fnames;
404 fnames.push_back( m_location.m_climateFilePath );
405
406 // TODO : add more file references with placeholders
407
408 // now we extract all the placeholders
409 for (std::vector<IBK::Path>::const_iterator it = fnames.begin(); it != fnames.end(); ++it) {
410 std::string placeholder;
411 IBK::Path myPath(*it);
412 IBK::Path dummy;
413 if ( myPath.extractPlaceholder( placeholder, dummy ) )
414 usedPlaceholders.insert(placeholder);
415 }
416
417 // remove default placeholders for project directory
418 usedPlaceholders.erase(IBK::PLACEHOLDER_PROJECT_DIR);
419
420 // no placeholders used? nothing to write
421 if (usedPlaceholders.empty())
422 return;
423
424 TiXmlComment::addComment(parent,
425 "DirectoryPlaceholders section defines strings to be substituted with directories");
426
427 TiXmlElement * e1 = new TiXmlElement( "DirectoryPlaceholders" );
428 parent->LinkEndChild( e1 );
429
430 for (std::set<std::string>::const_iterator it = usedPlaceholders.begin();
431 it != usedPlaceholders.end(); ++it)
432 {
433
434 std::map<std::string, IBK::Path>::const_iterator pit = m_placeholders.find(*it);
435 // placeholder should exist, if not user may have tempered with the file and
436 // manually inserted a placeholder without adding it to the placeholder section
437 // This typically occurs when typos are present in the placeholder name.
438 // In such cases, we silently ignore writing the placeholder, but simply write
439 // a warning to the user.
440 if (pit == m_placeholders.end()) {
441 IBK::IBK_Message(IBK::FormatString("Placeholder '%1' is being used in one or more "
442 "referenced files, but not defined within the placeholder section")
443 .arg(*it), IBK::MSG_WARNING, FUNC_ID);
444 continue;
445 }
446 // write placeholder to file
447 TiXmlElement::appendSingleAttributeElement(e1, "Placeholder", "name", *it, pit->second.str());
448 }
449}

461void Project::updatePointers() {
462 FUNCID(Project::updatePointers);
463
464 // This is a pretty important function as it check and guarantees data consistency in the model.
465 // It is called whenever something changes in the data structure, and thus should be pretty fast.
466 // If any check fails, it throws an exception - this is useful when developing code as it triggers
467 // when data was modified inconsistenty.
468 //
469 // Checks are:
470 // - unique object IDs
471 // - valid surface references from component instances
472 // - backward references from surfaces to component instances are unique
473
474
475 m_objectPtr.clear();
476 m_objectPtr[VICUS::INVALID_ID] = nullptr; // this will help us detect invalid IDs in the data model
477
478 // update hierarchy
479 for (VICUS::Building & b : m_buildings)
480 b.updateParents();
481
482 for (VICUS::Building & b : m_buildings) {
483 addAndCheckForUniqueness(&b);
484 for (VICUS::BuildingLevel & bl : b.m_buildingLevels) {
485 addAndCheckForUniqueness(&bl);
486 for (VICUS::Room & r : bl.m_rooms) {
487 addAndCheckForUniqueness(&r);
488 for (VICUS::Surface & s : r.m_surfaces) {
489 addAndCheckForUniqueness(&s);
490 s.m_componentInstance = nullptr;
491 for (VICUS::SubSurface & sub : const_cast<std::vector<VICUS::SubSurface> &>(s.subSurfaces()) ) {
492 addAndCheckForUniqueness(&sub);
493 sub.m_subSurfaceComponentInstance = nullptr;
494 }
495 addChildSurface(s);
496 }
497 }
498 }
499 }
500
501 // Now our m_objectsPtr map contains ALL ids in the model that are referenced elsewhere
502
503
504 // *** check component instances ***
505
506 // this map holds all surfaces referenced by components (key = surfaceID, value = componentID of component that references the surface)
507 // and is used to check, if several components reference the same surface
508 std::map<unsigned int, unsigned int> referencedSurface;
509
510 // also check, that backward referenced from surfaces to component instances are unique
511 // Note: invalid "component" is ok, as this does not affect model consistency (it is just meta-data)
512 for (VICUS::ComponentInstance & ci : m_componentInstances) {
513
514 // component instance must have at least one valid side ID
515 if (ci.m_idSideASurface == VICUS::INVALID_ID && ci.m_idSideBSurface == VICUS::INVALID_ID)
516 throw IBK::Exception(IBK::FormatString("Component instance #%1 has two invalid surface references.").arg(ci.m_id), FUNC_ID);
517
518 // component instance may not reference the same surface on both sides
519 if (ci.m_idSideASurface == ci.m_idSideBSurface)
520 throw IBK::Exception(IBK::FormatString("Component instance #%1 references the same surface on both sides.").arg(ci.m_id), FUNC_ID);
521
522 // lookup surfaces
523 if (ci.m_idSideASurface != VICUS::INVALID_ID) {
524 ci.m_sideASurface = surfaceByID(ci.m_idSideASurface);
525 if (ci.m_sideASurface == nullptr)
526 throw IBK::Exception(IBK::FormatString("Component instance #%1 references an invalid/unknown surface #%2.").arg(ci.m_id).arg(ci.m_idSideASurface), FUNC_ID);
527
528 // check that this surface hasn't been referenced before
529 std::map<unsigned int, unsigned int>::const_iterator it = referencedSurface.find(ci.m_idSideASurface);
530 if (it != referencedSurface.end())
531 throw IBK::Exception(IBK::FormatString("Surface #%1 name: '%2' is referenced by component instance #%3, but was previously also referenced by component instance #%4!")
532 .arg(ci.m_idSideASurface).arg(ci.m_sideASurface->m_displayName.toStdString()).arg(ci.m_id).arg(it->second), FUNC_ID);
533 // remember referenced surface
534 referencedSurface[ci.m_idSideASurface] = ci.m_id;
535
536 // store pointer to this component in referenced side
537 ci.m_sideASurface->m_componentInstance = &ci;
538 }
539 else
540 ci.m_sideASurface = nullptr; // set nullptr if now invalid id
541
542 if (ci.m_idSideBSurface != VICUS::INVALID_ID) {
543 ci.m_sideBSurface = surfaceByID(ci.m_idSideBSurface);
544 if (ci.m_sideBSurface == nullptr)
545 throw IBK::Exception(IBK::FormatString("Component instance #%1 references an invalid/unknown surface #%2.").arg(ci.m_id).arg(ci.m_idSideBSurface), FUNC_ID);
546
547 // check that this surface hasn't been referenced before
548 std::map<unsigned int, unsigned int>::const_iterator it = referencedSurface.find(ci.m_idSideBSurface);
549 if (it != referencedSurface.end())
550 throw IBK::Exception(IBK::FormatString("Surface #%1 name: '%2' is referenced by component instance #%3, but was previously also referenced by component instance #%4!")
551 .arg(ci.m_idSideBSurface).arg(ci.m_sideBSurface->m_displayName.toStdString()).arg(ci.m_id).arg(it->second), FUNC_ID);
552 // remember referenced surface
553 referencedSurface[ci.m_idSideBSurface] = ci.m_id;
554
555 // store pointer to this component in referenced side
556 ci.m_sideBSurface->m_componentInstance = &ci;
557 }
558 else
559 ci.m_sideBSurface = nullptr; // set nullptr if now invalid id
560
561 // NOTE: surface heating control zone may be referenced by several component instances (forward reference)
562 if (ci.m_idSurfaceHeatingControlZone != VICUS::INVALID_ID) {
563 bool found = false;
564 for (VICUS::Building & b : m_buildings) {
565 for (VICUS::BuildingLevel & bl : b.m_buildingLevels) {
566 for (VICUS::Room & r : bl.m_rooms) {
567 if (r.m_id == ci.m_idSurfaceHeatingControlZone) {
568 ci.m_surfaceHeatingControlZone = &r;
569 found = true;
570 break;
571 }
572 if (found) break;
573 }
574 if (found) break;
575 }
576 if (found) break;
577 }
578 if (!found)
579 throw IBK::Exception(IBK::FormatString("Component instance #%1 references an invalid/unknown surface heating control zone #%2.")
580 .arg(ci.m_id).arg(ci.m_idSurfaceHeatingControlZone), FUNC_ID);
581 }
582 } // for component instances
583
584
585 // *** check subsurface component instances ***
586
587 // same as referencedSurface
588 std::map<unsigned int, unsigned int> referencedSubSurface;
589
590 // update pointers in subsurfaces
591 for (VICUS::SubSurfaceComponentInstance & ci : m_subSurfaceComponentInstances) {
592 // subsurface component instance must have at least one valid side ID
593 if (ci.m_idSideASurface == VICUS::INVALID_ID && ci.m_idSideBSurface == VICUS::INVALID_ID)
594 throw IBK::Exception(IBK::FormatString("Subsurface component instance #%1 has two invalid references.").arg(ci.m_id), FUNC_ID);
595
596 // subsurface component instance may not reference the same surface on both sides
597 if (ci.m_idSideASurface == ci.m_idSideBSurface)
598 throw IBK::Exception(IBK::FormatString("Subsurface component instance #%1 references the same subsurface on both sides.").arg(ci.m_id), FUNC_ID);
599
600 // lookup surfaces
601 if (ci.m_idSideASurface != VICUS::INVALID_ID) {
602 ci.m_sideASubSurface = subSurfaceByID(ci.m_idSideASurface);
603 if (ci.m_sideASubSurface == nullptr)
604 throw IBK::Exception(IBK::FormatString("Subsurface component instance #%1 references an invalid/unknown subsurface #%2.").arg(ci.m_id).arg(ci.m_idSideASurface), FUNC_ID);
605
606 // check that this surface hasn't been referenced before
607 std::map<unsigned int, unsigned int>::const_iterator it = referencedSubSurface.find(ci.m_idSideASurface);
608 if (it != referencedSubSurface.end())
609 throw IBK::Exception(IBK::FormatString("Subsurface #%1 name: '%2' is referenced by subcomponent instance #%3, but was previously also "
610 "referenced by component instance #%4!")
611 .arg(ci.m_idSideASurface).arg(ci.m_sideASubSurface->m_displayName.toStdString()).arg(ci.m_id).arg(it->second), FUNC_ID);
612 // remember referenced surface
613 referencedSubSurface[ci.m_idSideASurface] = ci.m_id;
614
615 // store pointer to this component in referenced side
616 ci.m_sideASubSurface->m_subSurfaceComponentInstance = &ci;
617 }
618
619 if (ci.m_idSideBSurface != VICUS::INVALID_ID) {
620 ci.m_sideBSubSurface = subSurfaceByID(ci.m_idSideBSurface);
621 if (ci.m_sideBSubSurface == nullptr)
622 throw IBK::Exception(IBK::FormatString("Subsurface component instance #%1 references an invalid/unknown subsurface #%2.").arg(ci.m_id).arg(ci.m_idSideBSurface), FUNC_ID);
623
624 // check that this surface hasn't been referenced before
625 std::map<unsigned int, unsigned int>::const_iterator it = referencedSubSurface.find(ci.m_idSideBSurface);
626 if (it != referencedSubSurface.end())
627 throw IBK::Exception(IBK::FormatString("Subsurface #%1 name: '%2' is referenced by subcomponent instance #%3, but was previously also "
628 "referenced by component instance #%4!")
629 .arg(ci.m_idSideBSurface).arg(ci.m_sideBSubSurface->m_displayName.toStdString()).arg(ci.m_id).arg(it->second), FUNC_ID);
630 // remember referenced surface
631 referencedSubSurface[ci.m_idSideBSurface] = ci.m_id;
632
633 // store pointer to this component in referenced side
634 ci.m_sideBSubSurface->m_subSurfaceComponentInstance = &ci;
635 }
636 }
637
638 // *** plain geometry ***
639
640 for (VICUS::Surface & s : m_plainGeometry.m_surfaces)
641 addAndCheckForUniqueness(&s);
642
643
644 // *** networks ***
645
646 for (VICUS::Network & n : m_geometricNetworks) {
647 addAndCheckForUniqueness(&n);
648 for (VICUS::NetworkNode & nod : n.m_nodes)
649 addAndCheckForUniqueness(&nod);
650
651 // create note-edge-pointer links
652 try {
653 n.updateNodeEdgeConnectionPointers();
654 } catch (IBK::Exception & ex) {
655 throw IBK::Exception(ex, IBK::FormatString("Invalid edge-to-node references in network #%1").arg(n.m_id), FUNC_ID);
656 }
657
658 // network edges
659
660 // NOTE: VICUS::NetworkEdge objects do not save their unique IDs in the project file.
661 // Hence, we need to assign unique IDs on the first time the object is created,
662 // or, when the object pointer list is first updated.
663 //
664 // CAUTION: This should always be the last step in this function, otherwise we may assign object ids here,
665 // which are used by other objects that have not been added yet.
666
667 for (VICUS::NetworkEdge & e : n.m_edges) {
668 if (e.m_id == VICUS::INVALID_ID)
669 e.m_id = nextUnusedID();
670 addAndCheckForUniqueness(&e);
671 }
672 }
673
674
675}

680 return 1;
681
682 // the last element in m_objectPtr is always INVALID_ID
683 auto it = m_objectPtr.rbegin();
684 Q_ASSERT(it->first == VICUS::INVALID_ID);

871IBKMK::Vector3D Project::boundingBox(std::vector<const Surface*> &surfaces,
872 std::vector<const SubSurface*> &subsurfaces,
873 IBKMK::Vector3D &center)
874{
875 // NOTE: We do not reuse the other boundingBox() function, because this implementation is much faster.
876
877 // store selected surfaces
878 if ( surfaces.empty() && subsurfaces.empty())
879 return IBKMK::Vector3D ( 0,0,0 );
880
881 IBKMK::Vector3D lowerValues(std::numeric_limits<double>::max(), std::numeric_limits<double>::max(), std::numeric_limits<double>::max());
882 IBKMK::Vector3D upperValues(std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest());
883
884 // process all surfaces
885 for (const VICUS::Surface *s : surfaces ) {
886 if (!s->geometry().polygon3D().isValid()) continue;
887 // TODO : also skip invisible surfaces?
888 s->geometry().polygon3D().enlargeBoundingBox(lowerValues, upperValues);
889 }
890
891 // process all sub-surfaces
892 for (const VICUS::SubSurface *sub : subsurfaces ) {
893 const VICUS::Surface *s = dynamic_cast<const VICUS::Surface *>(sub->m_parent);
894 // parent geometry must be valid and correctly triangulated
895 if (!s->geometry().isValid()) continue;
896 // now find our selected subsurfaec
897 for (unsigned int i=0; i<s->subSurfaces().size(); ++i) {
898 if (&(s->subSurfaces()[i]) != sub) continue;
899 // TODO : also skip invisible sub-surfaces?
900 for ( const IBKMK::Vector3D & v : s->geometry().holeTriangulationData()[i].m_vertexes ) {
901 upperValues.m_x = std::max(upperValues.m_x, v.m_x);
902 upperValues.m_y = std::max(upperValues.m_y, v.m_y);
903 upperValues.m_z = std::max(upperValues.m_z, v.m_z);
904
905 lowerValues.m_x = std::min(lowerValues.m_x, v.m_x);
906 lowerValues.m_y = std::min(lowerValues.m_y, v.m_y);
907 lowerValues.m_z = std::min(lowerValues.m_z, v.m_z);
908 }
909 }
910 }
911
912 // center point of bounding box
913 center = 0.5*(lowerValues+upperValues);
914 // difference between upper and lower values gives bounding box (dimensions of selected geometry)
915 return (upperValues-lowerValues);
916}

973 // TODO : Review this
974
975 double maxX = std::numeric_limits<double>::lowest();
976 double maxY = std::numeric_limits<double>::lowest();
977 double maxZ = std::numeric_limits<double>::lowest();
978 double minX = std::numeric_limits<double>::max();
979 double minY = std::numeric_limits<double>::max();
980 double minZ = std::numeric_limits<double>::max();
981 for (const VICUS::Surface *s : surfaces ) {
982 try {
983 s->polygon3D().vertexes();

988 }
989
990 for ( IBKMK::Vector3D v : s->polygon3D().vertexes() ) {
991
992 IBKMK::Vector3D vLocal, point;
993
994 IBKMK::lineToPointDistance(offset, xAxis, v, vLocal.m_x, point);
995 IBKMK::lineToPointDistance(offset, yAxis, v, vLocal.m_y, point);

1003

1005 ( v.m_y < minY ) ? minY = v.m_y : 0;
1006 ( v.m_z < minZ ) ? minZ = v.m_z : 0;

1010 const VICUS::Surface *s = dynamic_cast<const VICUS::Surface *>(sub->m_parent);
1011 if (!s->geometry().isValid()) continue;
1012 for (unsigned int i=0; i<s->subSurfaces().size(); ++i) {
1013 if (&(s->subSurfaces()[i]) == sub) {
1014 for ( IBKMK::Vector3D v : s->geometry().holeTriangulationData()[i].m_vertexes ) {
1015
1016 IBKMK::Vector3D vLocal, point;

1019 IBKMK::lineToPointDistance(offset, yAxis, v, vLocal.m_y, point);
1020 IBKMK::lineToPointDistance(offset, zAxis, v, vLocal.m_z, point);
1021

1023
1024 ( v.m_x > maxX ) ? maxX = v.m_x : 0;
1025 ( v.m_y > maxY ) ? maxY = v.m_y : 0;
1026 ( v.m_z > maxZ ) ? maxZ = v.m_z : 0;
1027
1028 ( v.m_x < minX ) ? minX = v.m_x : 0;
1029 ( v.m_y < minY ) ? minY = v.m_y : 0;
1030 ( v.m_z < minZ ) ? minZ = v.m_z : 0;
1031 }
1032 }
1033 }
1034 }
1035
1036 double dX = maxX - minX;
1037 double dY = maxY - minY;
1038 double dZ = maxZ - minZ;

1041 offset.m_y + (minX + 0.5*dX) * xAxis.m_y + (minY + 0.5*dY) * yAxis.m_y + (minZ + 0.5*dZ) * zAxis.m_y ,

1043
1044 // set bounding box;
1045 return IBKMK::Vector3D ( dX, dY, dZ );
1046}

1759// assume that setpoints hold 4 values lower and upper value for heating (value 1. and 2.) and cooling (value 3. and 4.)
1760bool calculateSupplyTemperature(const std::vector<double> &supplySetpoints,const std::vector<double> &outdoorSetpoints,
1761 const std::vector<double> &outdoorTemperatureSpline, std::vector<double> &supplyTemperature){
1762 if(supplySetpoints.size() != outdoorSetpoints.size() && supplySetpoints.size() == 4)
1763 return false;
1764
1765
1766 double lowerSupplyHeatLimit = std::min<double>(supplySetpoints[0], supplySetpoints[1]);
1767 double upperSupplyHeatLimit = std::max<double>(supplySetpoints[0], supplySetpoints[1]);
1768 double lowerSupplyCoolLimit = std::min<double>(supplySetpoints[2], supplySetpoints[3]);
1769 double upperSupplyCoolLimit = std::max<double>(supplySetpoints[2], supplySetpoints[3]);
1770
1771 double lowerOutHeatLimit = std::min<double>(outdoorSetpoints[0], outdoorSetpoints[1]);
1772 double upperOutHeatLimit = std::max<double>(outdoorSetpoints[0], outdoorSetpoints[1]);
1773 double lowerOutCoolLimit = std::min<double>(outdoorSetpoints[2], outdoorSetpoints[3]);
1774 double upperOutCoolLimit = std::max<double>(outdoorSetpoints[2], outdoorSetpoints[3]);
1775
1776 if(upperOutHeatLimit > lowerOutCoolLimit)
1777 return false;
1778
1779 supplyTemperature.clear();
1780
1781 double deltaX = outdoorSetpoints[0] - outdoorSetpoints[1];
1782 double mHeat = 0;
1783 if(deltaX != 0)
1784 mHeat = (supplySetpoints[0] - supplySetpoints[1]) / deltaX;
1785 double nHeat = mHeat * supplySetpoints[0] - outdoorSetpoints[0];
1786
1787 deltaX = outdoorSetpoints[2] - outdoorSetpoints[3];
1788 double mCool = 0;
1789 if(deltaX != 0)
1790 mCool = (supplySetpoints[2] - supplySetpoints[3]) / deltaX;
1791 double nCool = mCool * supplySetpoints[2] - outdoorSetpoints[3];
1792
1793 for(unsigned int i=0; i<outdoorTemperatureSpline.size(); ++i){
1794 double tOut = outdoorTemperatureSpline[i];
1795 double tSupply;
1796 if(tOut <= lowerOutHeatLimit)
1797 tSupply = lowerSupplyHeatLimit;
1798 else if(tOut > lowerOutHeatLimit && tOut <= upperOutHeatLimit)
1799 tSupply = mHeat * tOut + nHeat;
1800 else if(tOut > upperOutHeatLimit && tOut < lowerOutCoolLimit)
1801 tSupply = upperSupplyHeatLimit;
1802 else if(tOut >= lowerOutCoolLimit && tOut < upperOutCoolLimit)
1803 tSupply = mCool * tOut + nCool;
1804 else
1805 tSupply = upperSupplyCoolLimit;
1806 supplyTemperature.push_back(tSupply);
1807 }
1808 return true;
1809}
