40/*! Stores all data for a room. */
41class Room : public Object {
42public:
43 /*! Type-info string. */
44 const char * typeinfo() const override { return "Room"; }
45
46 /*! Room parameters.
47 These parameters are either automatically computed when zone is created/modified, or manually
48 entered/adjusted by user.
49 */
50 enum para_t {
51 /*! Floor usable area of the zone, used for zone load definition. */
52 P_Area, // Keyword: Area [m2] 'Floor usable area of the zone'
53 /*! Geometrical violume of the zone, used for air exchange calulation + energy balance. */
54 P_Volume, // Keyword: Volume [m3] 'Volume of the zone'
55 NUM_P
56 };
57
58 // *** PUBLIC MEMBER FUNCTIONS ***
59
60 VICUS_READWRITE
61 VICUS_COMPARE_WITH_ID
62
63 /*! Calculates the floor area of the room. */
64 void calculateFloorArea();
65
66 /*! Calculates the volume of the room. Room must have a closed shell. */
67 void calculateVolume();
68
69 void updateParents() {
70 m_children.clear();
71 for (Surface & s : m_surfaces) {
72 m_children.push_back(&s);
73 s.m_parent = this;
74 s.updateParents();
75 }
76 }
77
78
79 // *** PUBLIC MEMBER VARIABLES ***
80
81 //:inherited unsigned int m_id = INVALID_ID; // XML:A:required
82 //:inherited QString m_displayName; // XML:A
83 //:inherited bool m_visible = true; // XML:A
84 //:inherited std::string m_ifcGUID; // XML:A
85
86 /*! Reference to assigned zone template (optional). */
87 IDType m_idZoneTemplate = INVALID_ID; // XML:E
88
89 /*! Reference to assigned acoustic template (optional). */
90 IDType m_idAcousticTemplate = INVALID_ID; // XML:E
91
92
93 /*! Zone parameters. */
94 IBK::Parameter m_para[NUM_P]; // XML:E
95
96 /*! Surfaces of the room. */
97 std::vector<Surface> m_surfaces; // XML:E
98
99 // *** RUNTIME VARIABLES ***
100
101 /*! Whole room net floor area in m2.
102 This area is updated either from user-defined area in m_para or automatically computed from
103 zone geometry. If the latter is not possible, user must enter a floor area.
104 */
105 double m_netFloorArea = -1;
106
107 /*! Whole room volume in m3.
108 This volume is updated either from user-defined volume in m_para or automatically computed from
109 zone geometry. If the latter is not possible, user must enter a valid volume.
110 */
111 double m_volume = -1;
112};
