34bool ZoneIdealHeatingCooling::isValid() const
35{
36 if(m_id == INVALID_ID)
37 return false;
38
39 try {
40 if(!m_para[P_HeatingLimit].empty())
41 m_para[P_HeatingLimit].checkedValue(VICUS::KeywordList::Keyword("ZoneIdealHeatingCooling::para_t", P_HeatingLimit),
42 "W/m2", "W/m2", 0, true, 1000, true, nullptr);
43 if(!m_para[P_CoolingLimit].empty())
44 m_para[P_CoolingLimit].checkedValue(VICUS::KeywordList::Keyword("ZoneIdealHeatingCooling::para_t", P_CoolingLimit),
45 "W/m2", "W/m2", 0, true, 1000, true, nullptr);
46
47 } catch (...) {
48 return false;
49 }
50
51 return true;
52}
