39/*! A network edge defines all parameters needed to generate pipe hydraulic network elements.
40 The pipe-specific hydraulic parameters are stored in the pipe database (NetworkPipe).
41 The type of pipe model is defined in m_modelType and the kind of heat exchange to be considered
42 is defined in m_heatExchangeType.
43 Parameters for generation of the hydraulic element are stored in m_para. Also, the double-parameters needed
44 for the heat exchange models are stored in m_para.
45
46 NOTE: NetworkEdge also uses the unique m_id to identify object and respective navigation tree items and scene
47 objects. However, the ID is nowhere else used (referenced) in the data model and thus not persistently
48 saved in the project file. It is generated on first use within VICUS::Project::updatePointers().
49
50 TODO Hauke, Dokumentation der Memberfunktionen!
51*/
52class NetworkEdge : public Object {
53public:
54 /*! Type-info string. */
55 const char * typeinfo() const override { return "NetworkEdge"; }
56
57 // *** PUBLIC MEMBER FUNCTIONS ***
58
59 VICUS_READWRITE
60
61 /*! Deafult constructor.
62 */
63 NetworkEdge() = default;
64
65 /*! Initializing constructor.
66 \param id id of this edge
67 \param nodeId1 id of node 1
68 \param nodeId2 id of node 2
69 \param supply specifies wether this is a 'supply' pipe, meaning that SubStation nodes can connect to this edge
70 \param length length of edge
71 \param pipeId id of pipe properties in database
72 */
73 NetworkEdge(const unsigned id, const unsigned nodeId1, const unsigned nodeId2, const bool supply, const double &length, const unsigned pipeId):
74 m_supply(supply),
75 m_idPipe(pipeId),
76 m_idNode1(nodeId1),
77 m_idNode2(nodeId2),
78 m_length(length)
79 {
80 m_id = id;
81 }
82
83 /*! Check connectivity of graph trhough recursive search */
84 void collectConnectedNodes(std::set<const NetworkNode*> & connectedNodes,
85 std::set<const NetworkEdge*> & connectedEdge) const;
86
87 /*! Used in recursive search, starting from one node (should be the soruce node), and trying to collect all edges in an ordered vector.
88 * Thereby for each edge, one node is set as inlet and the other node is set as outlet. This should anticipate the expected flow direction. */
89 void setInletOutletNode(std::set<const NetworkNode*> & visitedNodes, std::vector<const NetworkEdge*> & orderedEdges);
90
91 /*! Returns opposite node of the given one */
92 NetworkNode * neighbourNode(const NetworkNode *node) const;
93
94 /*! Returns opposite node id of the given one */
95 unsigned neighbourNode(unsigned nodeId) const;
96
97 /*! Returns the edge length */
98 double length() const { return m_length; }
99
100 /*! Re-set length, calculating it from coordinates of connected nodes */
101 void setLengthFromCoordinates();
102
103 unsigned int nodeId1() const { return m_idNode1; }
104
105 unsigned int nodeId2() const {return m_idNode2; }
106
107 /*! Swaps current nodeId1 with given node id, sets pointer and calculates the new length of this edge */
108 void changeNode1(NetworkNode *node);
109
110 /*! Swaps current nodeId2 with given node id, sets pointer and calculates the new length of this edge */
111 void changeNode2(NetworkNode *node);
112
113
114 // *** PUBLIC MEMBER VARIABLES ***
115
116 /*! If true, nodes of type SubStation can connect to this edge.
117 This is used for the automatic algorithm that connects buildings with the network */
118 bool m_supply; // XML:A
119
120 /*! ID of pipe in database */
121 IDType m_idPipe = INVALID_ID; // XML:A
122
123 //:inherited QString m_displayName; // XML:A
124
125 /*! Whether the node is visible or not. */
126 //:inherited bool m_visible = true; // XML:A
127
128 /*! Defines the heat exchange properties for this edge (ambient temperature, heat flux etc.) */
129 NANDRAD::HydraulicNetworkHeatExchange m_heatExchange; // XML:E
130
131
132 // *** RUNTIME VARIABLES ***
133
134 /*! The radius [m] used for the visualization of this edge in the 3D scene
135 Updated whenever the scale factor Network::m_scaleEdges changes, or the pipe ID.
136 */
137 double m_visualizationRadius;
138
139 /*! Color to be used for displaying (visible) nodes. */
140 mutable QColor m_color;
141
142 /*! Sum of maximum heating demand of all connected buildings [W], will be determined in sizePipeDimensions() */
143 double m_nominalHeatingDemand = 0;
144
145 /*! Mass flow [kg/s] at nominal temperature difference, will be determined in sizePipeDimensions() */
146 double m_nominalMassFlow = 0;
147
148 /*! Describes the fluid temperature change along this pipe with nominal mass flow, when there is a temperature difference of 1 K
149 between fluid and sourrunding soil. This information can be used to create according soil models.
150 Is dimensionless, but for interpretation, unit [K/K] can be used
151 */
152 double m_tempChangeIndicator = -1;
153 /*! The cumulative temperature change along a flow path */
154 double m_cumulativeTempChangeIndicator = -1;
155
156 /*! Pointers to connected nodes */
157 NetworkNode * m_node1 = nullptr;
158 NetworkNode * m_node2 = nullptr;
159
160 /*! Specifiy which node is the 'inlet' and which is the 'outlet' node. This anticipates the nominal flow direction during simulation. */
161 unsigned int m_idNodeInlet = INVALID_ID;
162 unsigned int m_idNodeOutlet = INVALID_ID;
163
164 /*! Used to map the edge to the according Nandrad supply / return pipe.*/
165 unsigned int m_idNandradSupplyPipe = INVALID_ID;
166 unsigned int m_idNandradReturnPipe = INVALID_ID;
167
168 /*! Used to map the edge to a soil model.*/
169 unsigned int m_idSoil = INVALID_ID;
170
171
172private:
173
174 // *** PRIVATE MEMBER VARIABLES ***
175
176 unsigned int m_idNode1 = 0; // XML:A:required
177 unsigned int m_idNode2 = 0; // XML:A:required
178
179 /*! Effective length [m], might be different from geometric length between nodes. */
180 double m_length = 0; // XML:E
181};
