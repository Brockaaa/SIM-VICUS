30bool EPDDataset::behavesLike(const EPDDataset & other) const
31{
32
33 if(m_referenceUnit != other.m_referenceUnit ||
34 m_referenceQuantity != other.m_referenceQuantity)
35 return false;
36
37 if(m_category != other.m_category)
38 return false;
39
40 for (unsigned int i=0; i<NUM_P; ++i) {
41 para_t t = static_cast<para_t>(i);
42 if(m_para[t].empty() && other.m_para[t].empty())
43 continue;
44 if(m_para[t] != other.m_para[t])
45 return false;
46 }
47
48 return true;
49
50}

52AbstractDBElement::ComparisonResult EPDDataset::equal(const AbstractDBElement *other) const {
53 const EPDDataset * otherEPD = dynamic_cast<const EPDDataset*>(other);
54 if (otherEPD == nullptr)
55 return Different;
56
57 //first check critical data
58
59 //check parameters
60 for(unsigned int i=0; i<NUM_P; ++i){
61 if(m_para[i] != otherEPD->m_para[i])
62 return Different;
63 }
64 if(m_uuid != otherEPD->m_uuid ||
65 m_referenceUnit != otherEPD->m_referenceUnit||
66 m_referenceQuantity != otherEPD->m_referenceQuantity ||
67 m_subtype != otherEPD->m_subtype ||
68 m_category != otherEPD->m_category ||
69 m_expireDate != otherEPD->m_expireDate)
70 return Different;
71
72 //check meta data
73
74 if(m_displayName != otherEPD->m_displayName ||
75 m_color != otherEPD->m_color ||
76 m_dataSource != otherEPD->m_dataSource ||
77 m_manufacturer != otherEPD->m_manufacturer ||
78 m_notes != otherEPD->m_notes)
79 return OnlyMetaDataDiffers;
80
81 return Equal;
82}
