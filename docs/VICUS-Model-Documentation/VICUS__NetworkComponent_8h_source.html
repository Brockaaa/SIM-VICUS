45/*! Data model object for network components, basically the same as NANDRAD::HydraulicNetworkComponent with
46 data members needed in the user interface.
47
48 We cannot use inheritence here, but since both objects are very similar it should be easy to copy/paste back
49 and forth between objects.
50*/
51class NetworkComponent : public AbstractDBElement {
52public:
53
54 NetworkComponent() = default;

56 /*! The various types (equations) of the hydraulic component. */
57 enum ModelType {
58 MT_SimplePipe, // Keyword: SimplePipe 'Pipe with a single fluid volume'
59 MT_DynamicPipe, // Keyword: DynamicPipe 'Pipe with discretized fluid volume'
60 MT_ConstantPressurePump, // Keyword: ConstantPressurePump 'Pump with defined pressure head'
61 MT_ConstantMassFluxPump, // Keyword: ConstantMassFluxPump 'Pump with defined mass flux'
62 MT_ControlledPump, // Keyword: ControlledPump 'Pump with controlled pressure head'
63 MT_VariablePressurePump, // Keyword: VariablePressurePump 'Pump with linear pressure head curve'
64 MT_HeatExchanger, // Keyword: HeatExchanger 'Simple heat exchanger'
65 MT_HeatPumpVariableIdealCarnotSourceSide, // Keyword: HeatPumpVariableIdealCarnotSourceSide 'Heat pump installed at source side, based on carnot efficiency'
66 MT_HeatPumpVariableIdealCarnotSupplySide, // Keyword: HeatPumpVariableIdealCarnotSupplySide 'Heat pump installed at supply side, based on carnot efficiency'
67 MT_HeatPumpVariableSourceSide, // Keyword: HeatPumpVariableSourceSide 'Heat pump installed at source side, based on polynom for COP'
68 MT_HeatPumpOnOffSourceSide, // Keyword: HeatPumpOnOffSourceSide 'Heat pump (on-off) at source side, based on polynoms for heating power and el. power'
69// xMT_HeatPumpOnOffSourceSideWithBuffer, // xKeyword: HeatPumpOnOffSourceSideWithBuffer 'On-off-type heat pump based on polynoms for heating power and el. power, installed at source side'
70 MT_ControlledValve, // Keyword: ControlledValve 'Controlled valve'
71 MT_ConstantPressureLossValve, // Keyword: ConstantPressureLossValve 'Valve with constant pressure loss'
72 MT_IdealHeaterCooler, // Keyword: IdealHeaterCooler 'Ideal heater with predefined supply temperature'
73 MT_PressureLossElement, // Keyword: PressureLossElement 'Adiabatic element with pressure loss defined by zeta-value'
74 NUM_MT
75 };

77 /*! Parameters for the component. */
78 enum para_t {
79 // from NANDRAD::HydraulicNetworkComponent
80 P_HydraulicDiameter, // Keyword: HydraulicDiameter [mm] 'Only used for pressure loss calculation with PressureLossCoefficient (NOT for pipes)'
81 P_PressureLossCoefficient, // Keyword: PressureLossCoefficient [---] 'Pressure loss coefficient for the component (zeta-value)'
82 P_PressureHead, // Keyword: PressureHead [Bar] 'Pump predefined pressure head'
83 P_MassFlux, // Keyword: MassFlux [kg/s] 'Pump predefined mass flux'
84 P_PumpMaximumEfficiency, // Keyword: PumpMaximumEfficiency [---] 'Pump efficiency in optimal operation point'
85 P_FractionOfMotorInefficienciesToFluidStream, // Keyword: FractionOfMotorInefficienciesToFluidStream [---] 'Fraction of pump heat loss due to inefficiency that heats up the fluid'
86 P_MaximumPressureHead, // Keyword: MaximumPressureHead [Bar] 'Pump maximum pressure head at point of minimal mass flow of pump'
87 P_PumpMaximumElectricalPower, // Keyword: PumpMaximumElectricalPower [W] 'Pump maximum electrical power at point of optimal operation'
88 P_DesignPressureHead, // Keyword: DesignPressureHead [Bar] 'Design pressure head of VariablePressureHeadPump'
89 P_DesignMassFlux, // Keyword: DesignMassFlux [kg/s] 'Design mass flux of VariablePressureHeadPump'
90 P_PressureHeadReductionFactor, // Keyword: PressureHeadReductionFactor [---] 'Factor to reduce pressure head of VariablePressureHeadPump'
91 P_Volume, // Keyword: Volume [m3] 'Water or air volume of the component'
92 P_PipeMaxDiscretizationWidth, // Keyword: PipeMaxDiscretizationWidth [m] 'Maximum width/length of discretized volumes in pipe'
93 P_CarnotEfficiency, // Keyword: CarnotEfficiency [---] 'Carnot efficiency eta'
94 P_MaximumHeatingPower, // Keyword: MaximumHeatingPower [W] 'Maximum heating power'
95 P_PressureLoss, // Keyword: PressureLoss [Bar] 'Pressure loss for valve'
96 P_MinimumOutletTemperature, // Keyword: MinimumOutletTemperature [C] 'Minimum outlet temperature of heat exchanger, used for clipping of heat extraction'
97 P_HeatingPowerB0W35, // Keyword: HeatingPowerB0W35 [W] 'Heat pump heating power at B0/W35
98 P_HeatingBufferSupplyTemperature, // Keyword: HeatingBufferSupplyTemperature [C] 'Heat pump supply temperature for heating buffer storage'
99 P_HeatingBufferReturnTemperature, // Keyword: HeatingBufferReturnTemperature [C] 'Heat pump return temperature for heating buffer storage'
100 P_DHWBufferSupplyTemperature, // Keyword: DHWBufferSupplyTemperature [C] 'Heat pump supply temperature for DHW buffer storage'
101 P_DHWBufferReturnTemperature, // Keyword: DHWBufferReturnTemperature [C] 'Heat pump return temperature for DHW buffer storage'
102 P_HeatingBufferVolume, // Keyword: HeatingBufferVolume [m3] 'Heat pump heating buffer storage volume'
103 P_DHWBufferVolume, // Keyword: DHWBufferVolume [m3] 'Heat pump DHW buffer storage volume' '
104 // additional parameters
105 P_PipeLength, // Keyword: PipeLength [m] 'Length of pipe'
106 NUM_P
107 };
