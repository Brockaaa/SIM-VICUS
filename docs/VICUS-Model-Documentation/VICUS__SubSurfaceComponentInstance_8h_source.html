36/*! Defines an embedded object (a subsurface).
37 It is basically just a connection data member and does not hold any
38 physical parameters itself.
39*/
40class SubSurfaceComponentInstance {
41public:
42 /*! Default c'tor. */
43 SubSurfaceComponentInstance() {}
44 /*! Initializing constructor. */
45 SubSurfaceComponentInstance(unsigned int id,
46 unsigned int componentID, unsigned int sideASurfaceID, unsigned int sideBSurfaceID)
47 : m_id(id), m_idSubSurfaceComponent(componentID), m_idSideASurface(sideASurfaceID), m_idSideBSurface(sideBSurfaceID)
48 {}
49
50 // *** PUBLIC MEMBER FUNCTIONS ***
51
52 VICUS_READWRITE
53 VICUS_COMPARE_WITH_ID
54
55 // *** PUBLIC MEMBER VARIABLES ***
56
57 /*! ID of component instance (will be used for NANDRAD::ConstructionInstance). */
58 unsigned int m_id = INVALID_ID; // XML:A:required
59 /*! ID of referenced component. */
60 unsigned int m_idSubSurfaceComponent = INVALID_ID; // XML:A
61 /*! ID of surface at side A (optional, ID = 0 reserved for "ambient", INVALID_ID means adiabatic). */
62 unsigned int m_idSideASurface = INVALID_ID; // XML:A
63 /*! ID of surface at side B (optional, ID = 0 reserved for "ambient", INVALID_ID means adiabatic). */
64 unsigned int m_idSideBSurface = INVALID_ID; // XML:A
65
66
67 /*! ID of shading system definition, to be used when shading system is active. */
68 unsigned int m_idShadingSystem = INVALID_ID; // XML:E
69 /*! ID of shading control model. */
70 unsigned int m_idShadingControlModel = INVALID_ID; // XML:E
71
72
73 // *** RUNTIME VARIABLES ***
74
75 // These pointers are updated in VICUS::Project::updatePointers() and can be used
76 // to quicky travers the data model.
77
78 VICUS::SubSurface *m_sideASubSurface = nullptr;
79 VICUS::SubSurface *m_sideBSubSurface = nullptr;
80};
