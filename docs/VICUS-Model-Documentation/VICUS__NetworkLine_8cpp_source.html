43double NetworkLine::distanceToPoint(const IBKMK::Vector3D &point) const{
44 double lineFactor = 0;
45 IBKMK::Vector3D projectionPoint;
46 double distance = IBKMK::lineToPointDistance(m_a, m_b, point, lineFactor, projectionPoint);
47 // if point is outside line, we calculate the distance to the closest endpoint
48 if (lineFactor > 1 || lineFactor < 0) {
49 double distanceA = (m_a - point).magnitude();
50 double distanceB = ((m_a + m_b) - point).magnitude();
51 distance = std::min(distanceA, distanceB);
52 }
53
54 return distance;
55}

58bool NetworkLine::containsPoint(const IBKMK::Vector3D &point) const {
59 double lineFactor = 0;
60 IBKMK::Vector3D projectionPoint;
61 double dist = IBKMK::lineToPointDistance(m_a, m_b, point, lineFactor, projectionPoint);
62 return dist < GeometricResolution && IBK::near_gr(lineFactor, 0) && IBK::near_le(lineFactor, 1);
63}
