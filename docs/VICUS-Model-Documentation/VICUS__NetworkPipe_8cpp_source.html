34double NetworkPipe::UValue() const {
35 // This function is only to be called when isValid() returns true. Hence, all parameters are present with
36 // valid values.
37
38 // some references for readability improvement
39 // all in SI units here (length unit "m")
40 const double dInsulation = m_para[VICUS::NetworkPipe::P_ThicknessInsulation].value;
41 const double lambdaInsulation = m_para[VICUS::NetworkPipe::P_ThermalConductivityInsulation].value;
42 const double da = m_para[VICUS::NetworkPipe::P_DiameterOutside].value;
43 const double lambdaWall = m_para[VICUS::NetworkPipe::P_ThermalConductivityWall].value;
44
45 double di = diameterInside();
46 double r;
47 if (dInsulation > 0)
48 r = 1./(2*lambdaWall) * IBK::f_log(da / di)
49 + 1./(2*lambdaInsulation) * IBK::f_log((da + 2*dInsulation) / da);
50 else
51 r = 1./(2*lambdaWall) * IBK::f_log(da / di);
52
53 double UValue = PI/r;
54 return UValue;
55}

63bool NetworkPipe::isValid() const {
64 if (m_id == INVALID_ID)
65 return false;
66
67 for (int i=0; i<NUM_P; ++i){
68 bool zeroAllowed = (para_t(i) == P_ThicknessInsulation);
69 try {
70 m_para[i].checkedValue(KeywordList::Keyword("NetworkPipe::para_t", i), KeywordList::Unit("NetworkPipe::para_t", i),
71 KeywordList::Unit("NetworkPipe::para_t", i), 0, zeroAllowed, std::numeric_limits<double>::max(),
72 false, nullptr);
73 } catch (...) {
74 return false;
75 }
76 }
77
78 // check if inside diameter is greater zero
79 if (diameterInside() <= 0)
80 return false;
81
82 return true;
83}

86VICUS::AbstractDBElement::ComparisonResult NetworkPipe::equal(const VICUS::AbstractDBElement *other) const {
87 const NetworkPipe * otherNetPipe = dynamic_cast<const NetworkPipe*>(other);
88 if (otherNetPipe == nullptr)
89 return Different;
90
91 // check parameters
92 for (unsigned int i=0; i<NUM_P; ++i){
93 if (m_para[i] != otherNetPipe->m_para[i])
94 return Different;
95 }
96
97 // check meta data
98 if (m_displayName != otherNetPipe->m_displayName ||
99 m_color != otherNetPipe->m_color ||
100 m_categoryName != otherNetPipe->m_categoryName)
101
102 return OnlyMetaDataDiffers;
103
104 return Equal;
105}
